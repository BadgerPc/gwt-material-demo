020package gwt.material.design.incubator.client.language;
021
022import com.google.gwt.core.client.GWT;
023import com.google.gwt.dom.client.Document;
024import com.google.gwt.event.logical.shared.ValueChangeEvent;
025import com.google.gwt.event.logical.shared.ValueChangeHandler;
026import com.google.gwt.event.shared.HandlerRegistration;
027import com.google.gwt.user.client.Window;
028import com.google.gwt.user.client.ui.HasValue;
029import gwt.material.design.client.MaterialDesignBase;
030import gwt.material.design.client.base.MaterialWidget;
031import gwt.material.design.client.ui.MaterialDropDown;
032import gwt.material.design.client.ui.MaterialImage;
033import gwt.material.design.incubator.client.AddinsIncubator;
034import gwt.material.design.incubator.client.base.IncubatorWidget;
035import gwt.material.design.incubator.client.base.constants.IncubatorCssName;
036
037import java.util.ArrayList;
038import java.util.List;
039
040/**
041 * A widget that provides a selection of {@link LanguageSelectorItem} for the site Internationalization (i18n)
042 *
043 * <p><i>
044 *     Note: This component is under the incubation process and subject to change.
045 * </i></p>
046 *
047 * @author kevzlou7979
048 */
049public class LanguageSelector extends MaterialWidget implements HasValue<Language> {
050
051    static {
052        if (AddinsIncubator.isDebug()) {
053            MaterialDesignBase.injectCss(LanguageSelectorDebugClientBundle.INSTANCE.languageSelectorDebugCss());
054        } else {
055            MaterialDesignBase.injectCss(LanguageSelectorClientBundle.INSTANCE.languageSelectorCss());
056        }
057    }
058
059    private MaterialImage image = new MaterialImage();
060    private MaterialDropDown dropdown = new MaterialDropDown();
061    private List<Language> _languages = new ArrayList<>();
062    private Language _language;
063
064    public LanguageSelector() {
065        super(Document.get().createDivElement(), IncubatorCssName.LANGUAGE_SELECTOR);
066        image.addStyleName(IncubatorCssName.LANGUAGE_ACTIVATOR);
067    }
068
069    @Override
070    protected void onLoad() {
071        super.onLoad();
072
073        IncubatorWidget.showWarning(this);
074        dropdown.clear();
075        _languages.forEach(language -> dropdown.add(new LanguageSelectorItem(language)));
076        dropdown.setConstrainWidth(false);
077
078        registerHandler(dropdown.addSelectionHandler(selectionEvent -> {
079            if (selectionEvent.getSelectedItem() instanceof LanguageSelectorItem) {
080                Language language = ((LanguageSelectorItem) selectionEvent.getSelectedItem()).getLanguage();
081
082                if (_language != language) {
083                    setValue(language, true);
084
085                    // Navigate to selected language value / locale and reload the browser
086                    String param = Window.Location.createUrlBuilder().setParameter("locale", language.getValue()).buildString();
087                    Window.Location.replace(param);
088                }
089            }
090        }));
091
092        image.add(dropdown);
093        add(image);
094
095        // Get the current locale inside the browser url
096        String currentLocale = Window.Location.getParameter("locale");
097        if (_languages.size() != 0) {
098            Language currentLanguage = _languages.stream().filter(language -> language.getValue().equals(currentLocale)).findFirst().orElse(_languages.get(0));
099            setValue(currentLanguage);
100        } else {
101            GWT.log("Please add at least one language for this selector.");
102        }
103    }
104
105    /**
106     * Ability to add language item {@link Language}
107     * contains name, value and image url
108     */
109    public void addLanguage(Language language) {
110        _languages.add(language);
111    }
112
113    /**
114     * Set lists of languages and repopulate inside the Language Selector Dropdown component
115     */
116    public void setLanguages(List<Language> languages) {
117        _languages = languages;
118    }
119
120    /**
121     * Get the lists of registered languages of this selector component
122     */
123    public List<Language> getLanguages() {
124        return _languages;
125    }
126
127    @Override
128    public Language getValue() {
129        return _language;
130    }
131
132    @Override
133    public void setValue(Language language) {
134        setValue(language, false);
135    }
136
137    @Override
138    public void setValue(Language language, boolean fireEvents) {
139        _language = language;
140        image.setUrl(language.getImage());
141
142        if (fireEvents) {
143            ValueChangeEvent.fire(this, language);
144        }
145    }
146
147    @Override
148    public HandlerRegistration addValueChangeHandler(ValueChangeHandler<Language> valueChangeHandler) {
149        return addHandler(valueChangeHandler, ValueChangeEvent.getType());
150    }
151}
