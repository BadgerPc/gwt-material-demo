022import com.google.gwt.dom.client.Style;
023import com.google.gwt.event.shared.HandlerRegistration;
024import com.google.gwt.user.client.ui.Widget;
025import gwt.material.design.addins.client.carousel.MaterialCarousel;
026import gwt.material.design.addins.client.carousel.MaterialCarouselFixedItem;
027import gwt.material.design.addins.client.carousel.constants.CarouselType;
028import gwt.material.design.client.constants.Color;
029import gwt.material.design.client.constants.Type;
030import gwt.material.design.client.constants.WavesType;
031import gwt.material.design.client.ui.MaterialButton;
032import gwt.material.design.incubator.client.base.IncubatorWidget;
033import gwt.material.design.incubator.client.walkthrough.events.CompleteEvent;

035//@formatter:off
036
037/**
038 * An extension to Onboard Type of {@link MaterialCarousel} component
039 * which provides additional features and handler registrations.
040 *
041 * <p><i>
042 *     Note: This component is under the incubation process and subject to change.
043 * </i></p>
044 *
045 * @author kevzlou7979
046 */
047//@formatter:on
048public class Walkthrough extends MaterialCarousel {
049
050    private MaterialCarouselFixedItem item = new MaterialCarouselFixedItem();
051    private MaterialButton submit = new MaterialButton("Get Started");
052
053    @Override
054    protected void onLoad() {
055        super.onLoad();
056
057        IncubatorWidget.showWarning(this);
058        item.add(submit);
059        submit.setFontSize("0.8em");
060        submit.setBackgroundColor(Color.WHITE);
061        submit.setTextColor(Color.GREY);
062        submit.setWaves(WavesType.DEFAULT);
063        add(item);
064
065        setType(CarouselType.ONBOARD);
066        setLayoutPosition(Style.Position.FIXED);
067        setEdgeFriction(0);
068
069        registerHandler(addAfterChangeHandler(event -> animate(event.getCurrentSlide())));
070        registerHandler(addInitHandler(event -> animate(0)));
071        registerHandler(submit.addClickHandler(clickEvent -> CompleteEvent.fire(this)));
072    }
073
074    protected void animate(int index) {
075        Widget w = getContainer().getWidget(index);
076        if (w instanceof WalkthroughItem) {
077            ((WalkthroughItem) w).animate();
078        }
079    }
080
081    public void register(String username, String appName) {
082        WalkthroughManager.register(username, appName);
083    }
084
085    public boolean hasWalkthrough(String username, String appName) {
086        return WalkthroughManager.hasWalkthrough(username, appName);
087    }
088
089    public HandlerRegistration addCompleteHandler(CompleteEvent.CompleteHandler handler) {
090        return addHandler(handler, CompleteEvent.TYPE);
091    }
092}
