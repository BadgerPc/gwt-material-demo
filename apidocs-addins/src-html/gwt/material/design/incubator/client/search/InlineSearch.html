020package gwt.material.design.incubator.client.search;
021
022import com.google.gwt.event.shared.HandlerRegistration;
023import gwt.material.design.addins.client.MaterialAddins;
024import gwt.material.design.addins.client.base.constants.AddinsCssName;
025import gwt.material.design.client.MaterialDesignBase;
026import gwt.material.design.client.base.mixin.CssNameMixin;
027import gwt.material.design.client.base.viewport.Resolution;
028import gwt.material.design.client.base.viewport.ViewPort;
029import gwt.material.design.client.ui.MaterialSearch;
030import gwt.material.design.incubator.client.base.IncubatorWidget;
031import gwt.material.design.incubator.client.search.constants.Theme;
032
033/**
034 * Extension to {@link MaterialSearch} which provides an inline
035 * style appended to {@link gwt.material.design.client.ui.MaterialNavBar} component.
036 *
037 * <p><i>
038 *     Note: This component is under the incubation process and subject to change.
039 * </i></p>
040 *
041 * @author kevzlou7979
042 */
043public class InlineSearch extends MaterialSearch {
044
045    private HandlerRegistration focusHandler;
046    private HandlerRegistration blurHandler;
047    private CssNameMixin<InlineSearch, Theme> cssNameMixin;
048
049    static {
050        if (MaterialAddins.isDebug()) {
051            MaterialDesignBase.injectCss(InlineSearchDebugClientBundle.INSTANCE.inlineSearchDebugCss());
052        } else {
053            MaterialDesignBase.injectCss(InlineSearchClientBundle.INSTANCE.inlineSearchCss());
054        }
055    }
056
057    public InlineSearch() {
058        super();
059        addStyleName(AddinsCssName.INLINE_SEARCH);
060    }
061
062    @Override
063    protected void onLoad() {
064        super.onLoad();
065
066        IncubatorWidget.showWarning(this);
067        ViewPort.when(Resolution.ALL_MOBILE).then(portChange -> {
068            focusHandler = registerHandler(addFocusHandler(focusEvent -> addStyleName(AddinsCssName.WIDE)));
069            blurHandler = registerHandler(addBlurHandler(blurEvent -> removeStyleName(AddinsCssName.WIDE)));
070        }, viewPortRect -> {
071            if (focusHandler != null && blurHandler != null) {
072                focusHandler.removeHandler();
073                blurHandler.removeHandler();
074            }
075
076            focusHandler = null;
077            blurHandler = null;
078            return true;
079        });
080    }
081
082    public void setTheme(Theme theme) {
083        getCssNameMixin().setCssName(theme);
084    }
085
086    protected CssNameMixin<InlineSearch, Theme> getCssNameMixin() {
087        if (cssNameMixin == null) {
088            cssNameMixin = new CssNameMixin<>(this);
089        }
090        return cssNameMixin;
091    }
092}
