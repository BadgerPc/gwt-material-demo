020package gwt.material.design.incubator.client.async;
021
022import gwt.material.design.client.MaterialDesignBase;
023import gwt.material.design.client.base.HasActive;
024import gwt.material.design.client.base.mixin.ActiveMixin;
025import gwt.material.design.client.constants.Color;
026import gwt.material.design.client.constants.IconType;
027import gwt.material.design.client.constants.SpinnerColor;
028import gwt.material.design.client.ui.MaterialButton;
029import gwt.material.design.client.ui.MaterialSpinner;
030import gwt.material.design.incubator.client.AddinsIncubator;
031import gwt.material.design.incubator.client.base.IncubatorWidget;
032import gwt.material.design.incubator.client.base.constants.IncubatorCssName;
033
034//@formatter:off
035
036/**
037 * Asynchronous buttons are components that is good ux for
038 * calling the server side component and automatically updates
039 * the states of the ui logic.
040 *
041 * <p><i>
042 *     Note: This component is under the incubation process and subject to change.
043 * </i></p>
044 * <h3>XML Namespace Declaration</h3>
045 *
046 * @author kevzlou7979
047 */
048public class AsyncButton extends MaterialButton implements HasActive {
049
050    static {
051        if (AddinsIncubator.isDebug()) {
052            MaterialDesignBase.injectCss(AsyncDebugClientBundle.INSTANCE.asyncButtonDebugCss());
053        } else {
054            MaterialDesignBase.injectCss(AsyncClientBundle.INSTANCE.asyncButtonCss());
055        }
056    }
057
058    private ActiveMixin<AsyncButton> activeMixin;
059
060    private MaterialSpinner loader = new MaterialSpinner();
061
062    public AsyncButton() {
063        super();
064        addStyleName(IncubatorCssName.ASYNC_BUTTON);
065    }
066
067    @Override
068    protected void onLoad() {
069        super.onLoad();
070
071        IncubatorWidget.showWarning(this);
072        loader.setColor(SpinnerColor.BLUE_ONLY);
073        add(loader);
074    }
075
076    public void loading() {
077        setActive(true);
078    }
079
080    public void loading(String message) {
081        loading();
082        setBackgroundColor(Color.WHITE);
083        setTextColor(Color.BLACK);
084        setText(message);
085    }
086
087    public void error(String message) {
088        setActive(false);
089        setBackgroundColor(Color.WHITE);
090        setTextColor(Color.RED);
091        setIconType(IconType.WARNING);
092        setText(message);
093    }
094
095    public void success(String message) {
096        setActive(false);
097        setBackgroundColor(Color.WHITE);
098        setTextColor(Color.BLACK);
099        setIconType(IconType.CHECK);
100        setText(message);
101    }
102
103    @Override
104    public void setActive(boolean value) {
105        getActiveMixin().setActive(value);
106    }
107
108    @Override
109    public boolean isActive() {
110        return getActiveMixin().isActive();
111    }
112
113    public ActiveMixin<AsyncButton> getActiveMixin() {
114        if (activeMixin == null) {
115            activeMixin = new ActiveMixin<>(this);
116        }
117        return activeMixin;
118    }
119}
