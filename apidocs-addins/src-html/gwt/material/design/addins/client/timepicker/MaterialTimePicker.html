256    /**
257     * Called after the lolliclock event <code>afterShow</code>.
258     */
259    protected void beforeShow() {
260        
261        this.input.getElement().blur();
262
263        // Add class 'valid' for visual feedback.
264        this.validMixin.setOn(true);
265    }
266    
267    /**
268     * Called after the lolliclock event <code>afterShow</code>.
269     */
270    protected void afterShow() {
271        this.fireOpenEvent();
272    }
273        
274    /**
275     * Called after the lolliclock event <code>afterHide</code>.
276     * 
277     * @param time
278     *            The time given by lolliclock in 12-hours <code>hh:mm aa</code>
279     *            format.
280     */
281    protected void afterHide() {
282        
283        String timeString = this.getTime(this.input.getElement());
284        
285        Date parsedDate = null;
286        
287        if(timeString.equals("") == false && timeString != null) {
288            try {
289                if(this.hour24 == true) {
290                    DateTimeFormat hour24DateTimeFormat = DateTimeFormat.getFormat("HH:mm");
291                    parsedDate = hour24DateTimeFormat.parse(timeString);
292                } else {
293                    DateTimeFormat hour12DateTimeFormat = DateTimeFormat.getFormat("hh:mm aa");
294                    parsedDate = hour12DateTimeFormat.parse(timeString);
295                }
296            } catch(Exception e) {
297                // Silently catch parse errors
298            }
299        }
300        
301        this.setValue(parsedDate);
302        
303        // Remove class 'valid' after hide.
304        this.validMixin.setOn(false);
305        
306        this.fireCloseEvent();
307    }

364    @Override
365    public void setValue(Date time, boolean fireEvents) {
366        
367        if(this.time != null) {
368            if(this.time.equals(time)) {
369                return;
370            }
371        }
372        
373        if(this.time == time) {
374            return;
375        }
376        
377        this.time = time;
378        
379        String timeString = null;
380        
381        if(this.hour24 == true) {
382            DateTimeFormat hour24DateTimeFormat = DateTimeFormat.getFormat("HH:mm");
383            timeString = hour24DateTimeFormat.format(time);
384        } else {
385            DateTimeFormat hour12DateTimeFormat = DateTimeFormat.getFormat("hh:mm aa");
386            timeString = hour12DateTimeFormat.format(time);
387        }
388        
389        this.setValue(this.input.getElement(), timeString);
390             
391        if(fireEvents == true) {
392            this.fireValueChangeEvent();
393        }
394    }
