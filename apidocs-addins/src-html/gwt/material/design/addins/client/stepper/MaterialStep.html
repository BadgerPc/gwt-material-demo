023import com.google.gwt.dom.client.Document;
024import com.google.gwt.event.dom.client.ClickEvent;
025import com.google.gwt.event.dom.client.ClickHandler;
026import com.google.gwt.event.logical.shared.HasSelectionHandlers;
027import com.google.gwt.event.logical.shared.SelectionEvent;
028import com.google.gwt.event.logical.shared.SelectionHandler;
029import com.google.gwt.event.shared.HandlerRegistration;
030import com.google.gwt.user.client.ui.Widget;
031import gwt.material.design.addins.client.stepper.base.mixin.ActiveMixin;
032import gwt.material.design.client.base.*;
033import gwt.material.design.client.constants.Axis;
034import gwt.material.design.client.constants.IconType;
035import gwt.material.design.client.ui.MaterialIcon;
036import gwt.material.design.client.ui.html.Div;
037
038//@formatter:off
039/**
040 * Material Step is a child element of Material Stepper, sometimes called Stepper Item, used to indicate the active
041 * inactive items on the Stepper Component.
042 *
043 * <h3>XML Namespace Declaration</h3>
044 * <pre>
045 * {@code
046 * xmlns:ma='urn:import:gwt.material.design.addins.client'
047 * }
048 * </pre>
049 *
050 * <h3>UiBinder Usage:</h3>
051 * <pre>
052 * {@code
053 *   <ma:stepper.MaterialStep step="1" title="Name of Step 1">
054 *       <m:MaterialPanel width="100%" height="300px" backgroundColor="grey lighten-2"/>
055 *       <m:MaterialButton ui:field="btnContinue1" text="Continue to Step 2" grid="l4" marginTop="12" backgroundColor="blue" textColor="white" waves="DEFAULT"/>
056 *       <m:MaterialButton ui:field="btnPrev1" text="Cancel" grid="l4" marginTop="12" type="FLAT" waves="DEFAULT"/>
057 *   </ma:stepper.MaterialStep>
058 * }
059 * </pre>
060 *
061 * @author kevzlou7979
062 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#steppers">Material Steppers</a>
063 */
064// @formatter:on
065public class MaterialStep extends MaterialWidget implements HasActive, HasTitle, HasError, HasAxis, HasSelectionHandlers<MaterialStep> {
066
067    private int step;
068    private String title;
069    private String description;
070
071    // containers
072    private Div conCircle = new Div();
073    private Div conBody = new Div();
074
075    // elements
076    private Div divCircle = new Div();
077    private Div divLine = new Div();
078    private Div divTitle = new Div();
079    private Div divDescription = new Div();
080    private Div divBody = new Div();
081
082    private MaterialIcon iconError = new MaterialIcon(IconType.REPORT_PROBLEM);
083    private MaterialIcon iconSuccess = new MaterialIcon(IconType.CHECK_CIRCLE);
084    private final ActiveMixin<MaterialStep> activeMixin = new ActiveMixin<>(this);
085    
086    private Axis axis = Axis.VERTICAL;
087
088    public MaterialStep() {
089        super(Document.get().createDivElement());
090        setStyleName("step");
091
092        super.add(conCircle);
093        conCircle.add(divCircle);
094        conCircle.add(divLine);
095
096        super.add(conBody);
097        conBody.add(divTitle);
098        conBody.add(divBody);
099
100        divCircle.setStyleName("circle");
101        divLine.setStyleName("line");
102        divTitle.setStyleName("title");
103        divBody.setStyleName("body");
104        
105        ClickHandler handler = new ClickHandler() {
106            @Override
107            public void onClick(ClickEvent event) {
108                if (isEnabled() && isVisible()){
109                    SelectionEvent.fire(MaterialStep.this, MaterialStep.this);                    
110                }
111            }
112        };
113        conCircle.addClickHandler(handler);
114        divTitle.addClickHandler(handler);
115        divDescription.addClickHandler(handler);
116    }
117
118    @Override
119    public void add(Widget child) {
120        divBody.add(child);
121    }
122
123    public int getStep() {
124        return step;
125    }
126
127    public void setStep(int step) {
128        this.step = step;
129        divCircle.getElement().setInnerHTML(String.valueOf(step));
130    }
131
132    @Override
133    public void setTitle(String title) {
134        this.title = title;
135        divTitle.getElement().setInnerHTML(title);
136    }
137
138    public String getTitle() {
139        return title;
140    }
141
142    @Override
143    public void setDescription(String description) {
144        this.description = description;
145        divDescription.setStyleName("description");
146        divDescription.getElement().setInnerHTML(description);
147        conBody.insert(divDescription, 1);
148    }
149
150    public String getDescription() {
151        return description;
152    }
153
154    @Override
155    public void setActive(boolean active) {
156        activeMixin.setActive(active);
157    }
158
159    @Override
160    public boolean isActive() {
161        return activeMixin.isActive();
162    }
163
164    @Override
165    public void setError(String error) {
166        removeStyleName("success");
167        addStyleName("error");
168        applyIconStatus(iconError, "red", error);
169    }
170
171    @Override
172    public void setSuccess(String success) {
173        removeStyleName("error");
174        addStyleName("success");
175        applyIconStatus(iconSuccess, "blue", success);
176    }
177
178    @Override
179    public void clearErrorOrSuccess() {
180        iconError.removeFromParent();
181        iconSuccess.removeFromParent();
182        conCircle.insert(divCircle, 0);
183        removeStyleName("error");
184        removeStyleName("success");
185    }
186
187
188    private void applyIconStatus(MaterialIcon icon, String color, String description){
189        iconError.removeFromParent();
190        iconSuccess.removeFromParent();
191        divCircle.removeFromParent();
192        conCircle.insert(icon, 0);
193        divDescription.getElement().setInnerHTML(description);
194    }
195
196    public Div getDivBody() {
197        return divBody;
198    }
199    
200    @Override
201    public void setAxis(Axis axis) {
202        if (axis == null){
203            axis = Axis.VERTICAL;
204        }
205        this.axis = axis;
206        switch (axis){
207        case HORIZONTAL:
208            conCircle.add(divTitle);
209            conCircle.add(divLine);
210            conCircle.add(divDescription);
211            break;
212        case VERTICAL:
213            conBody.insert(divTitle, 0);
214            conCircle.add(divLine);
215            break;
216        }
217    }
218    
219    @Override
220    public Axis getAxis() {
221        return axis;
222    }
223
224    @Override
225    public HandlerRegistration addSelectionHandler(SelectionHandler<MaterialStep> handler) {
226        return this.addHandler(handler, SelectionEvent.getType());
227    }
228}
