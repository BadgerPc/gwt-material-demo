020package gwt.material.design.addins.client.pathanimator;
021
022import com.google.gwt.dom.client.Element;
023import com.google.gwt.dom.client.Style;
024import com.google.gwt.user.client.ui.Widget;
025import gwt.material.design.addins.client.MaterialAddins;
026import gwt.material.design.addins.client.pathanimator.js.JsPathAnimator;
027import gwt.material.design.client.MaterialDesignBase;
028import gwt.material.design.jquery.client.api.Functions;
029
030import static gwt.material.design.jquery.client.api.JQuery.$;
031
032//@formatter:off
033
034/**
035 * Custom component that provides meaningfull transition between two elements to show visual continuity.
036 * <p>
037 * <pre>
038 * {@code
039 *
040 * // CAN BE CALLED AS A HELPER STATIC CONTEXT
041 * MaterialPathAnimator.animate(Element source, Element target, Functions.Func animateCallback);
042 *
043 * // INSTANTIATE THE PUSHPIN COMPONENT
044 * MaterialPathAnimator animator = new MaterialPathAnimator();
045 * animator.setSourceElement(btnSource1.getElement());
046 * animator.setTargetElement(panelTarget1.getElement());
047 * animator.animate();
048 * // Reverse Animate
049 * animator.reverseAnimate();
050 * }
051 * </pre>
052 *
053 * @author kevzlou7979
054 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#pathAnimator">Material Path Animator</a>
055 */
056//@formatter:on
057public class MaterialPathAnimator {
058
059    static {
060        if (MaterialAddins.isDebug()) {
061            MaterialDesignBase.injectJs(MaterialPathAnimatorDebugClientBundle.INSTANCE.pathanimatorDebugJs());
062        } else {
063            MaterialDesignBase.injectJs(MaterialPathAnimatorClientBundle.INSTANCE.pathanimatorJs());
064        }
065    }
066
067    private Element sourceElement;
068    private Element targetElement;
069    private Functions.Func animateCallback;
070    private Functions.Func reverseCallback;
071
072    public MaterialPathAnimator() {
073    }
074
075    /**
076     * Animate the path animator
077     */
078    public void animate() {
079        $("document").ready(() -> {
080            JsPathAnimator.cta(sourceElement, targetElement, () -> {
081                if (animateCallback != null) {
082                    animateCallback.call();
083                } else {
084                    // For default animateCallback when animateCallback is null
085                    targetElement.getStyle().setVisibility(Style.Visibility.VISIBLE);
086                    targetElement.getStyle().setOpacity(1);
087                }
088            });
089        });
090    }
091
092    /**
093     * Helper method to apply the path animator.
094     *
095     * @param source Source element to apply the Path Animator
096     * @param target Target element to apply the Path Animator
097     */
098    public static void animate(Element source, final Element target) {
099        animate(source, target, null);
100    }
101
102    /**
103     * Helper method to apply the path animator.
104     *
105     * @param source Source widget to apply the Path Animator
106     * @param target Target widget to apply the Path Animator
107     */
108    public static void animate(Widget source, final Widget target) {
109        animate(source.getElement(), target.getElement());
110    }
111
112    /**
113     * Helper method to apply the path animator with callback.
114     *
115     * @param source   Source widget to apply the Path Animator
116     * @param target   Target widget to apply the Path Animator
117     * @param callback The callback method to be called when the path animator is applied
118     */
119    public static void animate(Widget source, Widget target, Functions.Func callback) {
120        animate(source.getElement(), target.getElement(), callback);
121    }
122
123    /**
124     * Helper method to apply the path animator with callback.
125     *
126     * @param sourceElement   Source widget to apply the Path Animator
127     * @param targetElement   Target widget to apply the Path Animator
128     * @param animateCallback The callback method to be called when the path animator is applied
129     */
130    public static void animate(Element sourceElement, Element targetElement, Functions.Func animateCallback) {
131        MaterialPathAnimator animator = new MaterialPathAnimator();
132        animator.setSourceElement(sourceElement);
133        animator.setTargetElement(targetElement);
134        animator.setAnimateCallback(animateCallback);
135        animator.animate();
136    }
137
138    /**
139     * Reverse the Animation
140     */
141    public void reverseAnimate() {
142        $("document").ready(() -> {
143            if (reverseCallback != null) {
144                reverseCallback.call();
145            } else {
146                targetElement.getStyle().setVisibility(Style.Visibility.HIDDEN);
147                targetElement.getStyle().setOpacity(0);
148            }
149            JsPathAnimator.cta(targetElement, sourceElement);
150        });
151    }
152
153    /**
154     * Helper method to reverse animate the source element to target element.
155     *
156     * @param sourceElement Source element to apply the Path Animator
157     * @param targetElement Target element to apply the Path Animator
158     */
159    public static void reverseAnimate(final Element sourceElement, final Element targetElement) {
160        reverseAnimate(sourceElement, targetElement, null);
161    }
162
163    /**
164     * Helper method to reverse animate the source element to target element.
165     *
166     * @param source Source widget to apply the Path Animator
167     * @param target Target widget to apply the Path Animator
168     */
169    public static void reverseAnimate(final Widget source, final Widget target) {
170        reverseAnimate(source.getElement(), target.getElement());
171    }
172
173    /**
174     * Helper method to reverse animate the source element to target element with reverse callback.
175     *
176     * @param source          Source widget to apply the Path Animator
177     * @param target          Target widget to apply the Path Animator
178     * @param reverseCallback The reverse callback method to be called when the path animator is applied
179     */
180    public static void reverseAnimate(Widget source, Widget target, Functions.Func reverseCallback) {
181        reverseAnimate(source.getElement(), target.getElement(), reverseCallback);
182    }
183
184    /**
185     * Helper method to reverse animate the source element to target element with reverse callback
186     *
187     * @param sourceElement   Source element to apply the Path Animator
188     * @param targetElement   Target element to apply the Path Animator
189     * @param reverseCallback The reverse callback method to be called when the path animator is applied
190     */
191    public static void reverseAnimate(Element sourceElement, Element targetElement, Functions.Func reverseCallback) {
192        MaterialPathAnimator animator = new MaterialPathAnimator();
193        animator.setSourceElement(sourceElement);
194        animator.setTargetElement(targetElement);
195        animator.setReverseCallback(reverseCallback);
196        animator.reverseAnimate();
197    }
198
199    /**
200     * Get the source element
201     */
202    public Element getSourceElement() {
203        return sourceElement;
204    }
205
206    /**
207     * Set the source element
208     */
209    public void setSourceElement(Element sourceElement) {
210        this.sourceElement = sourceElement;
211    }
212
213    /**
214     * Get the target element
215     */
216    public Element getTargetElement() {
217        return targetElement;
218    }
219
220    /**
221     * Set the target element
222     */
223    public void setTargetElement(Element targetElement) {
224        this.targetElement = targetElement;
225    }
226
227    /**
228     * Get the callback method when the path animator is applied
229     */
230    public Functions.Func getAnimateCallback() {
231        return animateCallback;
232    }
233
234    /**
235     * Set the callback method when the path animator is applied
236     */
237    public void setAnimateCallback(Functions.Func animateCallback) {
238        this.animateCallback = animateCallback;
239    }
240
241    /**
242     * Get the reverse callback method when the path animator is applied
243     */
244    public Functions.Func getReverseCallback() {
245        return reverseCallback;
246    }
247
248    /**
249     * Set the reverse callback method when the path animator is applied
250     */
251    public void setReverseCallback(Functions.Func reverseCallback) {
252        this.reverseCallback = reverseCallback;
253    }
254}
