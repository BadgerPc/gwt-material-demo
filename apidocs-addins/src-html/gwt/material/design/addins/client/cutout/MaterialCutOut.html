046/**
047 * MaterialCutOut is a fullscreen modal-like component to show users about new
048 * features or important elements of the document.
049 * <p>
050 * You can use {@link CloseHandler}s to be notified when the cut out is closed.
051 * <p>
052 * <h3>XML Namespace Declaration</h3>
053 * <pre>
054 * {@code
055 * xmlns:ma='urn:import:gwt.material.design.addins.client'
056 * }
057 * </pre>
058 * <p>
059 * <h3>UiBinder Usage:</h3>
060 * <p>
061 * <pre>
062 * {@code
063 * <ma:cutout.MaterialCutOut ui:field="cutOut">
064 *      <!-- add any widgets here -->
065 * </ma:cutout.MaterialCutOut>
066 * }
067 * </pre>
068 * <p>
069 * <h3>Java Usage:</h3>
070 * {@code
071 * MaterialCutOut cutOut = ... //create using new or using UiBinder
072 * cutOut.setTarget(myTargetWidget); //the widget or element you want to focus
073 * cutOut.open(); //shows the modal over the page
074 * }
075 * <p>
076 * <h3>Custom styling:</h3> You use change the cut out style by using the
077 * <code>material-cutout</code> class, and <code>material-cutout-focus</code>
078 * class for the focusElement box.
079 * <p>
080 * <h3>Notice:</h3>On some iOS devices, on mobile Safari, the CutOut may not open when the
081 * {@link #setCircle(boolean)} is set to <code>true</code>. This is because of problems on Safari
082 * with box-shadows over rounded borders. To avoid this issue you can disable the circle. Check the
083 * <a href="https://github.com/GwtMaterialDesign/gwt-material/issues/227">issue 227</a> for details.
084 *
085 * @author gilberto-torrezan
086 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#cutouts">Material Cutouts</a>
087 */
088// @formatter:on

124    @Override

126        focusElement = Document.get().createDivElement();
127        getElement().appendChild(focusElement);
128
129        Style style = getElement().getStyle();
130        style.setWidth(100, Unit.PCT);
131        style.setHeight(100, Unit.PCT);
132        style.setPosition(Position.FIXED);
133        style.setTop(0, Unit.PX);
134        style.setLeft(0, Unit.PX);
135        style.setOverflow(Overflow.HIDDEN);
136        style.setZIndex(10000);
137        style.setDisplay(Display.NONE);
138
139        focusElement.setClassName(AddinsCssName.MATERIAL_CUTOUT_FOCUS);
140        style = focusElement.getStyle();
141        style.setProperty("content", "\'\'");
142        style.setPosition(Position.ABSOLUTE);
143        style.setZIndex(-1);
144    }

279    /**
280     * Opens the modal cut out taking all the screen. The target element should
281     * be set before calling this method.
282     *
283     * @throws IllegalStateException if the target element is <code>null</code>
284     * @see #setTarget(Widget)
285     */
286    public void open() {
287        if (targetElement == null) {
288            throw new IllegalStateException("The target element should be set before calling open().");
289        }
290        targetElement.scrollIntoView();
291
292        if (computedBackgroundColor == null) {
293            setupComputedBackgroundColor();
294        }
295
296        //temporarily disables scrolling by setting the overflow of the page to hidden
297        Style docStyle = Document.get().getDocumentElement().getStyle();
298        viewportOverflow = docStyle.getOverflow();
299        docStyle.setProperty("overflow", "hidden");
300
301        if(backgroundSize == null) {
302            backgroundSize = body().width() + 300 + "px";
303        }
304
305        setupTransition();
306        if (animated) {
307            focusElement.getStyle().setProperty("boxShadow", "0px 0px 0px 0rem " + computedBackgroundColor);
308
309            //the animation will take place after the boxshadow is set by the deferred command
310            Scheduler.get().scheduleDeferred(() -> {
311                focusElement.getStyle().setProperty("boxShadow", "0px 0px 0px " + backgroundSize + " " + computedBackgroundColor);
312            });
313        } else {
314            focusElement.getStyle().setProperty("boxShadow", "0px 0px 0px " + backgroundSize + " " + computedBackgroundColor);
315        }
316
317        if (circle) {
318            focusElement.getStyle().setProperty("WebkitBorderRadius", "50%");
319            focusElement.getStyle().setProperty("borderRadius", "50%");
320        } else {
321            focusElement.getStyle().clearProperty("WebkitBorderRadius");
322            focusElement.getStyle().clearProperty("borderRadius");
323        }
324        setupCutOutPosition(focusElement, targetElement, cutOutPadding, circle);
325
326        setupWindowHandlers();
327        getElement().getStyle().clearDisplay();
328
329        // verify if the component is added to the document (via UiBinder for
330        // instance)
331        if (getParent() == null) {
332            autoAddedToDocument = true;
333            RootPanel.get().add(this);
334        }
335        OpenEvent.fire(this, this);
336    }

346    /**
347     * Closes the cut out.
348     *
349     * @param autoClosed Notifies with the modal was auto closed or closed by user action
350     */
351    public void close(boolean autoClosed) {
352        //restore the old overflow of the page
353        Document.get().getDocumentElement().getStyle().setProperty("overflow", viewportOverflow);
354
355        getElement().getStyle().setDisplay(Display.NONE);
356
357        //remove old handlers to avoid memory leaks
358        if (resizeHandler != null) {
359            resizeHandler.removeHandler();
360            resizeHandler = null;
361        }
362        if (scrollHandler != null) {
363            scrollHandler.removeHandler();
364            scrollHandler = null;
365        }
366
367        // if the component added himself to the document, it must remove
368        // himself too
369        if (autoAddedToDocument) {
370            this.removeFromParent();
371            autoAddedToDocument = false;
372        }
373        CloseEvent.fire(this, this, autoClosed);
374    }

376    /**
377     * Setups the cut out position when the screen changes size or is scrolled.
378     */
379    protected void setupCutOutPosition(Element cutOut, Element relativeTo, int padding, boolean circle) {
380        float top = relativeTo.getAbsoluteTop() - body().scrollTop();
381        float left = relativeTo.getAbsoluteLeft();
382
383        float width = relativeTo.getOffsetWidth();
384        float height = relativeTo.getOffsetHeight();
385
386        if (circle) {
387            if (width != height) {
388                float dif = width - height;
389                if (width > height) {
390                    height += dif;
391                    top -= dif / 2;
392                } else {
393                    dif = -dif;
394                    width += dif;
395                    left -= dif / 2;
396                }
397            }
398        }
399
400        top -= padding;
401        left -= padding;
402        width += padding * 2;
403        height += padding * 2;
404
405        $(cutOut).css("top", top + "px");
406        $(cutOut).css("left", left + "px");
407        $(cutOut).css("width", width + "px");
408        $(cutOut).css("height", height + "px");
409    }
410
411    /**
412     * Configures a resize handler and a scroll handler on the window to
413     * properly adjust the Cut Out.
414     */
415    protected void setupWindowHandlers() {
416        if (resizeHandler != null) {
417            resizeHandler.removeHandler();
418        }
419        if (scrollHandler != null) {
420            scrollHandler.removeHandler();
421        }
422        resizeHandler = Window.addResizeHandler(event -> {
423            setupCutOutPosition(focusElement, targetElement, cutOutPadding, circle);
424        });
425        scrollHandler = Window.addWindowScrollHandler(event -> {
426            setupCutOutPosition(focusElement, targetElement, cutOutPadding, circle);
427        });
428    }
429
430    protected void setupTransition() {
431        if (animated) {
432            focusElement.getStyle().setProperty("WebkitTransition", "box-shadow " + getDuration() + "ms " + animationTimingFunction);
433            focusElement.getStyle().setProperty("transition", "box-shadow " + getDuration() + "ms " + animationTimingFunction);
434        } else {
435            focusElement.getStyle().clearProperty("WebkitTransition");
436            focusElement.getStyle().clearProperty("transition");
437        }
438    }

440    /**
441     * Gets the computed background color, based on the backgroundColor CSS
442     * class.
443     */
444    protected void setupComputedBackgroundColor() {
445        // temp is just a widget created to evaluate the computed background
446        // color
447        MaterialWidget temp = new MaterialWidget(Document.get().createDivElement());
448        temp.setBackgroundColor(backgroundColor);
449
450        // setting a style to make it invisible for the user
451        Style style = temp.getElement().getStyle();
452        style.setPosition(Position.FIXED);
453        style.setWidth(1, Unit.PX);
454        style.setHeight(1, Unit.PX);
455        style.setLeft(-10, Unit.PX);
456        style.setTop(-10, Unit.PX);
457        style.setZIndex(-10000);
458
459        // adding it to the body (on Chrome the component must be added to the
460        // DOM before getting computed values).
461        String computed = ColorHelper.setupComputedBackgroundColor(backgroundColor);
462
463        // convert rgb to rgba, considering the opacity field
464        if (opacity < 1 && computed.startsWith("rgb(")) {
465            computed = computed.replace("rgb(", "rgba(").replace(")", ", " + opacity + ")");
466        }
467
468        computedBackgroundColor = computed;
469    }
