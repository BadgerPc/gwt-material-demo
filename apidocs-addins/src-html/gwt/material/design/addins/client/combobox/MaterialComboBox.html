121    @Override
122    protected void onLoad() {
123        super.onLoad();
124
125        if (!initialized) {
126            label.setInitialClasses(AddinsCssName.SELECT2LABEL);
127            super.add(listbox);
128            super.add(label);
129            lblError.setLayoutPosition(Style.Position.ABSOLUTE);
130            lblError.setMarginTop(12);
131            super.add(lblError);
132            setId(uid);
133
134            listbox.setGwtDisplay(Style.Display.BLOCK);
135            initialized = true;
136        }
137
138        initialize();
139    }

141    public void initialize() {
142        JsComboBoxOptions options = new JsComboBoxOptions();
143        options.allowClear = allowClear;
144        options.placeholder = placeholder;
145        options.maximumSelectionLength = limit;
146        if (isHideSearch()) {
147            options.minimumResultsForSearch = "Infinity";
148        }
149
150        JsComboBox jsComboBox = $(listbox.getElement());
151        jsComboBox.select2(options);
152
153        jsComboBox.on(ComboBoxEvents.CHANGE, event -> {
154            ValueChangeEvent.fire(MaterialComboBox.this, getValue());
155            return true;
156        });
157
158        jsComboBox.on(ComboBoxEvents.SELECT, event -> {
159            SelectionEvent.fire(MaterialComboBox.this, getValue());
160            return true;
161        });
162
163        jsComboBox.on(ComboBoxEvents.UNSELECT, event -> {
164            RemoveItemEvent.fire(this, getValue());
165            return true;
166        });
167
168        jsComboBox.on(ComboBoxEvents.OPEN, (event1, o) -> {
169            OpenEvent.fire(MaterialComboBox.this, getValue());
170            return true;
171        });
172
173        jsComboBox.on(ComboBoxEvents.CLOSE, (event1, o) -> {
174            CloseEvent.fire(MaterialComboBox.this, getValue());
175            return true;
176        });
177    }

179    @Override
180    protected void onUnload() {
181        super.onUnload();
182
183        // Perform tear down on select2
184        JsComboBox jsComboBox = $(listbox.getElement());
185        jsComboBox.off(ComboBoxEvents.CHANGE);
186        jsComboBox.off(ComboBoxEvents.SELECT);
187        jsComboBox.off(ComboBoxEvents.UNSELECT);
188        jsComboBox.off(ComboBoxEvents.OPEN);
189        jsComboBox.off(ComboBoxEvents.CLOSE);
190        jsComboBox.select2("destroy");
191
192        if (valueChangeHandler != null) {
193            valueChangeHandler.removeHandler();
194        }
195    }

199    public void add(Widget child) {
200        if (child instanceof OptGroup) {
201            for (Widget w : ((OptGroup) child).getChildren()) {
202                if (w instanceof Option) {
203                    values.add((T) ((Option) w).getValue());
204                }
205            }
206        } else if (child instanceof Option) {
207            values.add((T) ((Option) child).getValue());
208        }
209        listbox.add(child);
210    }

329    @Override
330    public void setValue(T value, boolean fireEvents) {
331        int index = values.indexOf(value);
332        if (index > 0) {
333            T before = getValue();
334            setSelectedIndex(index);
335
336            if (fireEvents) {
337                ValueChangeEvent.fireIfNotEqual(this, before, value);
338            }
339        }
340    }

342    /**
343     * Set directly all the values that will be stored into
344     * combobox and build options into it.
345     */
346    public void setValues(List<T> values) {
347        String[] stringValues = new String[values.size()];
348        for (int i = 0; i < values.size(); i++) {
349            stringValues[i] = values.get(i).toString();
350        }
351        $(listbox.getElement()).val(stringValues).trigger("change", selectedIndex);
352    }

354    public Option addItem(T value) {
355        if (!values.contains(value)) {
356            Option opt = new Option(keyFactory.generateKey(value));
357            add(opt);
358            return opt;
359        } else {
360            GWT.log("Cannot add duplicate value: " + value);
361        }
362        return null;
363    }

410    /**
411     * Build the Option Element with provided params
412     */
413    protected Option buildOption(String text, T value) {
414        Option option = new Option();
415        option.setText(text);
416        option.setValue(keyFactory.generateKey(value));
417        return option;
418    }

420    /**
421     * Sets the currently selected index.
422     * <p>
423     * After calling this method, only the specified item in the list will
424     * remain selected. For a ListBox with multiple selection enabled.
425     *
426     * @param selectedIndex - the index of the item to be selected
427     */
428    public void setSelectedIndex(int selectedIndex) {
429        this.selectedIndex = selectedIndex;
430        T value = values.get(selectedIndex);
431        if (value != null) {
432            $(listbox.getElement()).val(value.toString()).trigger("change", selectedIndex);
433        } else {
434            GWT.log("Value index is not found.", new IndexOutOfBoundsException());
435        }
436    }

438    /**
439     * Gets the text for currently selected item. If multiple items are
440     * selected, this method will return the text of the first selected item.
441     *
442     * @return the text for selected item, or {@code null} if none is selected
443     */
444    public int getSelectedIndex() {
445        Object o = $("#" + uid).find("option:selected").last().prop("index");
446        if (o != null) {
447            return Integer.parseInt(o.toString());
448        }
449        return -1;
450    }

459    /**
460     * Get the selected vales from multiple combobox
461     */
462    public List<T> getSelectedValues() {
463        Object[] curVal = (Object[]) $(listbox.getElement()).val();
464
465        List<T> selectedValues = new ArrayList<>();
466        if(curVal == null || curVal.length < 1) {
467            return selectedValues;
468        }
469
470        List<String> keyIndex = getValuesKeyIndex();
471        for (Object val : curVal) {
472            if(val instanceof String) {
473                int selectedIndex = keyIndex.indexOf(val);
474                if (selectedIndex != -1) {
475                    selectedValues.add(values.get(selectedIndex));
476                }
477            }
478        }
479        return selectedValues;
480    }
481
482    protected List<String> getValuesKeyIndex() {
483        List<String> keys = new ArrayList<>();
484        for (T value : values) {
485            keys.add(keyFactory.generateKey(value));
486        }
487        return keys;
488    }
