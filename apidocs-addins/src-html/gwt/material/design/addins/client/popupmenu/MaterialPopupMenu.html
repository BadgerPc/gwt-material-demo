020package gwt.material.design.addins.client.popupmenu;
021
022import com.google.gwt.core.client.Scheduler;
023import com.google.gwt.dom.client.Element;
024import com.google.gwt.event.logical.shared.*;
025import com.google.gwt.event.shared.HandlerRegistration;
026import com.google.gwt.user.client.DOM;
027import gwt.material.design.addins.client.MaterialAddins;
028import gwt.material.design.addins.client.base.constants.AddinsCssName;
029import gwt.material.design.client.MaterialDesignBase;
030import gwt.material.design.client.constants.CssName;
031import gwt.material.design.client.ui.html.UnorderedList;
032import gwt.material.design.jquery.client.api.JQueryElement;
033
034import static gwt.material.design.jquery.client.api.JQuery.$;
035
036/**
037 * Popup Menu.
038 *
039 * @author Mark Kevin
040 * @author Ben Dol
041 */
042public class MaterialPopupMenu extends UnorderedList implements HasSelectionHandlers<Element>, HasOpenHandlers<MaterialPopupMenu>, HasCloseHandlers<MaterialPopupMenu> {
043
044    static {
045        if (MaterialAddins.isDebug()) {
046            MaterialDesignBase.injectCss(MaterialPopupMenuDebugClientBundle.INSTANCE.menuCssDebug());
047        } else {
048            MaterialDesignBase.injectCss(MaterialPopupMenuClientBundle.INSTANCE.menuCss());
049        }
050    }
051
052    private String id;
053    private Object selected;
054    private int popupX;
055    private int popupY;
056
057    public MaterialPopupMenu() {
058        id = DOM.createUniqueId();
059        setInitialClasses(AddinsCssName.POPUP_MENU, AddinsCssName.MENU_BAR, CssName.Z_DEPTH_3);
060    }
061
062    @Override
063    protected void onLoad() {
064        super.onLoad();
065
066        build();
067    }
068
069    @Override
070    protected void build() {
071        $(this).attr("tabindex", "0");
072        $(this).on("blur", e -> {
073            close();
074            return true;
075        });
076
077        $("*").on("scroll." + id, e -> {
078            close();
079            return true;
080        });
081
082        initializeSelectionEvent();
083
084        close();
085    }
086
087    @Override
088    protected void onUnload() {
089        super.onUnload();
090
091        $(this).off("." + id);
092        $("*").off("." + id);
093    }
094
095    private void initializeSelectionEvent() {
096        // Initialization of Selection event
097        $(".popup-menu li").off("click");
098        $(".popup-menu li").on("click", e -> {
099            e.stopPropagation();
100            SelectionEvent.fire(MaterialPopupMenu.this, $(e.getCurrentTarget()).asElement());
101            $(this).hide();
102            return true;
103        });
104
105        // Check if the dropdown is not visible anymore into it's container either left / bottom side
106        $(".popup-menu li").off("mouseover");
107        $(".popup-menu li").on("mouseover", (e, param1) -> {
108            JQueryElement item = $(e.getCurrentTarget()).find("a");
109            if (item.attr("data-activates") != null) {
110                JQueryElement dp = $("#" + item.attr("data-activates"));
111
112                double dpWidth = dp.width();
113                double dpLeft = dp.offset().left;
114                double conWidth = body().width();
115
116                double dpHeight = 200;
117                double dpTop = dp.offset().top;
118                double conHeight = body().height();
119
120                if (dpWidth + dpLeft > conWidth) {
121                    dp.addClass(AddinsCssName.EDGE_LEFT);
122                }
123
124                if (dpHeight + dpTop > conHeight) {
125                    dp.addClass(AddinsCssName.EDGE_BOTTOM);
126                }
127            }
128            return true;
129        });
130
131        $(".popup-menu li").off("mouseleave");
132        $(".popup-menu li").on("mouseleave", (e, param) -> {
133            JQueryElement item = $(e.getCurrentTarget()).find("a");
134            if (item.attr("data-activates") != null) {
135                JQueryElement dp = $("#" + item.attr("data-activates"));
136                dp.removeClass(AddinsCssName.EDGE_LEFT);
137                dp.removeClass(AddinsCssName.EDGE_BOTTOM);
138            }
139            return true;
140        });
141    }
142
143    /**
144     * Set the popup position of the context menu
145     *
146     * @param popupX window x position
147     * @param popupY window y position
148     */
149    public void setPopupPosition(int popupX, int popupY) {
150        // Will check if the popup is out of container
151        this.popupX = popupX;
152        this.popupY = popupY;
153        setLeft(popupX);
154        setTop(popupY);
155    }
156
157    @Override
158    public HandlerRegistration addSelectionHandler(SelectionHandler<Element> selectionHandler) {
159        return addHandler(selectionHandler, SelectionEvent.getType());
160    }
161
162    public void open() {
163        setVisible(true);
164        Scheduler.get().scheduleDeferred(() -> setFocus(true));
165
166        // Check if dropdown is out of the container (Left)
167        if ($(this).width() + $(this).offset().left > body().width()) {
168            setLeft(body().width() - $(this).width());
169        }
170        OpenEvent.fire(this, this);
171    }
172
173    public void close() {
174        setVisible(false);
175        CloseEvent.fire(this, this);
176    }
177
178    public Object getSelected() {
179        return selected;
180    }
181
182    public void setSelected(Object selected) {
183        this.selected = selected;
184    }
185
186    @Override
187    public HandlerRegistration addCloseHandler(CloseHandler<MaterialPopupMenu> closeHandler) {
188        return addHandler(new CloseHandler<MaterialPopupMenu>() {
189            @Override
190            public void onClose(CloseEvent<MaterialPopupMenu> closeEvent) {
191                if (isEnabled()) {
192                    closeHandler.onClose(closeEvent);
193                }
194            }
195        }, CloseEvent.getType());
196    }
197
198    @Override
199    public HandlerRegistration addOpenHandler(OpenHandler<MaterialPopupMenu> openHandler) {
200        return addHandler(new OpenHandler<MaterialPopupMenu>() {
201            @Override
202            public void onOpen(OpenEvent<MaterialPopupMenu> openEvent) {
203                if (isEnabled()) {
204                    openHandler.onOpen(openEvent);
205                }
206            }
207        }, OpenEvent.getType());
208    }
209
210    public int getPopupX() {
211        return popupX;
212    }
213
214    public int getPopupY() {
215        return popupY;
216    }
217}
