024import com.google.gwt.dom.client.Document;
025import com.google.gwt.dom.client.Element;
026import com.google.gwt.dom.client.Style;
027import com.google.gwt.event.dom.client.ClickEvent;
028import com.google.gwt.event.dom.client.ClickHandler;
029import gwt.material.design.client.base.MaterialWidget;
030import gwt.material.design.client.constants.IconType;
031import gwt.material.design.client.constants.WavesType;
032import gwt.material.design.client.ui.MaterialIcon;
033import gwt.material.design.client.ui.html.Span;
034
035public class MaterialUploadHeader extends MaterialWidget {
036
037    private Span uploadedFiles = new Span();
038    private MaterialIcon iconClose = new MaterialIcon(IconType.CLOSE);
039    private MaterialIcon iconColaps = new MaterialIcon(IconType.KEYBOARD_ARROW_DOWN);
040    private MaterialUploadPreview preview;
041
042    public MaterialUploadHeader() {
043        super(Document.get().createDivElement());
044        setStyleName("header");
045        iconClose.setId("upload-close");
046        iconClose.setCircle(true);
047        iconClose.setWaves(WavesType.DEFAULT);
048        iconClose.addClickHandler(new ClickHandler() {
049            @Override
050            public void onClick(ClickEvent clickEvent) {
051                preview.setVisibility(Style.Visibility.HIDDEN);
052            }
053        });
054        iconColaps.setId("upload-colaps");
055        iconColaps.setCircle(true);
056        iconColaps.setWaves(WavesType.DEFAULT);
057        uploadedFiles.setId("no-uploaded-files");
058        add(uploadedFiles);
059        add(iconClose);
060        add(iconColaps);
061    }
062
063    @Override
064    protected void onLoad() {
065        super.onLoad();
066        initUploadHeader(iconClose.getElement(), iconColaps.getElement(), getPreview().getElement(), getPreview().getUploadCollection().getElement());
067    }
068
069    private native void initUploadHeader(Element iconClose, Element iconColaps, Element preview, Element collection) /*-{
070        var toggle = true;
071
072        $wnd.jQuery(iconColaps).click(function() {
073            if(toggle){
074                $wnd.jQuery(collection).css('visibility', 'hidden');
075                $wnd.jQuery(iconColaps).html("keyboard_arrow_up");
076                $wnd.jQuery(collection).css('height', '0px');
077                toggle = false;
078            }else{
079                $wnd.jQuery(collection).css('visibility', 'visible');
080                $wnd.jQuery(iconColaps).html("keyboard_arrow_down");
081                $wnd.jQuery(collection).css('height', 'initial');
082                toggle = true;
083            }
084        });
085    }-*/;
086
087    public MaterialUploadPreview getPreview() {
088        return preview;
089    }
090
091    public void setPreview(MaterialUploadPreview preview) {
092        this.preview = preview;
093    }
094
095    public Span getUploadedFiles() {
096        return uploadedFiles;
097    }
098}
