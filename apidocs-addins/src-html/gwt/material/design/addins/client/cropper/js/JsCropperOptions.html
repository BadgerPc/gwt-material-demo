020package gwt.material.design.addins.client.cropper.js;
021
022import gwt.material.design.addins.client.cropper.constants.Shape;
023import jsinterop.annotations.JsOverlay;
024import jsinterop.annotations.JsPackage;
025import jsinterop.annotations.JsProperty;
026import jsinterop.annotations.JsType;
027
028/**
029 * JsCopper JsInterop Options
030 *
031 * @author kevzlou7979
032 */
033@JsType(isNative = true, name = "Object", namespace = JsPackage.GLOBAL)
034public class JsCropperOptions {
035
036    private JsCropperOptions() {}
037
038    @JsOverlay
039    public static final JsCropperOptions create() {
040        JsCropperOptions options = new JsCropperOptions();
041
042        JsCropperDimension viewPort = new JsCropperDimension();
043        viewPort.height = 200;
044        viewPort.width = 200;
045        viewPort.type = Shape.SQUARE.getCssName();
046
047        JsCropperDimension boundary = new JsCropperDimension();
048        boundary.height = 400;
049        boundary.width = 400;
050        boundary.type = Shape.SQUARE.getCssName();
051
052        options.viewport = viewPort;
053        options.boundary = boundary;
054        options.enableZoom = true;
055        options.enableOrientation = true;
056        options.enableResize = true;
057        options.mouseWheelZoom = true;
058        options.showZoomer = true;
059
060        return options;
061    }
062
063    /**
064     * The outer container of the cropper.
065     */
066    @JsProperty
067    public JsCropperDimension boundary;
068
069    /**
070     * The inner container of the cropper. The visible part of the image
071     */
072    @JsProperty
073    public JsCropperDimension viewport;
074
075    /**
076     * A class of your choosing to add to the container to add custom styles to your cropper
077     */
078    @JsProperty
079    public String customClass;
080
081    /**
082     * Enable or disable support for specifying a custom orientation when binding images.
083     */
084    @JsProperty
085    public boolean enableOrientation;
086
087    /**
088     * Enable zooming functionality. If set to false - scrolling and pinching would not zoom.
089     */
090    @JsProperty
091    public boolean enableZoom;
092
093    /**
094     * Restricts zoom so image cannot be smaller than viewport
095     */
096    @JsProperty
097    public boolean enforceBoundary;
098
099    /**
100     * Provides the ability to resize your viewport target
101     */
102    @JsProperty
103    public boolean enableResize;
104
105    /**
106     * Enable or disable the ability to use the mouse wheel to zoom in and out on a cropper instance
107     */
108    @JsProperty
109    public boolean mouseWheelZoom;
110
111    /**
112     * Hide or Show the zoom slider
113     */
114    @JsProperty
115    public boolean showZoomer;
116}
