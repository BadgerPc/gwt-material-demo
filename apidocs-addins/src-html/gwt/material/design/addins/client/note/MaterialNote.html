020package gwt.material.design.addins.client.note;
021
022import com.google.gwt.dom.client.Document;
023import com.google.gwt.dom.client.Style;
024import gwt.material.design.addins.client.MaterialAddins;
025import gwt.material.design.addins.client.base.constants.AddinsCssName;
026import gwt.material.design.client.MaterialDesignBase;
027import gwt.material.design.client.base.HasIcon;
028import gwt.material.design.client.base.MaterialWidget;
029import gwt.material.design.client.base.mixin.TextMixin;
030import gwt.material.design.client.constants.Color;
031import gwt.material.design.client.constants.IconPosition;
032import gwt.material.design.client.constants.IconSize;
033import gwt.material.design.client.constants.IconType;
034import gwt.material.design.client.ui.MaterialIcon;
035import gwt.material.design.client.ui.MaterialLabel;
036
037//@formatter:off
038
039/**
040 * MaterialNote is basic panel with text indication to quoate a certain part of your website
041 * <p>
042 * <h3>XML Namespace Declaration</h3>
043 * <pre>
044 * {@code
045 * xmlns:ma='urn:import:gwt.material.design.addins.client'
046 * }
047 * </pre>
048 * <p>
049 * <h3>UiBinder Usage:</h3>
050 * <pre>
051 * {@code
052 * <m:MaterialNote text="Some message here" iconType="WARNING" backgroundColor="RED_LIGHTEN_5" textColor="RED" />
053 * }
054 * </pre>
055 *
056 * @author kevzlou7979
057 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#note">Note</a>
058 */
059//@formatter:on
060public class MaterialNote extends MaterialWidget implements HasIcon {
061
062    static {
063        if (MaterialAddins.isDebug()) {
064            MaterialDesignBase.injectCss(MaterialNoteDebugClientBundle.INSTANCE.noteCssDebug());
065        } else {
066            MaterialDesignBase.injectCss(MaterialNoteClientBundle.INSTANCE.noteCss());
067        }
068    }
069
070    private MaterialIcon icon = new MaterialIcon(IconType.STAR);
071    private MaterialLabel lblMessage = new MaterialLabel();
072    private final TextMixin<MaterialLabel> textMixin = new TextMixin<>(lblMessage);
073
074    public MaterialNote() {
075        super(Document.get().createDivElement(), AddinsCssName.MATERIAL_NOTE);
076    }
077
078    @Override
079    protected void onLoad() {
080        super.onLoad();
081
082        add(icon);
083        add(lblMessage);
084    }
085
086    public String getText() {
087        return textMixin.getText();
088    }
089
090    public void setText(String text) {
091        textMixin.setText(text);
092    }
093
094    @Override
095    public MaterialIcon getIcon() {
096        return icon;
097    }
098
099    @Override
100    public void setIconType(IconType iconType) {
101        icon.setIconType(iconType);
102    }
103
104    @Override
105    public void setIconPosition(IconPosition iconPosition) {
106        icon.setIconPosition(iconPosition);
107    }
108
109    @Override
110    public void setIconSize(IconSize iconSize) {
111        icon.setIconSize(iconSize);
112    }
113
114    @Override
115    public void setIconFontSize(double v, Style.Unit unit) {
116        icon.setIconFontSize(v, unit);
117    }
118
119    @Override
120    public void setIconColor(Color color) {
121        icon.setIconColor(color);
122    }
123
124    @Override
125    public Color getIconColor() {
126        return icon.getIconColor();
127    }
128
129    @Override
130    public void setIconPrefix(boolean b) {
131        icon.setIconPrefix(b);
132    }
133
134    @Override
135    public boolean isIconPrefix() {
136        return icon.isIconPrefix();
137    }
138}
