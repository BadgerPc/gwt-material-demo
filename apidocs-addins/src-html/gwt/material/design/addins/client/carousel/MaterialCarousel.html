115    @Override
116    protected void onLoad() {
117        super.onLoad();
118
119        wrapper.setStyleName(AddinsCssName.MATERIAL_CAROUSEL_CONTAINER);
120        wrapper.add(container);
121
122        super.add(nextArrow);
123        super.add(previousArrow);
124        super.add(wrapper);
125
126        $(getElement()).on(CarouselEvents.AFTER_CHANGE, (e, slick, currentSlide) -> {
127            AfterChangeEvent.fire(this, Integer.parseInt(currentSlide.toString()));
128            return true;
129        });
130
131        $(getElement()).on(CarouselEvents.BEFORE_CHANGE, (e, slick, currentSlide, nextSlide) -> {
132            BeforeChangeEvent.fire(this, Integer.parseInt(currentSlide.toString()), Integer.parseInt(nextSlide.toString()));
133            e.stopPropagation();
134            return true;
135        });
136
137        $(getElement()).on(CarouselEvents.INIT, (e) -> {
138            InitEvent.fire(this);
139            return true;
140        });
141
142        $(getElement()).on(CarouselEvents.DESTROY, (e) -> {
143            DestroyEvent.fire(this);
144            return true;
145        });
146
147        $(getElement()).on(CarouselEvents.SWIPE, (e, slick, direction) -> {
148            SwipeEvent.fire(this, direction.toString());
149            return true;
150        });
151
152        load();
153    }

155    @Override
156    public void load() {
157        options.nextArrow = "#" + getBtnNextArrow().getId();
158        options.prevArrow = "#" + getBtnPrevArrow().getId();
159        $(container.getElement()).slick(options);
160    }
161
162    @Override
163    protected void onUnload() {
164        super.onUnload();
165
166        unload();
167    }
168
169    @Override
170    public void unload() {
171        $(getElement()).off(CarouselEvents.AFTER_CHANGE);
172        $(getElement()).off(CarouselEvents.BEFORE_CHANGE);
173        $(getElement()).off(CarouselEvents.INIT);
174        $(getElement()).off(CarouselEvents.DESTROY);
175        $(getElement()).off(CarouselEvents.SWIPE);
176
177        destroy();
178    }
179
180    @Override
181    public void reload() {
182        unload();
183        load();
184    }
185
186    public void destroy() {
187        if (container != null) {
188            command("destroy");
189        }
190    }

265    protected Object command(String action, Object... params) {
266        if (container == null && container.isAttached()) {
267            GWT.log("Your carousel container is not yet initialized", new IllegalStateException());
268        } else {
269            return $("#" + container.getId()).slick(action, params);
270        }
271        return null;
272    }

478    public void setTabNavigation(MaterialTab tab) {
