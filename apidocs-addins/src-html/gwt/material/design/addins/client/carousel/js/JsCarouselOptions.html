022import jsinterop.annotations.JsOverlay;
023import jsinterop.annotations.JsPackage;
024import jsinterop.annotations.JsProperty;
025import jsinterop.annotations.JsType;

027/**
028 * Options for carousel component
029 *
030 * @author kevzlou7979
031 */
032@JsType(isNative = true, name = "Object", namespace = JsPackage.GLOBAL)
033public class JsCarouselOptions {
034
035    private JsCarouselOptions() {}
036
037    @JsOverlay
038    public final static JsCarouselOptions create() {
039        JsCarouselOptions options = new JsCarouselOptions();
040        options.infinite = false;
041        options.centerMode = false;
042        options.variableWidth = false;
043        options.dots = true;
044        options.arrows = true;
045        options.adaptiveHeight = false;
046        options.autoplay = false;
047        options.fade = false;
048        options.slidesToShow = 1;
049        options.slidesToScroll = 1;
050        options.speed = 300;
051        options.autoplaySpeed = 3000;
052        options.edgeFriction = 0.15;
053        options.swipeToSlide = true;
054        options.focusOnSelect = false;
055        options.centerPadding = "100px";
056
057        // Responsive options
058        JsResponsiveOptions[] responsiveOptions = new JsResponsiveOptions[]{};
059        JsCarouselOptions tabletSettings = new JsCarouselOptions();
060        JsCarouselOptions mobileSettings = new JsCarouselOptions();
061
062        // Tablet Settings
063        if (tabletSettings != null) {
064            JsResponsiveOptions tabletOpt = new JsResponsiveOptions();
065            tabletOpt.breakpoint = 992;
066            tabletOpt.settings = tabletSettings;
067            responsiveOptions[0] = tabletOpt;
068        }
069
070        // Mobile Settings
071        if (mobileSettings != null) {
072            JsResponsiveOptions mobileOpt = new JsResponsiveOptions();
073            mobileOpt.breakpoint = 600;
074            mobileOpt.settings = mobileSettings;
075            responsiveOptions[1] = mobileOpt;
076        }
077
078        options.responsive = responsiveOptions;
079
080        return options;
081    }
