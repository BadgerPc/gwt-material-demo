020package gwt.material.design.addins.client.countup;
021
022import com.google.gwt.dom.client.Document;
023import com.google.gwt.dom.client.Element;
024import gwt.material.design.addins.client.MaterialAddins;
025import gwt.material.design.addins.client.countup.js.CountUp;
026import gwt.material.design.addins.client.countup.js.JsCountUpOptions;
027import gwt.material.design.client.MaterialDesignBase;
028import gwt.material.design.client.base.AbstractValueWidget;
029import gwt.material.design.jquery.client.api.Functions;
030
031//@formatter:off
032
033/**
034 * Animates a numerical value by counting it up with delightful animations.
035 * <p>
036 * <h3>XML Namespace Declaration</h3>
037 * <pre>
038 * {@code
039 * xmlns:ma='urn:import:gwt.material.design.addins.client'
040 * }
041 * </pre>
042 * <p>
043 * <h3>UiBinder Usage:</h3>
044 * <pre>
045 * {@code
046 * <ma:countup.MaterialCountUp ui:field="lblNumber" startValue="0" endValue="100" />
047 * }
048 * </pre>
049 * <p>
050 * <h3>Java Usage:</h3>
051 * <pre>
052 * {@code
053 * // To start the animation
054 * lblNumber.start();
055 * }
056 * </pre>
057 * </p>
058 *
059 * @author kevzlou7979
060 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#countup">Material CountUp</a>
061 * @see <a href="https://github.com/inorganik/countUp.js">CountUpJs 1.9.0</a>
062 */
063//@formatter:on
064public class MaterialCountUp extends AbstractValueWidget<Double> {
065
066    static {
067        if (MaterialAddins.isDebug()) {
068            MaterialDesignBase.injectDebugJs(MaterialCountUpDebugClientBundle.INSTANCE.countUpJsDebug());
069        } else {
070            MaterialDesignBase.injectJs(MaterialCountUpClientBundle.INSTANCE.countUpJs());
071        }
072    }
073
074    private CountUp countUp;
075
076
077    private JsCountUpOptions options = JsCountUpOptions.create();
078
079    public MaterialCountUp() {
080        super(Document.get().createLabelElement());
081    }
082
083    public void start() {
084        countUp = new CountUp(getElement(), getStartValue(), getEndValue(), getDecimals(), getDuration(), options);
085        countUp.start(getCallback());
086    }
087
088    /**
089     * Toggle pause/resume
090     */
091    public void pauseResume() {
092        countUp.pauseResume();
093    }
094
095    /**
096     * Reset an animation
097     */
098    public void reset() {
099        countUp.reset();
100    }
101
102    public double getStartValue() {
103        return options.startValue;
104    }
105
106    /**
107     * The value you want to begin at (Default : 0)
108     */
109    public void setStartValue(double startValue) {
110        options.startValue = startValue;
111    }
112
113    public double getEndValue() {
114        return options.endValue;
115    }
116
117    @Override
118    public Double getValue() {
119        return getEndValue();
120    }
121
122    /**
123     * The value you want to end at (Default : 0)
124     */
125    public void setEndValue(double endValue) {
126        options.endValue = endValue;
127    }
128
129    public double getDecimals() {
130        return options.decimals;
131    }
132
133    /**
134     * Number of decimal places in number (Default : 0)
135     */
136    public void setDecimals(double decimals) {
137        options.decimals = decimals;
138    }
139
140    public double getDuration() {
141        return options.duration;
142    }
143
144    /**
145     * Duration in seconds (Default : 2sec)
146     */
147    public void setDuration(double duration) {
148        options.duration = duration;
149    }
150
151    public String getSeparator() {
152        return options.separator;
153    }
154
155    /**
156     * Adds separator to your numerical value (Default : ",")
157     */
158    public void setSeparator(String separator) {
159        options.separator = separator;
160    }
161
162    public String getDecimal() {
163        return options.decimal;
164    }
165
166    /**
167     * Set the decimal symbol (Default ".")
168     */
169    public void setDecimal(String decimal) {
170        options.decimal = decimal;
171    }
172
173    public String getPrefix() {
174        return options.prefix;
175    }
176
177    /**
178     * Set the prefix, good for currency values like $ (dollar sign) (Default : " ")
179     */
180    public void setPrefix(String prefix) {
181        options.prefix = prefix;
182    }
183
184    public String getSuffix() {
185        return options.suffix;
186    }
187
188    /**
189     * Set the suffix value for your numbers
190     */
191    public void setSuffix(String suffix) {
192        options.suffix = suffix;
193    }
194
195    public Functions.Func getCallback() {
196        return options.callback;
197    }
198
199    /**
200     * Set the callback function after the count up reach the end value.
201     */
202    public void setCallback(Functions.Func callback) {
203        options.callback = callback;
204    }
205}
