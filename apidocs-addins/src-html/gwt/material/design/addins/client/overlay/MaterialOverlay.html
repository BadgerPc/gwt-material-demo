022import com.google.gwt.core.client.GWT;
023import com.google.gwt.dom.client.Document;
024import com.google.gwt.dom.client.Style;
025import com.google.gwt.event.logical.shared.*;
026import com.google.gwt.event.shared.HandlerRegistration;
027import gwt.material.design.addins.client.MaterialAddins;
028import gwt.material.design.addins.client.base.constants.AddinsCssName;
029import gwt.material.design.addins.client.pathanimator.MaterialPathAnimator;
030import gwt.material.design.client.MaterialDesignBase;
031import gwt.material.design.client.base.HasDurationTransition;
032import gwt.material.design.client.base.MaterialWidget;
033import gwt.material.design.client.constants.Color;
034import gwt.material.design.client.constants.IconType;
035import gwt.material.design.client.ui.MaterialIcon;
036
037import static gwt.material.design.jquery.client.api.JQuery.$;

039//@formatter:off

041/**
042 * It's an overlay panel layout wherein you can put as many widgets as you want and design it.
043 * You can do advance stuff by implementing Path Animator into the overlay panel container.
044 * <p>
045 * <h3>XML Namespace Declaration</h3>
046 * <pre>
047 * {@code
048 * xmlns:ma='urn:import:gwt.material.design.addins.client'
049 * }
050 * </pre>
051 * <p>
052 * <h3>UiBinder Usage:</h3>
053 * <pre>
054 * {@code
055 * <ma:overlay.MaterialOverlay background="blue">
056 *      &lt;-- Some content here -->
057 * </ma:overlay.MaterialOverlay>
058 * }
059 * </pre>
060 *
061 * @author kevzlou7979
062 */
063//@formatter:on
064public class MaterialOverlay extends MaterialWidget implements HasOpenHandlers<MaterialOverlay>,
065        HasCloseHandlers<MaterialOverlay>, HasDurationTransition {
066
067    static {
068        if (MaterialAddins.isDebug()) {
069            MaterialDesignBase.injectCss(MaterialOverlayDebugClientBundle.INSTANCE.overlayCssDebug());
070        } else {
071            MaterialDesignBase.injectCss(MaterialOverlayClientBundle.INSTANCE.overlayCss());
072        }
073    }
074
075    private MaterialWidget source;
076    private MaterialPathAnimator animator = new MaterialPathAnimator();
077    private MaterialOverlayTab overlayTab;
078    private MaterialIcon minimizeIcon = new MaterialIcon(IconType.KEYBOARD_ARROW_DOWN);
079
080    public MaterialOverlay() {
081        super(Document.get().createDivElement(), AddinsCssName.OVERLAY_PANEL);
082    }
083
084    public MaterialOverlay(Color backgroundColor) {
085        this();
086        setBackgroundColor(backgroundColor);
087    }
088
089    public MaterialOverlay(Color backgroundColor, Style.Visibility visibility, Double opacity) {
090        this(backgroundColor);
091        setVisibility(visibility);
092        setOpacity(opacity);
093    }
094
095    /**
096     * Open the Overlay Panel with Path Animator applied
097     */
098    public void open(MaterialWidget source) {
099        this.source = source;
100        $("body").attr("style", "overflow: hidden !important");
101        animator.setSourceElement(source.getElement());
102        animator.setTargetElement(getElement());
103        animator.animate();
104        OpenEvent.fire(this, this);
105    }
106
107    /**
108     * Open the Overlay Panel without Path Animator
109     */
110    public void open() {
111        setVisibility(Style.Visibility.VISIBLE);
112        setOpacity(1);
113        OpenEvent.fire(this, this);
114    }
115
116    /**
117     * Close the Overlay Panel with Path Animator applied
118     */
119    public void close() {
120        body().attr("style", "overflow: auto !important");
121        if (source != null) {
122            animator.reverseAnimate();
123        } else {
124            setOpacity(0);
125            setVisibility(Style.Visibility.HIDDEN);
126        }
127        CloseEvent.fire(this, this);
128    }
129
130    /**
131     * Get source widget for path animator
132     */
133    public MaterialWidget getSource() {
134        return source;
135    }
136
137    /**
138     * Set source widget for path animator
139     */
140    public void setSource(MaterialWidget source) {
141        this.source = source;
142    }
143
144    @Override
145    public int getDuration() {
146        return animator.getDuration();
147    }
148
149    @Override
150    public void setDuration(int duration) {
151        animator.setDuration(duration);
152    }
153
154    public int getTargetShowDuration() {
155        return animator.getTargetShowDuration();
156    }
157
158    /**
159     * Duration (in milliseconds) of targetElement to become visible, if hidden initially. The library will automatically try to figure this out from the element's computed styles. Default is 0 seconds.
160     */
161    public void setTargetShowDuration(int targetShowDuration) {
162        animator.setTargetShowDuration(targetShowDuration);
163    }
164
165    public int getExtraTransitionDuration() {
166        return animator.getExtraTransitionDuration();
167    }
168
169    /**
170     * Extra duration (in milliseconds) of targetElement to provide visual continuity between the animation and the rendering of the targetElement. Default is 1 second
171     */
172    public void setExtraTransitionDuration(int extraTransitionDuration) {
173        animator.setExtraTransitionDuration(extraTransitionDuration);
174    }
175
176    @Override
177    public HandlerRegistration addCloseHandler(CloseHandler<MaterialOverlay> closeHandler) {
178        return addHandler(new CloseHandler<MaterialOverlay>() {
179            @Override
180            public void onClose(CloseEvent<MaterialOverlay> closeEvent) {
181                if (isEnabled()) {
182                    closeHandler.onClose(closeEvent);
183                }
184            }
185        }, CloseEvent.getType());
186    }
187
188    @Override
189    public HandlerRegistration addOpenHandler(OpenHandler<MaterialOverlay> openHandler) {
190        return addHandler(new OpenHandler<MaterialOverlay>() {
191            @Override
192            public void onOpen(OpenEvent<MaterialOverlay> openEvent) {
193                if (isEnabled()) {
194                    openHandler.onOpen(openEvent);
195                }
196            }
197        }, OpenEvent.getType());
198    }
199
200    public MaterialOverlayTab getOverlayTab() {
201        return overlayTab;
202    }
203
204    public void setOverlayTab(MaterialOverlayTab overlayTab) {
205        this.overlayTab = overlayTab;
206        minimizeIcon.addStyleName(AddinsCssName.MINIMIZE_ICON);
207        minimizeIcon.addMouseDownHandler(e -> minimize());
208        add(minimizeIcon);
209    }
210
211    protected void minimize() {
212        if (overlayTab != null) {
213            overlayTab.minimize(this);
214        } else {
215            GWT.log("You must set the overlay container before minimizing the overlay.", new IllegalStateException());
216        }
217    }
218
219    public MaterialIcon getMinimizeIcon() {
220        return minimizeIcon;
221    }
222}
