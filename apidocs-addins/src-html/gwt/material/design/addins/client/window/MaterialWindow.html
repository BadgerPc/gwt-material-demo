024import com.google.gwt.dom.client.Document;
025import com.google.gwt.dom.client.Style;

035import gwt.material.design.client.base.MaterialWidget;
036import gwt.material.design.client.base.mixin.ColorsMixin;
037import gwt.material.design.client.base.mixin.ToggleStyleMixin;
038import gwt.material.design.client.constants.IconType;
039import gwt.material.design.client.constants.WavesType;
040import gwt.material.design.client.ui.MaterialIcon;
041import gwt.material.design.client.ui.MaterialLink;
042import gwt.material.design.client.ui.animate.MaterialAnimator;
043import gwt.material.design.client.ui.animate.Transition;

045//@formatter:off

047/**
048 * Window is another kind of Modal but it has a header toolbar for maximizing and close the window. Also you can attached a tab component on its content.
049 *
050 * <h3>XML Namespace Declaration</h3>
051 * <pre>
052 * {@code
053 * xmlns:ma='urn:import:gwt.material.design.addins.client'
054 * }
055 * </pre>
056 *
057 * <h3>UiBinder Usage:</h3>
058 * <pre>
059 * {@code
060 *  <ma:window.MaterialWindow ui:field="window" />
061 * }
062 * </pre>
063 *
064 * <h3>UiBinder Usage:</h3>
065 * <pre>
066 * {@code
067 *  // Opening a window
068 *  window.openWindow();
069 *
070 *  // Closing a window
071 *  window.closeWindow();
072 * }
073 * </pre>
074 *
075 * @author kevzlou7979
076 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#window">Material Window</a>
077 */
078//@formatter:on
079public class MaterialWindow extends MaterialWidget implements HasCloseHandlers<Boolean>, HasOpenHandlers<Boolean>{
080
081    static {
082        if(MaterialResourceInjector.isDebug()) {
083            MaterialResourceInjector.injectCss(MaterialWindowDebugClientBundle.INSTANCE.windowCssDebug());
084        } else {
085            MaterialResourceInjector.injectCss(MaterialWindowClientBundle.INSTANCE.windowCss());
086        }
087    }
088
089    private MaterialWidget window = new MaterialWidget(Document.get().createDivElement());
090    private MaterialWidget content = new MaterialWidget(Document.get().createDivElement());
091
092    // Toolbar elements
093    private MaterialWidget toolbar = new MaterialWidget(Document.get().createDivElement());
094    private MaterialIcon iconMaximize = new MaterialIcon(IconType.CHECK_BOX_OUTLINE_BLANK);
095    private MaterialIcon iconClose = new MaterialIcon(IconType.CLOSE);
096    private MaterialLink link = new MaterialLink();
097    private String title = "";
098    private String toolbarColor;
099
100    private final ColorsMixin<MaterialWidget> toolbarColorMixin = new ColorsMixin<>(toolbar);
101    private final ToggleStyleMixin<MaterialWidget> maximizeMixin = new ToggleStyleMixin<>(window, "maximize");
102    private final ToggleStyleMixin<MaterialWindow> closeMixin = new ToggleStyleMixin<>(this, "open");
103    private boolean maximize = true;
104    private boolean open = false;
105
106
107    public MaterialWindow() {
108        super(Document.get().createDivElement());
109        window.setStyleName("window");
110        content.setStyleName("content");
111        super.add(window);
112        setStyleName("window-overlay");
113        initWindow();
114    }
115
116    /**
117     * Builds the toolbar
118     */
119    private void initWindow() {
120        toolbar.setStyleName("window-toolbar");
121        link.setStyleName("window-title");
122        iconClose.addStyleName("window-action");
123        iconMaximize.addStyleName("window-action");
124        iconClose.setCircle(true);
125        iconClose.setWaves(WavesType.DEFAULT);
126        iconMaximize.setCircle(true);
127        iconMaximize.setWaves(WavesType.DEFAULT);
128        toolbar.add(link);
129        toolbar.add(iconClose);
130        toolbar.add(iconMaximize);
131        window.add(toolbar);
132        window.add(content);
133
134        // Add handlers to action buttons
135        iconMaximize.addClickHandler(new ClickHandler() {
136            @Override
137            public void onClick(ClickEvent event) {
138                if(maximize){
139                    setMaximize(true);
140                    maximize = false;
141                }else{
142                    setMaximize(false);
143                    maximize = true;
144                }
145            }
146        });
147        iconClose.addClickHandler(new ClickHandler() {
148            @Override
149            public void onClick(ClickEvent event) {
150                if(open){
151                    openWindow();
152                    open = false;
153                }else{
154                    closeWindow();
155                    open = true;
156                }
157            }
158        });
159
160        // Add a draggable header
161        MaterialDnd dnd = new MaterialDnd();
162        dnd.setTarget(window);
163        dnd.setIgnoreFrom(content);
164        dnd.setRestriction(new Restriction(Restriction.Restrict.PARENT, true, -0.3, 0, 1.1, 1));
165    }
166
167    @Override
168    public void add(Widget child) {
169        content.add(child);
170    }
171
172    @Override
173    public String getTitle() {
174        return title;
175    }
176
177    @Override
178    public void setTitle(String title) {
179        this.title = title;
180        link.setText(title);
181    }
182
183    public boolean isMaximize() {
184        return maximizeMixin.isOn();
185    }
186
187    public void setMaximize(boolean maximize) {
188        maximizeMixin.setOn(maximize);
189        if(maximizeMixin.isOn()){
190            iconMaximize.setIconType(IconType.FILTER_NONE);
191        }else{
192            iconMaximize.setIconType(IconType.CHECK_BOX_OUTLINE_BLANK);
193        }
194    }
195
196    /**
197     * Open the window
198     */
199    public void openWindow() {
200        this.open = false;
201        OpenEvent.fire(this, true);
202        MaterialAnimator.animate(Transition.ZOOMIN, window, 0, 200);
203        closeMixin.setOn(true);
204    }
205
206    /**
207     * Close the window
208     */
209    public void closeWindow() {
210        this.open = true;
211        CloseEvent.fire(this, false);
212        Runnable callback = new Runnable() {
213            @Override
214            public void run() {
215                closeMixin.setOn(false);
216                RootPanel.get().getElement().getStyle().setCursor(Style.Cursor.DEFAULT);
217            }
218        };
219        MaterialAnimator.animate(Transition.ZOOMOUT, window, 0, callback);
220    }
221
222    public String getToolbarColor() {
223        return toolbarColor;
224    }
225
226    public void setToolbarColor(String toolbarColor) {
227        this.toolbarColor = toolbarColor;
228        toolbarColorMixin.setBackgroundColor(toolbarColor);
229    }
230
231    @Override
232    public HandlerRegistration addCloseHandler(CloseHandler<Boolean> handler) {
233        return addHandler(handler, CloseEvent.getType());
234    }
235
236    @Override
237    public HandlerRegistration addOpenHandler(OpenHandler<Boolean> handler) {
238        return addHandler(handler, OpenEvent.getType());
239    }
240
241    @Override
242    public void setWidth(String width) {
243        window.setWidth(width);
244    }
245
246    @Override
247    public void setHeight(String height) {
248        window.setHeight(height);
249    }
250
251    @Override
252    public void setBackgroundColor(String bgColor) {
253        window.setBackgroundColor(bgColor);
254    }
255
256    public boolean isOpen() {
257        return open;
258    }
259}
