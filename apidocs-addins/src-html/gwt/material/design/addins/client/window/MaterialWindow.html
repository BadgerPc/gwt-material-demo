047/**
048 * Window is another kind of Modal but it has a header toolbar for maximizing and close the windowContainer. Also you can attached a tab component on its content.
049 * <p>
050 * <h3>XML Namespace Declaration</h3>
051 * <pre>
052 * {@code
053 * xmlns:ma='urn:import:gwt.material.design.addins.client'
054 * }
055 * </pre>
056 * <p>
057 * <h3>UiBinder Usage:</h3>
058 * <pre>
059 * {@code
060 *  <ma:windowContainer.MaterialWindow ui:field="windowContainer" />
061 * }
062 * </pre>
063 * <p>
064 * <h3>UiBinder Usage:</h3>
065 * <pre>
066 * {@code
067 *  // Opening a windowContainer
068 *  windowContainer.open();
069 *
070 *  // Closing a windowContainer
071 *  windowContainer.close();
072 * }
073 * </pre>
074 *
075 * @author kevzlou7979
076 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#window">Material Window</a>
077 */
078//@formatter:on
079public class MaterialWindow extends MaterialWidget implements HasCloseHandlers<Boolean>, HasOpenHandlers<Boolean> {
080
081    static {
082        if (MaterialAddins.isDebug()) {
083            MaterialDesignBase.injectCss(MaterialWindowDebugClientBundle.INSTANCE.windowCssDebug());
084        } else {
085            MaterialDesignBase.injectCss(MaterialWindowClientBundle.INSTANCE.windowCss());
086        }
087    }
088
089    private MaterialWidget windowContainer = new MaterialWidget(Document.get().createDivElement());
090    private MaterialWidget content = new MaterialWidget(Document.get().createDivElement());
091
092    // Toolbar elements
093    private MaterialLink labelTitle = new MaterialLink();
094    private MaterialWidget toolbar = new MaterialWidget(Document.get().createDivElement());
095    private MaterialIcon iconMaximize = new MaterialIcon(IconType.CHECK_BOX_OUTLINE_BLANK);
096    private MaterialIcon iconClose = new MaterialIcon(IconType.CLOSE);
097
098    private String title = "";
099    private Color toolbarColor;
100
101    private final ColorsMixin<MaterialWidget> toolbarColorMixin = new ColorsMixin<>(toolbar);
102    private final ToggleStyleMixin<MaterialWidget> maximizeMixin = new ToggleStyleMixin<>(windowContainer, AddinsCssName.MAXIMIZE);
103    private final ToggleStyleMixin<MaterialWindow> openMixin = new ToggleStyleMixin<>(this, AddinsCssName.OPEN);
104
105    private MaterialAnimation openAnimation;
106    private MaterialAnimation closeAnimation;
107
108    public MaterialWindow() {
109        super(Document.get().createDivElement(), AddinsCssName.WINDOW_OVERLAY);
110        windowContainer.setStyleName(AddinsCssName.WINDOW);
111        content.setStyleName(AddinsCssName.CONTENT);
112        super.add(windowContainer);
113
114        initialize();
115    }
116
117    public MaterialWindow(String title) {
118        this();
119        setTitle(title);
120    }
121
122    public MaterialWindow(String title, Color backgroundColor, Color textColor) {
123        this(title);
124        setBackgroundColor(backgroundColor);
125        setTextColor(textColor);
126    }
127
128    public MaterialWindow(String title, Color backgroundColor, Color textColor, Color toolbarColor) {
129        this(title, backgroundColor, textColor);
130        setToolbarColor(toolbarColor);
131    }
132
133    /**
134     * Builds the toolbar
135     */
136    protected void initialize() {
137        toolbar.setStyleName(AddinsCssName.WINDOW_TOOLBAR);
138        labelTitle.setStyleName(AddinsCssName.WINDOW_TITLE);
139        iconClose.addStyleName(AddinsCssName.WINDOW_ACTION);
140        iconMaximize.addStyleName(AddinsCssName.WINDOW_ACTION);
141        iconClose.setCircle(true);
142        iconClose.setWaves(WavesType.DEFAULT);
143        iconMaximize.setCircle(true);
144        iconMaximize.setWaves(WavesType.DEFAULT);
145        toolbar.add(labelTitle);
146        toolbar.add(iconClose);
147        toolbar.add(iconMaximize);
148        toolbar.addDomHandler(event -> {
149            toggleMaximize();
150            Document.get().getDocumentElement().getStyle().setCursor(Style.Cursor.DEFAULT);
151        }, DoubleClickEvent.getType());
152        windowContainer.add(toolbar);
153        windowContainer.add(content);
154
155        // Add handlers to action buttons
156        iconMaximize.addClickHandler(event -> toggleMaximize());
157        iconClose.addClickHandler(event -> {
158            if (!isOpen()) {
159                open();
160            } else {
161                close();
162            }
163        });
164
165        // Add a draggable header
166        MaterialDnd dnd = MaterialDnd.draggable(windowContainer,
167                JsDragOptions.create(new Restriction(Restriction.Restrict.PARENT, true, -0.3, 0, 1.1, 1)));
168        dnd.ignoreFrom(".content, .window-action");
169    }
170
171    protected void toggleMaximize() {
172        setMaximize(!isMaximized());
173    }
174
175    @Override
176    public void add(Widget child) {
177        content.add(child);
178    }
179
180    @Override
181    public String getTitle() {
182        return title;
183    }
184
185    @Override
186    public void setTitle(String title) {
187        this.title = title;
188        labelTitle.setText(title);
189    }
190
191    public boolean isMaximized() {
192        return maximizeMixin.isOn();
193    }
194
195    public void setMaximize(boolean maximize) {
196        maximizeMixin.setOn(maximize);
197        if (maximizeMixin.isOn()) {
198            iconMaximize.setIconType(IconType.FILTER_NONE);
199        } else {
200            iconMaximize.setIconType(IconType.CHECK_BOX_OUTLINE_BLANK);
201        }
202    }
203
204    /**
205     * Open the windowContainer.
206     */
207    public void open() {
208        if (!isAttached()) {
209            RootPanel.get().add(this);
210        }
211        OpenEvent.fire(this, true);
212        if (openAnimation == null) {
213            openMixin.setOn(true);
214        } else {
215            openAnimation.animate(windowContainer, () -> openMixin.setOn(true));
216        }
217    }
218
219    /**
220     * Close the windowContainer.
221     */
222    public void close() {
223        CloseEvent.fire(this, false);
224        // Turn back the cursor to POINTER
225        RootPanel.get().getElement().getStyle().setCursor(Style.Cursor.DEFAULT);
226        if (closeAnimation == null) {
227            openMixin.setOn(false);
228        } else {
229            closeAnimation.animate(windowContainer, () -> openMixin.setOn(false));
230        }
231    }
232
233    public Color getToolbarColor() {
234        return toolbarColor;
235    }
236
237    public void setToolbarColor(Color toolbarColor) {
238        this.toolbarColor = toolbarColor;
239        toolbarColorMixin.setBackgroundColor(toolbarColor);
240    }
241
242    public void setOpenAnimation(final MaterialAnimation openAnimation) {
243        this.openAnimation = openAnimation;
244    }
245
246    public void setCloseAnimation(final MaterialAnimation closeAnimation) {
247        this.closeAnimation = closeAnimation;
248    }
249
250    @Override
251    public HandlerRegistration addCloseHandler(final CloseHandler<Boolean> handler) {
252        return addHandler(handler, CloseEvent.getType());
253    }
254
255    @Override
256    public HandlerRegistration addOpenHandler(final OpenHandler<Boolean> handler) {
257        return addHandler(handler, OpenEvent.getType());
258    }
259
260    @Override
261    public void setWidth(String width) {
262        windowContainer.setWidth(width);
263    }
264
265    @Override
266    public void setHeight(String height) {
267        windowContainer.setHeight(height);
268    }
269
270    @Override
271    public void setBackgroundColor(Color bgColor) {
272        windowContainer.setBackgroundColor(bgColor);
273    }
274
275    public boolean isOpen() {
276        return openMixin.isOn();
277    }
278
279    public MaterialWidget getWindowContainer() {
280        return windowContainer;
281    }
282
283    public MaterialWidget getToolbar() {
284        return toolbar;
285    }
286
287    public MaterialWidget getContent() {
288        return content;
289    }
290
291    public MaterialIcon getIconMaximize() {
292        return iconMaximize;
293    }
294
295    public MaterialIcon getIconClose() {
296        return iconClose;
297    }
298
299    public MaterialLink getLabelTitle() {
300        return labelTitle;
301    }
