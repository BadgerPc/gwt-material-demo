049/**
050 * A panel that allows any of its nested children to be swiped away.
051 *
052 * <h3>XML Namespace Declaration</h3>
053 * <pre>
054 * {@code
055 * xmlns:ma='urn:import:gwt.material.design.addins.client'
056 * }
057 * </pre>
058 *
059 * <h3>UiBinder Usage:</h3>
060 * <pre>
061 *{
062 * @code
063 * <ma:swipeable.MaterialSwipeablePanel ui:field="swipeablePanel" shadow="1" backgroundColor="white" padding="12">
064 *   <m:MaterialLabel text="You can swipe native components. This is a plain label" backgroundColor="yellow" padding="12" />
065 *   <m:MaterialCard>
066 *     <m:MaterialCardContent>
067 *       <m:MaterialLabel text="This is another Card Component that is swipeable." />
068 *       </m:MaterialCardContent>
069 *     </m:MaterialCard>
070 * </ma:swipeable.MaterialSwipeablePanel>
071 * }
072 * </pre>
073 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#swipeable">Material Swipeable</a>
074 * @author kevzlou7979
075 * @author Ben Dol
076 */
077//@formatter:on
078public class MaterialSwipeablePanel extends MaterialWidget implements HasSwipeable<Widget> {
079
080    static {
081        if(MaterialResourceInjector.isDebug()) {
082            MaterialResourceInjector.injectCss(MaterialSwipeableDebugClientBundle.INSTANCE.swipeableCssDebug());
083        } else {
084            MaterialResourceInjector.injectCss(MaterialSwipeableClientBundle.INSTANCE.swipeableCss());
085        }
086    }
087
088    private final String DISABLED = "disabled-swipe";
089
090    public MaterialSwipeablePanel() {
091        super(Document.get().createDivElement());
092        setStyleName("swipeable");
093    }
094
095    @Override
096    protected void onLoad() {
097        super.onLoad();
098        for(Widget w : getChildren()) {
099            if(!w.getStyleName().contains(DISABLED)) {
100                initSwipeable(w.getElement(), w);
101            }
102        }
103    }
104
105
106    @Override
107    public void initSwipeable(Element element, Widget target) {
108        initSwipeableElement(element, target);
109    }

189    @Override
190    public HandlerRegistration addSwipeLeft(SwipeLeftEvent.SwipeLeftHandler<Widget> handler) {
191        return addHandler(handler, SwipeLeftEvent.getType());
192    }
193
194    @Override
195    public HandlerRegistration addSwipeRight(SwipeRightEvent.SwipeRightHandler<Widget> handler) {
196        return addHandler(handler, SwipeRightEvent.getType());
197    }
198
199    /**
200     * Fire the Swipe left event listener
201     * @param target
202     */
203    public void fireSwipeLeftEvent(Widget target) {
204        SwipeLeftEvent.fire(MaterialSwipeablePanel.this, target);
205    }
206
207    /**
208     * Fire the Swipe right event listener
209     * @param target
210     */
211    public void fireSwipeRightEvent(Widget target) {
212        SwipeRightEvent.fire(MaterialSwipeablePanel.this, target);
213    }
214
215    public void setDisable(Widget... widgets) {
216        for(Widget w : widgets){
217            w.addStyleName(DISABLED);
218        }
219    }
