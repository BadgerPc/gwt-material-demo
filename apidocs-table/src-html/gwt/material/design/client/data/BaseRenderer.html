051/**
052 * Base Component Renderer used by {@link AbstractDataView}.
053 * <br><br>
054 * This can be extended upon or can be replaced all together.
055 *
056 * @author Ben Dol
057 */
058public class BaseRenderer<T> implements Renderer<T> {
059
060    private final Logger logger = Logger.getLogger(BaseRenderer.class.getName());
061
062    // Configurations
063    private int calculatedRowHeight = 55;
064    private int expectedRowHeight = calculatedRowHeight;
065
066    private IconType sortAscIcon = IconType.ARROW_UPWARD;
067    private IconType sortDescIcon = IconType.ARROW_DOWNWARD;
068    private IconSize sortIconSize = IconSize.TINY;
069
070    @Override
071    public TableRow drawRow(DataView<T> dataView, RowComponent<T> rowComponent, Object valueKey,
072                            List<Column<T, ?>> columns, boolean redraw) {
073        T data = rowComponent.getData();
074        TableRow row = rowComponent.getElement();
075        if(row == null) {
076            // Create a new row element
077            row = new TableRow();
078            row.getElement().getStyle().setDisplay(Display.NONE);
079            row.getElement().getStyle().setProperty("height", getExpectedRowHeight() + "px");
080            row.getElement().getStyle().setProperty("maxHeight", getExpectedRowHeight() + "px");
081            row.getElement().getStyle().setProperty("minHeight", getExpectedRowHeight() + "px");
082            row.setStyleName(TableCssName.DATA_ROW);
083            rowComponent.setElement(row);
084
085            if(!dataView.getSelectionType().equals(SelectionType.NONE)) {
086                row.add(drawSelectionCell());
087            }
088
089            // Build the columns
090            int colOffset = dataView.getColumnOffset();
091            for(int c = 0; c < columns.size(); c++) {
092                int colIndex = c + colOffset;
093                Context context = new Context(rowComponent.getIndex(), colIndex, valueKey);
094                drawColumn(row, context, data, columns.get(c), colIndex, dataView.isHeaderVisible(colIndex));
095            }
096
097            if(dataView.isUseRowExpansion()) {
098                TableData expand = new TableData();
099                expand.setId("colex");
100                MaterialIcon expandIcon = new MaterialIcon();
101                expandIcon.setId("expand");
102                expandIcon.setWidth("100%");
103                expandIcon.setIconType(IconType.KEYBOARD_ARROW_DOWN);
104                expandIcon.setWaves(WavesType.LIGHT);
105                expandIcon.getElement().getStyle().setCursor(Cursor.POINTER);
106                expand.add(expandIcon);
107                row.add(expand);
108            }
109        } else {
110            if(redraw || rowComponent.isRedraw()) {
111                // Rebuild the columns
112                int colOffset = dataView.getColumnOffset();
113                for(int c = 0; c < columns.size(); c++) {
114                    int colIndex = c + colOffset;
115                    Context context = new Context(rowComponent.getIndex(), colIndex, valueKey);
116                    drawColumn(row, context, rowComponent.getData(), columns.get(c), colIndex,
117                        dataView.isHeaderVisible(colIndex));
118                }
119                rowComponent.setRedraw(false);
120            }
121        }
122
123        Scheduler.get().scheduleDeferred(() -> {
124            calculateRowHeight(rowComponent);
125        });
126        return row;
127    }
128
129    @Override
130    public TableSubHeader drawCategory(CategoryComponent category) {
131        if(category != null) {
132            TableSubHeader subHeader = category.getElement();
133            if(subHeader == null) {
134                subHeader = category.render();
135            }
136            return subHeader;
137        }
138
139        // No subheader was added
140        return null;
141    }
142
143    @Override
144    public TableRow drawCustom(Component<?> component) {
145        return new TableRow();
146    }
147
148    @Override
149    public TableData drawSelectionCell() {
150        TableData checkBox = new TableData();
151        checkBox.setId("col0");
152        checkBox.setStyleName(TableCssName.SELECTION);
153        new MaterialCheckBox(checkBox.getElement());
154        return checkBox;
155    }
156
157    @Override
158    @SuppressWarnings("unchecked")
159    public TableData drawColumn(TableRow row, Context context, T rowValue, Column<T, ?> column,
160                                int beforeIndex, boolean visible) {
161        TableData data = null;
162        if(row != null && rowValue != null) {
163            data = row.getColumn(beforeIndex);
164            if(data == null) {
165                data = new TableData();
166                row.insert(data, beforeIndex);
167            } else {
168                data.clear();
169            }
170
171            Div wrapper = new Div();
172
173            // Render the column cell
174            if(column instanceof WidgetColumn) {
175                wrapper.setStyleName(TableCssName.WIDGET_CELL);
176                wrapper.add(((WidgetColumn) column).render(context, rowValue));
177            } else {
178                SafeHtmlBuilder sb = new SafeHtmlBuilder();
179                column.render(context, rowValue, sb);
180                wrapper.getElement().setInnerHTML(sb.toSafeHtml().asString());
181                wrapper.setStyleName(TableCssName.CELL);
182            }
183
184            data.add(wrapper);
185
186            data.setId("col" + beforeIndex);
187            data.setDataTitle(column.getName());
188            HideOn hideOn = column.getHideOn();
189            if(hideOn != null) {
190                data.setHideOn(hideOn);
191            }
192            TextAlign textAlign = column.getTextAlign();
193            if(textAlign != null) {
194                data.setTextAlign(textAlign);
195            }
196            if(column.isNumeric()) {
197                data.addStyleName(TableCssName.NUMERIC);
198            }
199
200            // Hide if defined as not visible
201            // This can be the case when a header is toggled off.
202            if(!visible) {
203                data.$this().hide();
204            }
205        }
206        return data;
207    }
208
209    @Override
210    public TableHeader drawColumnHeader(Column<T, ?> column, String header, int index) {
211        MaterialIcon sortIcon = new MaterialIcon();
212        sortIcon.setIconSize(sortIconSize);
213
214        TableHeader th = new TableHeader(sortIcon);
215        th.setId("col" + index);
216        th.setHeader(header);
217        HideOn hideOn = column.getHideOn();
218        if(hideOn != null) {
219            th.setHideOn(hideOn);
220        }
221        TextAlign textAlign = column.getTextAlign();
222        if(textAlign != null) {
223            th.setTextAlign(textAlign);
224        }
225        if(column.isNumeric()) {
226            th.addStyleName(TableCssName.NUMERIC);
227        }
228        th.setVisible(true);
229        return th;
230    }
231
232    @Override
233    public void drawSortIcon(TableHeader th, SortContext<T> sortContext) {
234        if(sortContext.getSortDir().equals(SortDir.ASC)) {
235            th.getSortIcon().setIconType(sortAscIcon);
236        } else {
237            th.getSortIcon().setIconType(sortDescIcon);
238        }
239    }
240
241    @Override
242    public int getExpectedRowHeight() {
243        return expectedRowHeight;
244    }
245
246    @Override
247    public void setExpectedRowHeight(int expectedRowHeight) {
248        if(expectedRowHeight < 33) {
249            logger.warning("Expected row height must be 33px or higher, setting row height to 33px.");
250        }
251        this.expectedRowHeight = expectedRowHeight;
252    }
253
254    @Override
255    public void calculateRowHeight(RowComponent<T> row) {
256        TableRow element = row.getElement();
257        if(element != null) {
258            int rowHeight = element.$this().outerHeight(true);
259            if (rowHeight > 0 && rowHeight != calculatedRowHeight) {
260                calculatedRowHeight = rowHeight;
261            }
262        }
263    }
264
265    @Override
266    public int getCalculatedRowHeight() {
267        return calculatedRowHeight;
268    }
