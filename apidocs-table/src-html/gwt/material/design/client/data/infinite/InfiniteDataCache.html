023import java.util.ArrayList;
024import java.util.List;
025
026/**
027 * A Data Source for the {@link InfiniteDataView}, supports asynchronous calls
028 * expecting that {@link InfiniteDataView#loaded(int, List, int, boolean)} is called when ready.
029 * <br><br>
030 * <b>Note</b> that if {@link InfiniteDataView#loaded(int, List, int, boolean)} is not called
031 * the data will not be populated.
032 *
033 * @author Ben Dol
034 */
035public class InfiniteDataCache<T> extends ArrayList<T> {
036
037    public List<T> getCache(int startIndex, int viewSize) {
038        int size = size();
039        int end = (startIndex + viewSize);
040
041        List<T> cache = new ArrayList<>();
042        if(startIndex < size && end <= size) {
043            for(int i = startIndex; i < end; i++) {
044                T data = get(i);
045                if(data != null) {
046                    cache.add(data);
047                } else {
048                    // If any data record is null we
049                    // need to invoke a full data request
050                    cache.clear();
051                    break;
052                }
053            }
054        }
055        return cache;
056    }
057
058    public void addCache(int startIndex, List<T> data) {
059        // Create placeholders up to the specified index.
060        int cacheOffset = Math.max(0, startIndex - size());
061        for (int i = 0; i < cacheOffset; i++) {
062            add(null);
063        }
064
065        // Insert the new values into the data array.
066        for (int i = startIndex; i < startIndex + data.size(); i++) {
067            T value = data.get(i - startIndex);
068            if (i < size()) {
069                set(i, value);
070            } else {
071                add(value);
072            }
073        }
074    }
075}
