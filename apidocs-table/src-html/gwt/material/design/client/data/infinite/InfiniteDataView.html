054/**
055 * The InfiniteDataView is designed for lazy loading data, while using minimal DOM row elements.<br/>
056 * The data source will invoke a load when the view index changes on scrolling.<br/>
057 * <br/>
058 * <b>How to use:</b>
059 * <ul>
060 *   <li>View size can be configured manually or can be set to dynamic view using {@link #DYNAMIC_VIEW}.</li>
061 *   <li>Provide a valid {@link DataSource} implementation.</li>
062 * </ul>
063 *
064 * @author Ben Dol
065 */
066public class InfiniteDataView<T> extends AbstractDataView<T> {
067

150    @Override
151    public void setup(TableScaffolding scaffolding) throws Exception {
152        super.setup(scaffolding);
153
154        dynamicView = viewSize == DYNAMIC_VIEW;
155        if(dynamicView) {
156            setVisibleRange(0, getVisibleRowCapacity());
157            setViewSize(range.getLength());
158        }
159
160        JQueryElement topWrapper = $("<div>");
161        bufferTop = $("<div class='bufferTop'>");
162        topWrapper.append(bufferTop);
163        tbody.insert(new MaterialWidget(topWrapper.asElement()), 0);
164
165        bufferBottom = $("<div class='bufferBottom'>");
166        tableBody.append(bufferBottom);
167
168        container.off(TableEvents.CATEGORY_OPENED);
169        container.on(TableEvents.CATEGORY_OPENED, (e, category) -> {
170            dataCache.clear();
171            updateRows(viewIndex, true);
172            forceScroll = true;
173            return true;
174        });
175
176        container.off(TableEvents.CATEGORY_CLOSED);
177        container.on(TableEvents.CATEGORY_CLOSED, (e, category) -> {
178            dataCache.clear();
179            updateRows(viewIndex, true);
180            forceScroll = true;
181            return true;
182        });
183
184        container.off(TableEvents.ROW_SELECT);
185        container.on(TableEvents.ROW_SELECT, new EventFunc3<T, Element, Boolean>() {
186            @Override
187            public Object call(Event e, T model, Element element, Boolean selected) {
188                if(selected) {
189                    if(!selectedModels.contains(model)) {
190                        selectedModels.add(model);
191                    }
192                } else {
193                    selectedModels.remove(model);
194                }
195                return true;
196            }
197        });
198
199        container.off(TableEvents.SELECT_ALL);
200        container.on(TableEvents.SELECT_ALL, new EventFunc3<List<T>, List<Element>, Boolean>() {
201            @Override
202            public Object call(Event e, List<T> models, List<Element> elements, Boolean selected) {
203                for(T model : models) {
204                    if (selected) {
205                        if (!selectedModels.contains(model)) {
206                            selectedModels.add(model);
207                        }
208                    } else {
209                        selectedModels.remove(model);
210                    }
211                }
212                return true;
213            }
214        });
215
216        // Setup the scroll event handlers
217        JQueryExtension.$(tableBody).scrollY(id, (e, scroll) ->  onScrollY());
218    }

220    @Override
221    public void render(Components<Component<?>> components) {
222        int calcRowHeight = getCalculatedRowHeight();
223        int topHeight = loaderIndex * calcRowHeight;
224        bufferTop.height(topHeight);
225
226        int categoryMod = isUseCategories() ? categories.size() : 0;
227        int bottomHeight = ((totalRows + categoryMod) * calcRowHeight) - (topHeight - calcRowHeight);
228        bufferBottom.height(bottomHeight);
229
230        super.render(components);
231    }

233    @Override
234    public void renderRows(Components<RowComponent<T>> rows) {
235        int prevCategories = categories.size();
236        super.renderRows(rows);
237
238        if(isUseCategories()) {
239            // Update the view size to accommodate the new categories
240            int newCatCount = categories.size() - prevCategories;
241            if(newCatCount != 0) {
242                setVisibleRange(viewIndex, viewSize - newCatCount);
243                setViewSize(range.getLength());
244            }
245
246            // show all the categories
247            List<CategoryComponent> lastHidden = new ArrayList<>();
248            for (CategoryComponent category : categories) {
249                if (category.isRendered()) {
250                    category.getElement().setVisible(true);
251                }
252
253                boolean hidden = false;
254                if (isCategoryEmpty(category)) {
255                    Range range = getVisibleRange();
256                    int reach = range.getStart() + range.getLength();
257
258                    if (reach < getTotalRows()) {
259                        if (category.isRendered()) {
260                            category.getElement().setVisible(false);
261                        }
262                        lastHidden.add(category);
263                        hidden = true;
264                    }
265                }
266
267                if (!hidden) {
268                    // Reshow the previously hidden categories
269                    // This is because we have found a valid category
270                    // after these were hidden, implying valid data.
271                    for (CategoryComponent hiddenCategory : lastHidden) {
272                        if (hiddenCategory.isRendered()) {
273                            hiddenCategory.getElement().setVisible(true);
274                        }
275                    }
276                }
277            }
278
279            // hide passed empty categories
280            for (CategoryComponent category : getPassedCategories()) {
281                if (category.isRendered()) {
282                    category.getElement().setVisible(false);
283                }
284            }
285
286            subheaderLib.recalculate(true);
287        }
288    }

294    @Override
295    public void refreshView() {
296        super.refreshView();
297        int rangeStart = range.getStart();
298        setVisibleRange(rangeStart, dynamicView ? getVisibleRowCapacity() : viewSize);
299        setViewSize(range.getLength());
300        updateRows(viewIndex, true);
301        forceScroll = true;
302    }

304    @Override
305    public void setDataSource(final DataSource<T> dataSource) {
306        if(!dataSource.useRemoteSort()) {
307            logger.warning("It is recommended that an InfiniteDataView only use a remote sorting " +
308                "DataSource, this will be set for you.");
309
310            super.setDataSource(new DataSource<T>() {
311                @Override
312                public void load(LoadConfig<T> loadConfig, LoadCallback<T> callback) {
313                    dataSource.load(loadConfig, callback);
314                }
315                @Override
316                public boolean useRemoteSort() {
317                    return true;
318                }
319            });
320        } else {
321            super.setDataSource(dataSource);
322        }
323    }

333    public double getVisibleHeight() {
334        // We only want to account for row space.
335        return tableBody.height() - topPanel.height() - headerRow.$this().height();
336    }

338    protected Object onScrollY() {
339        if(!rendering) {
340            int index = (int) Math.ceil(tableBody.scrollTop() / getCalculatedRowHeight());
341            if(index == 0 || index != viewIndex) {
342                updateRows(index, false);
343            }
344        }
345        return true;
346    }
347
348    protected void updateRows(int newIndex, boolean reload) {
349        // the number of rows visible within the grid's viewport
350        viewIndex = Math.min(newIndex, Math.max(0, totalRows - viewSize));
351        requestData(viewIndex, !reload);
352    }
353
354    protected boolean requestData(int index, boolean checkCache) {
355        if(loading) {
356            // Avoid loading again before the last load
357            return false;
358        }
359        logger.finest("requestData() offset: " + index + ", viewSize: " + viewSize);
360        loaderIndex = Math.max(0, index - indexOffset);
361        loaderSize = viewSize + indexOffset;
362        if (loaderTask == null) {
363            loaderTask = new InterruptibleTask() {
364                @Override
365                public void onExecute() {
366                    doLoad();
367                }
368            };
369        }
370
371        if(checkCache) {
372            List<T> cachedData = dataCache.getCache(loaderIndex, loaderSize);
373            if (!cachedData.isEmpty()) {
374                // Found in the cache
375                loaderCache = cachedData;
376            }
377        }
378
379        loaderTask.delay(loaderDelay);
380        return true;
381    }
382
383    protected void doLoad() {
384        loading = true;
385        // Check if the data was found in the cache
386        if(loaderCache != null && !loaderCache.isEmpty()) {
387            loaded(loaderIndex, loaderCache, false);
388            loaderCache.clear();
389            loaderCache = null;
390        } else {
391            Scheduler.get().scheduleFinally(() -> display.setLoadMask(true));
392
393            final InfiniteDataView<T> dataView = this;
394
395            dataSource.load(new LoadConfig<T>() {
396                @Override
397                public int getOffset() {
398                    return loaderIndex;
399                }
400                @Override
401                public int getLimit() {
402                    return loaderSize;
403                }
404                @Override
405                public SortContext<T> getSortContext() {
406                    return dataView.getSortContext();
407                }
408                @Override
409                public List<CategoryComponent> getOpenCategories() {
410                    return dataView.getOpenCategories();
411                }
412            }, new LoadCallback<T>() {
413                @Override
414                public void onSuccess(LoadResult<T> loadResult) {
415                    dataView.loaded(loadResult.getOffset(), loadResult.getData(), loadResult.getTotalLength(), loadResult.isCacheData());
416                }
417                @Override
418                public void onFailure(Throwable caught) {
419                    GWT.log("Load failure", caught);
420                    //TODO: What we need to do on failure? Maybe clear table?
421                }
422            });
423        }
424    }

426    @Override
427    public void setLoadMask(boolean loadMask) {
428        if(loadMask || !forceScroll) {
429            super.setLoadMask(loadMask);
430
431            // Ensure the mask element uses max height
432            Scheduler.get().scheduleDeferred(() -> {
433                if (loadMask && maskElement != null) {
434                    maskElement.height(bufferBottom.outerHeight(true) + bufferTop.outerHeight(true)
435                        + tableBody.outerHeight(true) + 1000 + "px");
436                }
437            });
438        }
439    }

453    /**
454     * With infinite data loading it is often required to
455     * change the total rows, upon loading of new data.
456     * See {@link #loaded(int, List)} for standard use.
457     *
458     * @param startIndex the new start index
459     * @param data the new list of data loaded
460     * @param totalRows the new total row count
461     */
462    public void loaded(int startIndex, List<T> data, int totalRows, boolean cacheData) {
463        setTotalRows(totalRows);
464        setVisibleRange(startIndex, loaderSize);
465
466        if(cacheData) {
467            dataCache.addCache(startIndex, data);
468        }
469        super.loaded(startIndex, data);
470        loading = false;
471
472        if(forceScroll) {
473            forceScroll = false;
474            updateRows((int) Math.ceil(tableBody.scrollTop() / getCalculatedRowHeight()), false);
475        }
476
477        // Ensure selection persistence
478        for(T model : selectedModels) {
479            Element row = getRowElementByModel(model);
480            if(row != null) {
481                selectRow(row, false);
482            }
483        }
484    }

486    /**
487     * Returns the total number of rows that are visible given
488     * the current grid height.
489     */
490    public int getVisibleRowCapacity() {
491        int rh = getCalculatedRowHeight();
492        double visibleHeight = getVisibleHeight();
493        int rows = (int) ((visibleHeight < 1) ? 0 : Math.floor(visibleHeight / rh));
494
495        int calcHeight = rh * rows;
496
497        while (calcHeight < visibleHeight) {
498            rows++;
499            calcHeight = rh * rows;
500        }
501
502        logger.finest("row height: " + rh + " visibleHeight: " + visibleHeight + " visible rows: "
503            + rows + " calcHeight: " + calcHeight);
504        return rows + (isUseCategories() ? categories.size() : 1);
505    }

532    protected List<CategoryComponent> getPassedCategories() {
533        List<CategoryComponent> passed = new ArrayList<>();
534        int scrollTop = tableBody.scrollTop();
535        for(CategoryComponent category : categories) {
536            if(isCategoryEmpty(category) && scrollTop > (getRowHeight() + thead.$this().height())) {
537                passed.add(category);
538            } else {
539                // Hit the current category
540                return passed;
541            }
542        }
543        // No categories are populated.
544        return new ArrayList<>();
545    }
