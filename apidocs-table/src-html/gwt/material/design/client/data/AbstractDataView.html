173    @Override
174    public void render(Components<Component<?>> components) {
175        // Clear the current row components
176        // This does not clear the rows DOM elements
177        this.rows.clearComponents();
178
179        // Render the new components
180        for(Component<?> component : components) {
181            renderComponent(component);
182        }
183
184        redraw = false;
185        prepareRows();
186
187        // Reset category indexes and row counts
188        if(isUseCategories()) {
189            for (CategoryComponent category : categories) {
190                category.setCurrentIndex(-1);
191                category.setRowCount(-1);
192            }
193        }
194
195        if(!components.isEmpty()) {
196            // Remove the last attach handler
197            if(attachHandler != null) {
198                attachHandler.removeHandler();
199            }
200            // When the last component has been rendered we
201            // will set the rendering flag to false.
202            // This can be improved later.
203            Scheduler.get().scheduleDeferred(() -> {
204                Component<?> component = components.get(components.size() - 1);
205                Widget componentWidget = component.getElement();
206                if (componentWidget == null || componentWidget.isAttached()) {
207                    rendering = false;
208                } else {
209                    attachHandler = componentWidget.addAttachHandler(event -> rendering = false);
210                }
211            });
212        } else {
213            rendering = false;
214        }
215    }

217    /**
218     * Compile list of {@link RowComponent}'s and invoke a render.
219     * Controls the building of category components and custom components.
220     * Which then {@link #render(Components)} is invoked to perform DOM render.
221     *
222     * Rows which are already rendered are reprocessed based on their equals method.
223     * If the data is equal to the row in the same index it use the existing row.
224     *
225     * @param rows list of rows to be rendered against the existing rows.
226     */
227    protected void renderRows(Components<RowComponent<T>> rows) {
228        rendering = true;
229        Range visibleRange = getVisibleRange();
230
231        // Check if we need to redraw categories
232        if(redrawCategories) {
233            redrawCategories = false;
234
235            // When we perform a category redraw we have
236            // to clear the row elements also.
237            this.rows.clearElements();
238
239            if (isUseCategories()) {
240                List<CategoryComponent> openCategories = getOpenCategories();
241                categories.clearElements();
242
243                for (CategoryComponent category : categories) {
244                    // Re-render the category component
245                    renderComponent(category);
246
247                    if (openCategories.contains(category) && category.isRendered()) {
248                        subheaderLib.open(category.getElement().$this());
249                    }
250                }
251            } else {
252                categories.clearElements();
253            }
254        }
255
256        // The linear component list.
257        // This component list will be the rendering
258        // blueprint for the current view, so the sequence
259        // is compiled according to the component generation.
260        Components<Component<?>> components = new Components<>(visibleRange.getLength());
261
262        int index = 0;
263        for(RowComponent<T> row : rows) {
264            if (components.isFull()) {
265                break; // Component stack is full, break
266            }
267
268            if (isUseCategories()) {
269                CategoryComponent category = getCategory(row.getDataCategory());
270                if(category == null && !hasCategory(row.getDataCategory())) {
271                    category = buildCategoryComponent(row);
272                }
273
274                if(category != null) {
275                    // Create or assign the category
276                    if (!categories.contains(category)) {
277                        categories.add(category);
278                    }
279
280                    if (!components.contains(category)) {
281                        //components.add(category);
282
283                        if (category.isRendered()) {
284                            category.getElement().setVisible(true);
285                        }
286                    }
287                }
288            }
289
290            // Do we have an existing row to use
291            if (index < this.rows.size()) {
292                RowComponent<T> existingRow = this.rows.get(index);
293                if(existingRow != null) {
294                    // Replace the rows element with the
295                    // existing indexes element.
296                    row.setElement(existingRow.getElement());
297                    row.setRedraw(true);
298
299                    // Rebuild the rows custom components
300                    //existingRow.destroyChildren();
301                    //buildCustomComponents(existingRow);
302                }
303            }
304            row.setIndex(index++);
305            components.add(row);
306        }
307
308        // Render the component stack
309        render(components);
310    }

312    @SuppressWarnings("unchecked")
313    protected void renderComponent(Component<?> component) {
314        if(component != null) {
315            TableRow row;
316            int index = -1;
317
318            if(component instanceof RowComponent) {
319                RowComponent<T> rowComponent = (RowComponent<T>)component;
320
321                // Check if the row has a category
322                // Categories have been rendered before the rows
323                CategoryComponent category = null;
324                String rowCategory = rowComponent.getDataCategory();
325                if(isUseCategories()){
326                    category = getCategory(rowCategory);
327
328                    // Ensure the category exists and is rendered
329                    if(category != null && !category.isRendered()) {
330                        renderComponent(category);
331                    }
332                }
333                T data = rowComponent.getData();
334
335                // Draw the table row
336                row = renderer.drawRow(this, rowComponent, getValueKey(data), columns, redraw);
337
338                if(row != null) {
339                    if(category != null){
340                        if(categories.size() > 1) {
341                            int categoryIndex = category.getCurrentIndex();
342                            if(categoryIndex == -1) {
343                                categoryIndex = tbody.getWidgetIndex(category.getElement());
344                                category.setCurrentIndex(categoryIndex);
345                            }
346
347                            int categoryCount = category.getRowCount() + 1;
348                            category.setRowCount(categoryCount);
349
350                            // Calculate the rows index
351                            index = categoryIndex + categoryCount;
352                        }
353
354                        // Check the display of the row
355                        TableSubHeader element = category.getElement();
356                        if (element != null && element.isOpen()) {
357                            row.getElement().getStyle().clearDisplay();
358                        }
359                    } else {
360                        // Not using categories
361                        row.getElement().getStyle().clearDisplay();
362                    }
363
364                    rows.add(rowComponent);
365                }
366            } else if(component instanceof CategoryComponent) {
367                row = bindCategoryEvents(renderer.drawCategory((CategoryComponent)component));
368            } else {
369                row = renderer.drawCustom(component);
370            }
371
372            if(row != null) {
373                if(row.getParent() == null) {
374                    if (index < 0) {
375                        tbody.add(row);
376                    } else {
377                        tbody.insert(row, index + 1);
378                    }
379                } else {
380                    // TODO: calculate row element repositioning.
381                    // This will only apply if its index is different
382                    // to the new index generated based on the category.
383                }
384            } else {
385                logger.warning("Attempted to add a null TableRow to tbody, the row was ignored.");
386            }
387
388            // Render the components children
389            /*for(Component<?> child : component.getChildren()) {
390                renderComponent(child);
391            }*/
392        }
393    }

427    public void setup(TableScaffolding scaffolding) throws Exception {
428        try {
429            container = $(getContainer());
430            table = scaffolding.getTable();
431            tableBody = $(scaffolding.getTableBody());
432            topPanel = $(scaffolding.getTopPanel());
433            tbody = table.getBody();
434            thead = table.getHead();
435            $table = table.getJsElement();
436
437            headerRow = new TableRow();
438            thead.add(headerRow);
439
440            // Create progress widget
441            progressWidget = new MaterialProgress();
442            progressWidget.setTop(0);
443            progressWidget.setGwtDisplay(Display.NONE);
444            TableRow progressRow = new TableRow();
445            progressRow.addStyleName(TableCssName.STICKYEXCLUDE);
446            progressRow.setHeight("3px");
447            TableData progressTd = new TableData();
448            progressTd.getElement().setAttribute("colspan", "999");
449            progressTd.setPadding(0);
450            progressTd.setHeight("0px");
451            progressTd.add(progressWidget);
452            progressRow.add(progressTd);
453            thead.add(progressRow);
454
455            if(useRowExpansion) {
456                // Add the expand header
457                TableHeader expandHeader = new TableHeader();
458                expandHeader.setStyleName(TableCssName.COLEX);
459                addHeader(0, expandHeader, null);
460            }
461
462            if(!selectionType.equals(SelectionType.NONE)) {
463                setupHeaderSelectionBox();
464
465                if(selectionType.equals(SelectionType.MULTIPLE)) {
466                    setupShiftDetection();
467                }
468            }
469
470            // Setup the sticky header bar
471            if (useStickyHeader) {
472                setupStickyHeader();
473            }
474
475            // Setup the subheaders for categories
476            setupSubHeaders();
477
478            // Setup the resize event handlers
479            tableBody.on("resize." + id, e -> {
480                refreshView();
481                return true;
482            });
483
484            // We will check the window resize just in case
485            // it has updated the view size of the data view.
486            $(window()).on("resize." + id, e -> {
487                refreshView();
488                return true;
489            });
490
491            setup = true;
492        }
493        catch (Exception ex) {
494            logger.log(Level.SEVERE, "Problem setting up the DataView.", ex);
495            throw ex;
496        }
497    }

509    /**
510     * Prepare all row specific functionality.
511     */
512    protected void prepareRows() {
513        JQueryElement rows = $table.find("tr.data-row");
514        rows.off("." + id);
515
516        if(!selectionType.equals(SelectionType.NONE)) {
517            // Select row click bind
518            // This will also update the check status of check all input.
519            rows.on("tap." + id + " click." + id, (e, o) -> {
520                if (!selectionType.equals(SelectionType.NONE)) {
521                    Element row = $(e.getCurrentTarget()).asElement();
522                    int rowIndex = getRowIndexByElement(row);
523                    if (selectionType.equals(SelectionType.MULTIPLE) && shiftDown) {
524                        if (lastSelected < rowIndex) {
525                            // Increment
526                            for (int i = lastSelected; i < rowIndex; i++) {
527                                if (i < getVisibleItemCount()) {
528                                    RowComponent<T> rowComponent = this.rows.get(i);
529                                    if (rowComponent != null && rowComponent.isRendered()) {
530                                        selectRow(rowComponent.getElement().getElement(), true);
531                                    }
532                                }
533                            }
534                        } else {
535                            // Decrement
536                            for (int i = lastSelected - 1; i >= rowIndex - 1; i--) {
537                                if (i >= 0) {
538                                    RowComponent<T> rowComponent = this.rows.get(i);
539                                    if (rowComponent != null && rowComponent.isRendered()) {
540                                        selectRow(rowComponent.getElement().getElement(), true);
541                                    }
542                                }
543                            }
544                        }
545                    } else {
546                        toggleRowSelect(row);
547                    }
548                }
549                return false;
550            });
551        }
552
553        rows.on("contextmenu." + id, (e, o) -> {
554            Element row = $(e.getCurrentTarget()).asElement();
555
556            // Fire row select event
557            container.trigger(TableEvents.ROW_CONTEXTMENU, new Object[] {
558                    e, getModelByRowElement(row), row
559            });
560            return false;
561        });
562
563        rows.on("dblclick." + id, (e, o) -> {
564            Element row = $(e.getCurrentTarget()).asElement();
565
566            // Fire row select event
567            container.trigger(TableEvents.ROW_DOUBLECLICK, new Object[] {
568                    e, getModelByRowElement(row), row
569            });
570            return false;
571        });
572
573        JQueryExtension.$(rows).longpress(e -> {
574            Element row = $(e.getCurrentTarget()).asElement();
575
576            // Fire row select event
577            container.trigger(TableEvents.ROW_LONGPRESS, new Object[] {
578                    e, getModelByRowElement(row), row
579            });
580            return true;
581        }, e -> {
582            Element row = $(e.getCurrentTarget()).asElement();
583
584            // Fire row select event
585            container.trigger(TableEvents.ROW_SHORTPRESS, new Object[] {
586                    e, getModelByRowElement(row), row
587            });
588            return true;
589        }, longPressDuration);
590
591        JQueryElement expands = $table.find("i#expand");
592        expands.off("." + id);
593        if(useRowExpansion) {
594            // Expand current row extra information
595            expands.on("tap." + id + " click." + id, e -> {
596                final boolean[] recalculated = {false};
597
598                JQueryElement tr = $(e.getCurrentTarget()).parent().parent();
599                if (!tr.hasClass("disabled") && !tr.is("[disabled]")) {
600                    JQueryElement[] expansion = new JQueryElement[]{tr.next().find("td.expansion div")};
601
602                    if (expansion[0].length() < 1) {
603                        expansion[0] = $("<tr class='expansion'>" +
604                                "<td class='expansion' colspan='100%'>" +
605                                "<div>" +
606                                "<section class='overlay'>" +
607                                "<div class='progress' style='height:4px;top:-1px;'>" +
608                                "<div class='indeterminate'></div>" +
609                                "</div>" +
610                                "</section>" +
611                                "<div class='content'><br/><br/><br/></div>" +
612                                "</div>" +
613                                "</td></tr>");
614
615                        expansion[0].insertAfter(tr);
616                        expansion[0] = expansion[0].find("td.expansion div");
617                    }
618
619                    final boolean expanding = !expansion[0].hasClass("expanded");
620                    final JQueryElement expandRow = tr.next();
621                    final T model = getModelByRowElement(tr.asElement());
622
623                    expansion[0].one("transitionend webkitTransitionEnd oTransitionEnd MSTransitionEnd",
624                        (e1, param1) -> {
625                            if (!recalculated[0]) {
626                                // Recalculate subheaders
627                                subheaderLib.recalculate(true);
628                                recalculated[0] = true;
629
630                                // Apply overlay
631                                JQueryElement overlay = expandRow.find("section.overlay");
632                                overlay.height(expandRow.outerHeight(false));
633
634                                // Fire table expand event
635                                container.trigger(TableEvents.ROW_EXPANDED, new RowExpand<>(model, expandRow, expanding));
636                            }
637                            return true;
638                        });
639
640                    // Fire table expand event
641                    container.trigger(TableEvents.ROW_EXPAND, new RowExpand<>(model, expandRow, expanding));
642
643                    Scheduler.get().scheduleDeferred(() -> {
644                        expansion[0].toggleClass("expanded");
645                    });
646                }
647
648                e.stopPropagation();
649                return true;
650            });
651        }
652
653        subheaderLib.detect();
654        subheaderLib.recalculate(true);
655    }

664    protected void setupSubHeaders() {
665        if($table != null && display != null) {
666            subheaderLib = JsTableSubHeaders.newInstance(
667                    $(".table-body", getContainer()), "tr.subheader");
668
669            final JQueryElement header = $table.find("thead");
670            $(subheaderLib).on("before-recalculate", e -> {
671                boolean updateMargin = header.is(":visible") && isUseStickyHeader();
672                subheaderLib.setMarginTop(updateMargin ? header.outerHeight() : 0);
673                return true;
674            });
675
676            // Load the subheaders after binding.
677            subheaderLib.load();
678        }

685    protected boolean isWithinView(int start, int length, boolean canOverflow) {
686        int end = start + length;
687        int rangeStart = range.getStart();
688        int rangeEnd = rangeStart + range.getLength();
689        return ((canOverflow ? (end > rangeStart && start < rangeEnd) : (start >= rangeStart && end <= rangeEnd)));
690    }

692    /**
693     * Ensure that the cached data is consistent with the data size.
694     */
695    private void updateCachedData() {
696        int rangeStart = range.getStart();
697        int expectedLastIndex = Math.max(0, Math.min(range.getLength(), getRowCount() - rangeStart));
698        int lastIndex = getVisibleItemCount() - 1;
699        while (lastIndex >= expectedLastIndex) {
700            rows.remove(lastIndex);
701            lastIndex--;
702        }
703    }

737        if (count == getRowCount() && isExact == isRowCountExact()) {
738            return;
739        }
740        setRowCount(count);
741        this.isExact = isExact;
742
743        // Update the cached data.
744        updateCachedData();
745
746        // Fire the row count change event
747        container.trigger(TableEvents.ROW_COUNT_CHANGE, new Object[]{count, isExact});
748    }

760    protected void setVisibleRange(Range range, boolean clearData, boolean forceRangeChangeEvent) {
761        final int start = range.getStart();
762        final int length = range.getLength();
763        if (start < 0) {
764            throw new IllegalArgumentException("Range start cannot be less than 0");
765        }
766        if (length < 0) {
767            throw new IllegalArgumentException("Range length cannot be less than 0");
768        }
769
770        // Update the page start.
771        final int pageStart = this.range.getStart();
772        final int pageSize = this.range.getLength();
773        final boolean pageStartChanged = (pageStart != start);
774        if (pageStartChanged) {
775            // Trim the data if we aren't clearing it.
776            if (!clearData) {
777                if (start > pageStart) {
778                    int increase = start - pageStart;
779                    if (getVisibleItemCount() > increase) {
780                        // Remove the data we no longer need.
781                        for (int i = 0; i < increase; i++) {
782                            rows.remove(rows.size()-1);
783                        }
784                    }
785                } else {
786                    int decrease = pageStart - start;
787                    if ((getVisibleItemCount() > 0) && (decrease < pageSize)) {
788                        // Add null data
789                        for (int i = 0; i < decrease; i++) {
790                            rows.add(null);
791                        }
792                    }
793                }
794            }
795
796            // Update the range start
797            this.range = new Range(start, this.range.getLength());
798        }
799
800        // Update the page size
801        final boolean pageSizeChanged = (pageSize != length);
802        if (pageSizeChanged) {
803            this.range = new Range(this.range.getStart(), length);
804        }
805
806        // Clear the rows
807        if (clearData) {
808            rows.clear();
809        }
810
811        // Trim the row values if needed
812        updateCachedData();
813
814        // Update the pager and data source if the range changed
815        if (pageStartChanged || pageSizeChanged || forceRangeChangeEvent) {
816            RangeChangeEvent.fire(display, getVisibleRange());
817        }
818    }

888    protected void sort(TableHeader th, Column<T, ?> column, int index) {
889        SortContext<T> oldSortContext = this.sortContext;
890        updateSortContext(th, column);
891
892        Components<RowComponent<T>> rows = new Components<>(this.rows, RowComponent::new);
893        if(doSort(sortContext, rows)) {
894            th.addStyleName(TableCssName.SELECTED);
895
896            // Draw and apply the sort icon.
897            renderer.drawSortIcon(th, sortContext);
898
899            // Render the new sort order.
900            renderRows(rows);
901
902            container.trigger(TableEvents.SORT_COLUMN, new Object[]{sortContext, index});
903        } else {
904            // revert the sort context
905            sortContext = oldSortContext;
906        }

909    /**
910     * Perform a sort on the a set of {@link RowComponent}'s.
911     * Sorting will check for each components category and sort per category if found.
912     * @return true if the data was sorted, false if no sorting was performed.
913     */
914    protected boolean doSort(SortContext<T> sortContext, Components<RowComponent<T>> rows) {
915
916        if (dataSource != null && dataSource.useRemoteSort()) {
917            // The sorting should be handled by an external
918            // data source rather than re-ordered by the
919            // client comparator.
920            return true;
921        }
922
923        Comparator<? super RowComponent<T>> comparator = sortContext != null
924                ? sortContext.getSortColumn().getSortComparator() : null;
925        if (isUseCategories() && !categories.isEmpty()) {
926            // Split row data into categories
927            Map<String, List<RowComponent<T>>> splitMap = new HashMap<>();
928            List<RowComponent<T>> orphanRows = new ArrayList<>();
929
930            for (RowComponent<T> row : rows) {
931                if(row != null) {
932                    String category = row.getDataCategory();
933                    if(category != null) {
934                        List<RowComponent<T>> data = splitMap.get(category);
935                        if (data == null) {
936                            data = new ArrayList<>();
937                            splitMap.put(category, data);
938                        }
939                        data.add(row);
940                    } else {
941                        orphanRows.add(row);
942                    }
943                }
944            }
945
946            if(!orphanRows.isEmpty()) {
947                splitMap.put("<@orphans@>", orphanRows);
948            }
949
950            rows.clearComponents();
951            for (Map.Entry<String, List<RowComponent<T>>> entry : splitMap.entrySet()) {
952                List<RowComponent<T>> list = entry.getValue();
953                if (comparator != null) {
954                    Collections.sort(list, new DataSort<>(comparator, sortContext.getSortDir()));
955                }
956                rows.addAll(list);
957            }
958        } else {
959            if (comparator != null) {
960                Collections.sort(rows, new DataSort<>(comparator, sortContext.getSortDir()));
961            } else {
962                return false;
963            }
964        }
965        return true;
966    }

1295    @Override
1296    public void setRowData(int start, List<? extends T> values) {
1297        int length = values.size();
1298        int end = start + length;
1299
1300        // Make sure we have a valid range
1301        if(range == null) {
1302            setVisibleRange(0, length);
1303        }
1304
1305        // Current range start and end
1306        int rangeStart = range.getStart();
1307        int rangeEnd = rangeStart + range.getLength();
1308
1309        // Calculated boundary scope
1310        int boundedStart = Math.max(start, rangeStart);
1311        int boundedEnd = Math.min(end, rangeEnd);
1312
1313        if (start != rangeStart && boundedStart >= boundedEnd) {
1314            // The data is out of range for the current range.
1315            // Intentionally allow empty lists that start on the range start.
1316            return;
1317        }
1318
1319        // Merge the existing data with the new data
1320        Components<RowComponent<T>> rows = new Components<>(this.rows);
1321        for (int i = boundedStart; i < boundedEnd; i++) {
1322            RowComponent<T> newRow = buildRowComponent(values.get(i - boundedStart));
1323            if(i < rows.size()) {
1324                // Within the existing data set
1325                rows.set(i, newRow);
1326            } else {
1327                // Expanding the data set
1328                rows.add(newRow);
1329            }
1330        }
1331
1332        // Ensure sort order is applied for new rows
1333        doSort(sortContext, rows);
1334
1335        // Render the new rows
1336        renderRows(rows);
1337    }

1540    protected Map<CategoryComponent, List<RowComponent<T>>> generateRowMap(Components<RowComponent<T>> rows, boolean autoCreate) {
1541        Map<CategoryComponent, List<RowComponent<T>>> dataMap = new HashMap<>();
1542
1543        for(CategoryComponent category : categories) {
1544            dataMap.put(category, new ArrayList<>());
1545        }
1546
1547        for(RowComponent<T> row : rows) {
1548            CategoryComponent category = getCategory(row.getDataCategory());
1549            if(category == null && !hasCategory(row.getDataCategory())) {
1550                if(autoCreate) {
1551                    category = buildCategoryComponent(row);
1552                } else {
1553                    // When we are not auto creating we will
1554                    // orphan the data without a category.
1555                    category = getOrphansCategory();
1556                }
1557            }
1558
1559            if(category != null) {
1560                List<RowComponent<T>> rowList = dataMap.get(category);
1561                if(rowList == null) {
1562                    rowList = new ArrayList<>();
1563                    dataMap.put(category, rowList);
1564                }
1565                rowList.add(row);
1566            }
1567        }
1568        return dataMap;
1569    }
