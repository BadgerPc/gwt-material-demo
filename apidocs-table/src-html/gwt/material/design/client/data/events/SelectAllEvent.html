020package gwt.material.design.client.data.events;
021
022import com.google.gwt.dom.client.Element;
023import com.google.gwt.event.shared.GwtEvent;
024import gwt.material.design.client.data.DataView;
025
026import java.util.List;
027
028public class SelectAllEvent<T> extends GwtEvent<SelectAllHandler<T>> {
029
030    public final static Type<SelectAllHandler> TYPE = new Type<>();
031
032    public static <T> void fire(DataView<T> source, List<T> models, List<Element> rows, boolean selected) {
033        source.fireEvent(new SelectAllEvent<>(models, rows, selected));
034    }
035
036    private final List<T> models;
037    private final List<Element> rows;
038    private final boolean selected;
039
040    public SelectAllEvent(List<T> models, List<Element> rows, boolean selected) {
041        this.models = models;
042        this.rows = rows;
043        this.selected = selected;
044    }
045
046    public List<T> getModels() {
047        return models;
048    }
049
050    public List<Element> getRows() {
051        return rows;
052    }
053
054    public boolean isSelected() {
055        return selected;
056    }
057
058    @Override
059    @SuppressWarnings("unchecked")
060    public Type<SelectAllHandler<T>> getAssociatedType() {
061        return (Type) TYPE;
062    }
063
064    @Override
065    protected void dispatch(SelectAllHandler<T> handler) {
066        handler.onSelectAll(this);
067    }
068}
