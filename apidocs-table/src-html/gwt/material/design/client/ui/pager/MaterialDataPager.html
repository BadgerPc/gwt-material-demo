024import gwt.material.design.client.data.DataSource;
025import gwt.material.design.client.ui.table.MaterialDataTable;
026
027/**
028 * Material Data Pager - a simple pager for Material Data Table component
029 *
030 * @author kevzlou7979
031 */
032public class MaterialDataPager<T> extends MaterialDataPagerBase<T> {
033
034    private int[] rowCountOptions = new int[] {5, 10, 15};
035    private int numPages, firstRow, lastRow, totalRows;
036
037    private boolean useRowCountOptions = true;
038
039    public MaterialDataPager(MaterialDataTable<T> table, DataSource<T> dataSource) {
040        super(table, dataSource);
041    }
042
043    @Override
044    protected void onLoad() {
045        super.onLoad();
046
047        if(!useRowCountOptions) {
048            getRowsPerPagePanel().setVisible(false);
049            getNumPagePanel().setOffset("l6");
050        } else {
051            setRowCount(getRowCountOptions()[0]);
052            getRowsPerPagePanel().setVisible(true);
053        }
054
055        // Must set the total rows for the calculation of page numbers
056        setTotalRows(getDataSource().getDataSize());
057
058        getListRowsPerPage().clear();
059        for(int i : rowCountOptions) {
060            getListRowsPerPage().addItem(String.valueOf(i));
061        }
062
063        getListRowsPerPage().addValueChangeHandler(valueChangeEvent -> {
064            goToPage(Integer.parseInt(valueChangeEvent.getValue()), getCurrentPage());
065        });
066
067        getListPages().addValueChangeHandler(valueChangeEvent -> {
068            goToPage(getRowCount(), Integer.parseInt(valueChangeEvent.getValue()));
069        });
070
071        getIconNext().addClickHandler(clickEvent -> next());
072        getIconPrev().addClickHandler(clickEvent -> previous());
073        goToPage(getRowCount(), 1);
074    }
075
076    /**
077     * Update the pager properties.
078     */
079    private void goToPage(int rowCount, int currentPage) {
080
081        // Check if the row's num page is less than the current page selection
082        // If yes, then we will navigate to last page
083        if(getNumPages() < currentPage) {
084            lastPage();
085            return;
086        }
087
088        // Check if data pager is on first page, if yes then the previous button will be disabled
089        if(currentPage >= getNumPages()) {
090            getIconNext().setEnabled(false);
091            getIconPrev().setEnabled(true);
092        } else {
093            getIconNext().setEnabled(true);
094        }
095
096        // Check if data pager is on last page, if yes then the next button will be disabled.
097        if(currentPage <= 1) {
098            getIconNext().setEnabled(true);
099            getIconPrev().setEnabled(false);
100        } else {
101            getIconPrev().setEnabled(true);
102        }
103
104        // Check if data pager has only one page, if yes then the next and previous button will be disabled.
105        if(getNumPages() == 1) {
106            getIconNext().setEnabled(false);
107            getIconPrev().setEnabled(false);
108        }
109        setRowCount(rowCount);
110        setFirstRow(((Math.max(0, currentPage - 1)) * rowCount) + 1);
111        setLastRow(currentPage * rowCount);
112        updateDataTable();
113        setCurrentPage(currentPage);
114    }
115
116    public void refresh() {
117        updateDataTable();
118    }
119
120    /**
121     * Refresh and redraw the table and set the visible range with the given params.
122     */
123    private void updateDataTable() {
124        getListPages().clear();
125        for(int i = 1; i <= getNumPages(); i++) {
126            getListPages().addItem(String.valueOf(i));
127        }
128        getActionLabel().setText(getFirstRow() + "-" + getLastRow() + " of " + getTotalRows());
129        table.setVisibleRange(getFirstRow(), getRowCount());
130
131        // Check if the first row and row count  is bigger than the getTotalRows
132        if(getFirstRow() + getRowCount() > getTotalRows()) {
133            if(getRowCount() > getTotalRows()) {
134                getDataSource().load(table, getFirstRow(), getTotalRows());
135            } else {
136                getDataSource().load(table, getFirstRow(), getTotalRows() - (getFirstRow() - 1));
137            }
138        } else {
139            getDataSource().load(table, getFirstRow(), getRowCount());
140        }
141    }
142
143    /**
144     * Get the row count options.
145     * @return an array of integers for the series of options (How many rows should be displayed)
146     */
147    public int[] getRowCountOptions() {
148        return rowCountOptions;
149    }
150
151    /**
152     * Set how many rows should be displayed into the data table,
153     * a series of options (e.g 5, 10, 15) which will be added to listbox options
154     */
155    public void setRowCountOptions(int... rowCountOptions) {
156        this.rowCountOptions = rowCountOptions;
157    }
158
159    /**
160     * Get the specific selected row count list box option.
161     * @return the row count
162     */
163    public int getRowCount() {
164        if(getListRowsPerPage().getSelectedItemText() != null && isUseRowCountOptions()) {
165            return Integer.parseInt(getListRowsPerPage().getSelectedItemText());
166        }
167        return table.getVisibleRange().getLength();
168    }
169
170    /**
171     * Set directly the selected row count
172     */
173    public void setRowCount(int rowCount) {
174        getListRowsPerPage().setSelectedValue(String.valueOf(rowCount));
175    }
176
177    /**
178     * Get the current page.
179     * @return currentPage
180     */
181    public int getCurrentPage() {
182        if(getListPages().getSelectedItemText() != null) {
183            return Integer.parseInt(getListPages().getSelectedItemText());
184        }
185        return 0;
186    }
187
188    /**
189     * Set the current page.
190     */
191    public void setCurrentPage(int currentPage) {
192        getListPages().setSelectedValue(String.valueOf(currentPage));
193    }
194
195    /**
196     * Get the total number of pages.
197     */
198    public int getNumPages() {
199        if(getTotalRows() % getRowCount() > 0) {
200            return (totalRows / getRowCount()) + 1;
201        }
202        return totalRows / getRowCount();
203    }
204
205    /**
206     * Advance the starting row by 'pageSize' rows.
207     */
208    public void next() {
209        goToPage(getRowCount(), getCurrentPage() + 1);
210    }
211
212    /**
213     * Move the starting row back by 'pageSize' rows.
214     */
215    public void previous() {
216        goToPage(getRowCount(), getCurrentPage() - 1);
217    }
218
219    /**
220     * Go to the first page.
221     */
222    public void firstPage() {
223        goToPage(getRowCount(), 1);
224    }
225
226    /**
227     * Go to the last page.
228     */
229    public void lastPage() {
230        goToPage(getRowCount(), getNumPages());
231    }
232
233    public int getFirstRow() {
234        return firstRow;
235    }
236
237    public void setFirstRow(int firstRow) {
238        this.firstRow = firstRow;
239    }
240
241    public int getLastRow() {
242        if(lastRow > getTotalRows()) {
243            lastRow = getTotalRows();
244        }
245        return lastRow;
246    }
247
248    public void setLastRow(int lastRow) {
249        this.lastRow = lastRow;
250    }
251
252    public int getTotalRows() {
253        return totalRows;
254    }
255
256    public void setTotalRows(int totalRows) {
257        this.totalRows = totalRows;
258    }
259
260    public boolean isUseRowCountOptions() {
261        return useRowCountOptions;
262    }
263
264    public void setUseRowCountOptions(boolean useRowCountOptions) {
265        this.useRowCountOptions = useRowCountOptions;
266    }
267}
