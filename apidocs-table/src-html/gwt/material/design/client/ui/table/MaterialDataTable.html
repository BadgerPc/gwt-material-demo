043/**
044 * The standard Material data table with custom "outer" components such as,
045 * table icon, table title, stretch functionality, column toggling, etc.
046 *
047 * @author Ben Dol
048 */
049public class MaterialDataTable<T> extends AbstractDataTable<T> {
050
051    private JQueryElement stretchContainer;
052
053    // Interface
054    private MaterialIcon tableIcon;
055    private Span tableTitle;
056    private MaterialIcon stretchIcon;
057    private MaterialIcon columnMenuIcon;
058    private MaterialDropDown menu;
059
060    public MaterialDataTable() {
061    }
062
063    public MaterialDataTable(DataView<T> dataView) {
064        super(dataView);
065    }
066
067    public MaterialDataTable(TableScaffolding scaffolding) {
068        super(scaffolding);
069    }
070
071    public MaterialDataTable(DataView<T> dataView, TableScaffolding scaffolding) {
072        super(dataView, scaffolding);
073    }
074
075    @Override
076    public void setup(TableScaffolding scaffolding) throws Exception {
077        super.setup(scaffolding);
078
079        Panel tableBody = scaffolding.getTableBody();
080        Panel infoPanel = scaffolding.getInfoPanel();
081        Panel toolPanel = scaffolding.getToolPanel();
082
083        // table icon
084        tableIcon = new MaterialIcon(IconType.VIEW_LIST);
085        infoPanel.add(tableIcon);
086
087        // table title
088        tableTitle = new Span("Table Title");
089        tableTitle.addStyleName(TableCssName.TITLE);
090        infoPanel.add(tableTitle);
091
092        // stretch icon
093        stretchIcon = new MaterialIcon(IconType.FULLSCREEN);
094        stretchIcon.setWaves(WavesType.LIGHT);
095        stretchIcon.setCircle(true);
096        stretchIcon.setId("stretch");
097        $(stretchIcon).css("cursor", "pointer");
098        toolPanel.add(stretchIcon);
099
100        // menu icon
101        columnMenuIcon = new MaterialIcon(IconType.MORE_VERT);
102        columnMenuIcon.setHideOn(HideOn.HIDE_ON_SMALL_DOWN);
103        columnMenuIcon.setWaves(WavesType.LIGHT);
104        columnMenuIcon.setCircle(true);
105        columnMenuIcon.setId("columnToggle");
106        $(columnMenuIcon).css("cursor", "pointer");
107        toolPanel.add(columnMenuIcon);
108
109        // stretch container
110        stretchContainer = $("body");
111
112        setupToolPanel();
113        setupMenu();
114    }
115
116    protected void setupToolPanel() {
117        // Stretch click handler
118        $(scaffolding.getToolPanel()).find("i#stretch").on("click", e -> {
119            stretch();
120
121            e.preventDefault();
122            return true;
123        });
124    }
125
126    protected void setupMenu() {
127        // Setup menu checkboxes
128        // This will allow the user to toggle columns
129
130        if(menu == null) {
131            // dropdown structure
132            menu = new MaterialDropDown(columnMenuIcon);
133            scaffolding.getToolPanel().add(menu);
134
135            // Menu initialization
136            menu.setInDuration(300);
137            menu.setOutDuration(225);
138            menu.setConstrainWidth(false);
139            menu.setHover(false);
140            menu.setGutter(0);
141            menu.setBelowOrigin(false);
142            menu.setAlignment(Alignment.LEFT);
143            menu.setHideOn(HideOn.HIDE_ON_SMALL_DOWN);
144            menu.getElement().getStyle().setProperty("minWidth", "200px");
145        }
146
147        JQueryElement $menu = $(menu);
148        $menu.find("li label").off("tap click");
149        $menu.find("li label").on("tap click", e -> {
150            JQueryElement $this = $(e.getCurrentTarget());
151
152            String forBox = ((String) $this.attr("for")).replace(getViewId() + "-", "");
153            if(Js.isTrue(forBox)) {
154                JQueryElement thd = $("th#" + forBox + ",td#" + forBox, this);
155                boolean checked = $this.prev().is(":checked");
156
157                thd.each((index, el) -> {
158                    JQueryElement cell = $(el);
159                    if(checked) {
160                        cell.hide();
161                    } else {
162                        cell.show();
163                    }
164                });
165
166                // Update the sticky table header widths
167                scaffolding.getTable().getJsElement().stickyTableHeaders("updateWidth");
168
169                // Recalculate the subheader
170                getSubheaderLib().recalculate(true);
171            }
172            return true;
173        });
174
175        // Stop each menu item from closing the dropdown
176        $menu.find("li").off("touchstart click");
177        $menu.find("li").on("touchstart click", e -> {
178            e.stopPropagation();
179            return true;
180        });
181    }
182
183    @Override
184    public void insertColumn(int beforeIndex, Column<T, ?> col, String header) {
185        super.insertColumn(beforeIndex, col, header);
186
187        int index = beforeIndex + getColumnOffset();
188        String ref = getViewId() + "-col" + index;
189
190        MaterialCheckBox toggleBox = new MaterialCheckBox(new ListItem().getElement());
191        JQueryElement input = $(toggleBox).find("input");
192        input.attr("id", ref);
193
194        JQueryElement label = $(toggleBox).find("label");
195        label.text(col.getName());
196        label.attr("for", ref);
197
198        toggleBox.setValue(true);
199        menu.add(toggleBox);
200
201        // We will hide the empty header menu items
202        if(header.isEmpty()) {
203            toggleBox.setVisible(false);
204        }
205
206        setupMenu();
207        reindexToggles();
208    }
209
210    @Override
211    public void removeColumn(int colIndex) {
212        super.removeColumn(colIndex);
213
214        int index = colIndex + getColumnOffset();
215        $(menu).find("li input#col" + index).parent().remove();
216        reindexToggles();
217    }
218
219    private void reindexToggles() {
220        int colOffset = getColumnOffset();
221        $("li", menu).each((index, e) -> {
222            String ref = getViewId() + "-col" + ((Double)index + colOffset);
223
224            JQueryElement input = $(e).find("input");
225            input.attr("id", ref);
226
227            JQueryElement label = $(e).find("label");
228            label.attr("for", ref);
229        });
230    }
231
232    public void stretch() {
233        stretch(true);
234    }
235
236    public void stretch(boolean fireEvent) {
237        $this().toggleClass(TableCssName.STRETCH);
238
239        // Make sure the body doesn't display scrollbar
240        body().toggleClass(TableCssName.OVERFLOW_HIDDEN);
241
242        // Update table header widths
243        JsTableElement tableJs = scaffolding.getTable().getJsElement();
244        tableJs.stickyTableHeaders("updateWidth");
245        tableJs.stickyTableHeaders("toggleHeaders");
246
247        // Recalculate subheaders
248        getSubheaderLib().recalculate(true);
249
250        if(fireEvent) {
251            // Fire table stretch event
252            $this().trigger(TableEvents.STRETCH, new Object[]{$this().hasClass(TableCssName.STRETCH)});
253        }
254    }
255
256    public MaterialIcon getStretchIcon() {
257        return stretchIcon;
258    }
259
260    public MaterialIcon getColumnMenuIcon() {
261        return columnMenuIcon;
262    }
263
264    public MaterialIcon getTableIcon() {
265        return tableIcon;
266    }
267
268    public MaterialDropDown getMenu() {
269        return menu;
270    }
271
272    public Span getTableTitle() {
273        return tableTitle;
274    }
