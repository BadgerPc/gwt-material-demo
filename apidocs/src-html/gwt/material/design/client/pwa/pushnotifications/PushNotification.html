020package gwt.material.design.client.pwa.pushnotifications;
021
022import jsinterop.annotations.JsMethod;
023import jsinterop.annotations.JsType;
024
025import static jsinterop.annotations.JsPackage.GLOBAL;
026
027//@formatter:off
028
029/**
030 * Push Notification Core Services for Progressive Web Application
031 *
032 * @author kevzlou7979
033 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#notification">PWA Notification</a>
034 */
035//@formatter:on
036@JsType(isNative = true, namespace = GLOBAL, name = "Push")
037public class PushNotification {
038
039    /**
040     * Creates a simple push notification with title as param
041     */
042    @JsMethod
043    public static native void create(String title);
044
045    /**
046     * Creates a push notification with additional options
047     * such as icons, body, data etc.
048     */
049    @JsMethod
050    public static native void create(String title, PushNotificationOptions options);
051
052    /**
053     * Return how many notifications are currently open.
054     */
055    @JsMethod
056    public static native int count();
057
058    /**
059     * Will close / hide the Push Notification Dialog
060     */
061    @JsMethod
062    public static native void close();
063
064    /**
065     * Will close / hide the Push Notification Dialog with tag as param
066     */
067    @JsMethod
068    public static native void close(String tag);
069
070    /**
071     * Will clear all the notifications that are currently opened.
072     */
073    @JsMethod
074    public static native void clear();
075}
