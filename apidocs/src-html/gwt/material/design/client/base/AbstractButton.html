023import com.google.gwt.dom.client.Element;
024import com.google.gwt.dom.client.Style;
025import com.google.gwt.event.dom.client.ClickEvent;
026import com.google.gwt.event.dom.client.ClickHandler;
027import com.google.gwt.event.dom.client.DoubleClickEvent;
028import com.google.gwt.event.dom.client.DoubleClickHandler;
029import com.google.gwt.event.dom.client.HasAllMouseHandlers;
030import com.google.gwt.event.dom.client.HasClickHandlers;
031import com.google.gwt.event.dom.client.HasDoubleClickHandlers;
032import com.google.gwt.event.dom.client.MouseDownEvent;
033import com.google.gwt.event.dom.client.MouseDownHandler;
034import com.google.gwt.event.dom.client.MouseMoveEvent;
035import com.google.gwt.event.dom.client.MouseMoveHandler;
036import com.google.gwt.event.dom.client.MouseOutEvent;
037import com.google.gwt.event.dom.client.MouseOutHandler;
038import com.google.gwt.event.dom.client.MouseOverEvent;
039import com.google.gwt.event.dom.client.MouseOverHandler;
040import com.google.gwt.event.dom.client.MouseUpEvent;
041import com.google.gwt.event.dom.client.MouseUpHandler;
042import com.google.gwt.event.dom.client.MouseWheelEvent;
043import com.google.gwt.event.dom.client.MouseWheelHandler;
044import com.google.gwt.event.shared.HandlerRegistration;
045import com.google.gwt.user.client.History;
046import gwt.material.design.client.base.mixin.ActivatesMixin;
047import gwt.material.design.client.base.mixin.CssTypeMixin;
048import gwt.material.design.client.constants.ButtonSize;
049import gwt.material.design.client.constants.ButtonType;
050import gwt.material.design.client.constants.WavesType;
051import gwt.material.design.client.ui.html.Span;
052
053/**
054 * @author Ben Dol
055 */
056public abstract class AbstractButton extends MaterialWidget implements HasHref, HasGrid, HasActivates,
057        HasTargetHistoryToken, HasType<ButtonType>, HasClickHandlers, HasAllMouseHandlers,
058        HasDoubleClickHandlers {
059
060    private final ActivatesMixin<AbstractButton> activatesMixin = new ActivatesMixin<>(this);
061    private final CssTypeMixin<ButtonType, AbstractButton> cssTypeMixin = new CssTypeMixin<>(this);
062
063    private Span span = new Span();
064    private ButtonSize size;
065
066    private String targetHistoryToken;
067
068    /** Creates button with RAISED type.
069     */
070    protected AbstractButton() {
071        setElement(createElement());
072
073        getElement().getStyle().setCursor(Style.Cursor.POINTER);
074    }
075
076    protected AbstractButton(String... initialClass) {
077        this();
078        setInitialClasses(initialClass);
079    }
080
081    protected AbstractButton(String text, String bgColor, WavesType waves) {
082        this(null, text, bgColor);
083        setWaves(waves);
084    }
085
086    protected AbstractButton(final ButtonType type, String text, String bgColor, WavesType waves) {
087        this(type, text, bgColor);
088        setWaves(waves);
089    }
090
091    protected AbstractButton(final ButtonType type, String text, String bgColor) {
092        this(type, text);
093        setBackgroundColor(bgColor);
094    }
095
096    protected AbstractButton(final ButtonType type, String text) {
097        this(type);
098        setText(text);
099    }
100
101    protected AbstractButton(final ButtonType type) {
102        this();
103        setType(type);
104    }
105
106    protected abstract Element createElement();
107
108    @Override
109    public void setHref(String href) {
110        getElement().setAttribute("href", href);
111    }
112
113    @Override
114    public String getHref() {
115        return getElement().getAttribute("href");
116    }
117
118    @Override
119    public void setTarget(String target) {
120        getElement().setAttribute("target", target);
121    }
122
123    @Override
124    public String getTarget() {
125        return getElement().getAttribute("target");
126    }
127
128    @Override
129    public void setActivates(String activates) {
130        removeStyleName(getActivates() + " dropdown-button");
131        activatesMixin.setActivates(activates);
132        addStyleName(activates + " dropdown-button");
133    }
134
135    @Override
136    public String getActivates() {
137        return activatesMixin.getActivates();
138    }
139
140    @Override
141    public void setType(ButtonType type) {
142        cssTypeMixin.setType(type);
143    }
144
145    @Override
146    public ButtonType getType() {
147        return cssTypeMixin.getType();
148    }
149
150    public void setSize(ButtonSize size) {
151        if(this.size != null) {
152            removeStyleName(this.size.getCssName());
153        }
154        this.size = size;
155
156        if(size != null) {
157            addStyleName(size.getCssName());
158        }
159    }
160
161    public ButtonSize getSize() {
162        return size;
163    }
164
165    public String getText() {
166        return span.getText();
167    }
168
169    public void setText(String text) {
170        span.setText(text);
171        add(span);
172    }
173
174    /**
175     * Set the target history token for the widget. Note, that you should use either
176     * {@link #setTargetHistoryToken(String)}or {@link #setHref(String)}, but not both as
177     * {@link #setHref(String)} resets the target history token.
178     * @param targetHistoryToken String target history token of the widget
179     */
180    @Override
181    public void setTargetHistoryToken(final String targetHistoryToken) {
182        this.targetHistoryToken = targetHistoryToken;
183        if (targetHistoryToken != null) {
184            setHref("#" + History.encodeHistoryToken(targetHistoryToken));
185        }
186    }
187
188    /**
189     * Get the target history token for the widget. May return {@code null} if no
190     * history token has been set or if it has been reset by {@link #setHref(String)}.
191     * @return String the widget's target history token.
192     */
193    @Override
194    public String getTargetHistoryToken() {
195        return targetHistoryToken;
196    }
197
198    @Override
199    public HandlerRegistration addClickHandler(final ClickHandler handler) {
200        return addDomHandler(new ClickHandler() {
201            @Override
202            public void onClick(ClickEvent event) {
203                if(isEnabled()){
204                    handler.onClick(event);
205                }
206            }
207        }, ClickEvent.getType());
208    }
209
210    @Override
211    public HandlerRegistration addMouseDownHandler(final MouseDownHandler handler) {
212        return addDomHandler(new MouseDownHandler() {
213            @Override
214            public void onMouseDown(MouseDownEvent event) {
215                if(isEnabled()){
216                    handler.onMouseDown(event);
217                }
218            }
219        }, MouseDownEvent.getType());
220    }
221
222    @Override
223    public HandlerRegistration addMouseMoveHandler(final MouseMoveHandler handler) {
224        return addDomHandler(new MouseMoveHandler() {
225            @Override
226            public void onMouseMove(MouseMoveEvent event) {
227                if(isEnabled()){
228                    handler.onMouseMove(event);
229                }
230            }
231        }, MouseMoveEvent.getType());
232    }
233
234    @Override
235    public HandlerRegistration addMouseOutHandler(final MouseOutHandler handler) {
236        return addDomHandler(new MouseOutHandler() {
237            @Override
238            public void onMouseOut(MouseOutEvent event) {
239                if(isEnabled()) {
240                    handler.onMouseOut(event);
241                }
242            }
243        }, MouseOutEvent.getType());
244    }
245
246    @Override
247    public HandlerRegistration addMouseOverHandler(final MouseOverHandler handler) {
248        return addDomHandler(new MouseOverHandler() {
249            @Override
250            public void onMouseOver(MouseOverEvent event) {
251                if(isEnabled()) {
252                    handler.onMouseOver(event);
253                }
254            }
255        }, MouseOverEvent.getType());
256    }
257
258    @Override
259    public HandlerRegistration addMouseUpHandler(final MouseUpHandler handler) {
260        return addDomHandler(new MouseUpHandler() {
261            @Override
262            public void onMouseUp(MouseUpEvent event) {
263                if(isEnabled()) {
264                    handler.onMouseUp(event);
265                }
266            }
267        }, MouseUpEvent.getType());
268    }
269
270    @Override
271    public HandlerRegistration addMouseWheelHandler(final MouseWheelHandler handler) {
272        return addDomHandler(new MouseWheelHandler() {
273            @Override
274            public void onMouseWheel(MouseWheelEvent event) {
275                if(isEnabled()) {
276                    handler.onMouseWheel(event);
277                }
278            }
279        }, MouseWheelEvent.getType());
280    }
281
282    @Override
283    public HandlerRegistration addDoubleClickHandler(final DoubleClickHandler handler) {
284        return addDomHandler(new DoubleClickHandler() {
285            @Override
286            public void onDoubleClick(DoubleClickEvent event) {
287                if(isEnabled()) {
288                    handler.onDoubleClick(event);
289                }
290            }
291        }, DoubleClickEvent.getType());
292    }
293}
