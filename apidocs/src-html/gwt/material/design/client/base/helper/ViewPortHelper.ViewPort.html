020package gwt.material.design.client.base.helper;
021
022import gwt.material.design.client.js.Window;
023
024public class ViewPortHelper {
025
026    public enum ViewPort {
027        MOBILE_SMALL("(max-width: 320px)"),
028        MOBILE_MEDIUM("(min-width: 321px) and (max-width: 375px)"),
029        MOBILE_LARGE("(min-width: 376px) and (max-width: 425px)"),
030        TABLET("(min-width: 426px) and (max-width: 768px)"),
031        LAPTOP("(min-width: 769px) and (max-width: 1024px)"),
032        LAPTOP_LARGE("(min-width: 1025px) and (max-width: 1440px)"),
033        LAPTOP_4K("(min-width: 1441px) and (max-width: 2560px)");
034
035        private String mediaQuery;
036
037        ViewPort(String mediaQuery) {
038            this.mediaQuery = mediaQuery;
039        }
040
041        public String getMediaQuery() {
042            return mediaQuery;
043        }
044
045        public void setMediaQuery(String mediaQuery) {
046            this.mediaQuery = mediaQuery;
047        }
048    }
049
050    public static boolean isMobile() {
051        return matchMedia(ViewPort.MOBILE_SMALL) || matchMedia(ViewPort.MOBILE_MEDIUM ) || matchMedia(ViewPort.MOBILE_LARGE);
052    }
053
054    public static boolean isTablet() {
055        return matchMedia(ViewPort.TABLET);
056    }
057
058    public static boolean isLaptop() {
059        return matchMedia(ViewPort.LAPTOP) || matchMedia(ViewPort.LAPTOP_LARGE) || matchMedia(ViewPort.LAPTOP_4K);
060    }
061
062    public static native boolean isTouchScreenDevice() /*-{
063        return 'ontouchstart' in document.documentElement;
064    }-*/;
065
066    public static boolean matchMedia(ViewPort viewPort) {
067        return Window.matchMedia(viewPort.getMediaQuery());
068    }
069}
