020package gwt.material.design.client.base.mixin;
021
022import com.google.gwt.user.client.ui.HasText;
023import com.google.gwt.user.client.ui.UIObject;
024import gwt.material.design.client.base.HasError;
025import gwt.material.design.client.constants.CssName;
026
027/**
028 * @author Ben Dol
029 */
030public class ErrorMixin<T extends UIObject & HasError, H extends UIObject & HasText>
031        extends AbstractMixin<T> implements HasError {
032
033    private H textObject;
034    private UIObject target;
035    private UIObject lblPlaceholder;
036    private String helperText;
037
038    public ErrorMixin(final T widget) {
039        this(widget, null);
040    }
041
042    public ErrorMixin(final T widget, final H textObject) {
043        this(widget, textObject, widget);
044    }
045
046    public ErrorMixin(final T widget, final H textObject, UIObject target) {
047        super(widget);
048
049        this.textObject = textObject;
050        this.target = target;
051    }
052
053    public ErrorMixin(final T widget, final H textObject, UIObject target, UIObject lblPlaceholder) {
054        this(widget, textObject, target);
055        this.lblPlaceholder = lblPlaceholder;
056    }
057
058    @Override
059    public void setError(String error) {
060        if (textObject != null) {
061            textObject.setText(error);
062            textObject.addStyleName(CssName.FIELD_ERROR_LABEL);
063            textObject.removeStyleName(CssName.FIELD_HELPER_LABEL);
064            textObject.removeStyleName(CssName.FIELD_SUCCESS_LABEL);
065            textObject.setVisible(true);
066        }
067        if (target != null) {
068            target.addStyleName(CssName.FIELD_ERROR);
069            target.removeStyleName(CssName.FIELD_SUCCESS);
070        }
071        if (lblPlaceholder != null) {
072            lblPlaceholder.removeStyleName("green-text");
073            lblPlaceholder.addStyleName("red-text");
074        }
075    }
076
077    @Override
078    public void setSuccess(String success) {
079        if (textObject != null) {
080            textObject.setText(success);
081            textObject.addStyleName(CssName.FIELD_SUCCESS_LABEL);
082            textObject.removeStyleName(CssName.FIELD_HELPER_LABEL);
083            textObject.removeStyleName(CssName.FIELD_ERROR_LABEL);
084            textObject.setVisible(true);
085        }
086        if (target != null) {
087            target.addStyleName(CssName.FIELD_SUCCESS);
088            target.removeStyleName(CssName.FIELD_ERROR);
089        }
090        if (lblPlaceholder != null) {
091            lblPlaceholder.removeStyleName("red-text");
092            lblPlaceholder.addStyleName("green-text");
093        }
094    }
095
096    @Override
097    public void setHelperText(String helperText) {
098        this.helperText = helperText;
099        clearErrorOrSuccess();
100    }
101
102    @Override
103    public void clearErrorOrSuccess() {
104        if (textObject != null) {
105            textObject.setText(helperText == null ? "" : helperText);
106            if (helperText != null) {
107                textObject.addStyleName(CssName.FIELD_HELPER_LABEL);
108            } else {
109                textObject.removeStyleName(CssName.FIELD_HELPER_LABEL);
110            }
111            textObject.removeStyleName(CssName.FIELD_ERROR_LABEL);
112            textObject.removeStyleName(CssName.FIELD_SUCCESS_LABEL);
113            textObject.setVisible(helperText != null);
114        }
115        if (target != null) {
116            target.removeStyleName(CssName.FIELD_ERROR);
117            target.removeStyleName(CssName.FIELD_SUCCESS);
118        }
119        if (lblPlaceholder != null) {
120            lblPlaceholder.removeStyleName("red-text");
121            lblPlaceholder.removeStyleName("green-text");
122        }
123    }
124}
