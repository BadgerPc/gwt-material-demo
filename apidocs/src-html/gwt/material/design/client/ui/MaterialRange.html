020package gwt.material.design.client.ui;
021
022import com.google.gwt.core.client.GWT;
023import com.google.gwt.dom.client.Document;
024import com.google.gwt.dom.client.Element;
025import com.google.gwt.dom.client.Style.Unit;
026import com.google.gwt.event.dom.client.ChangeEvent;
027import com.google.gwt.event.dom.client.ChangeHandler;
028import com.google.gwt.event.dom.client.HasChangeHandlers;
029import com.google.gwt.event.shared.HandlerRegistration;
030import gwt.material.design.client.base.AbstractValueWidget;
031import gwt.material.design.client.base.HasError;
032import gwt.material.design.client.base.mixin.ErrorMixin;
033import gwt.material.design.client.constants.CssName;
034import gwt.material.design.client.constants.InputType;
035import gwt.material.design.client.ui.html.Paragraph;
036import gwt.material.design.client.ui.html.Span;
037
038import static gwt.material.design.jquery.client.api.JQuery.$;
039
040//@formatter:off
041
042/**
043 * Material Range - a slider that initialize the minimum and maximum values.
044 * <p>
045 * <h3>UiBinder Usage:</h3>
046 * <pre>
047 * {@code <m:MaterialRange value="2" min="20" max="50" value="25"/>}
048 * </pre>
049 *
050 * @author kevzlou7979
051 * @author Ben Dol
052 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#forms">Material Range</a>
053 */
054//@formatter:on
055public class MaterialRange extends AbstractValueWidget<Integer> implements HasChangeHandlers, HasError {
056
057    private Paragraph paragraph = new Paragraph();
058    private MaterialInput rangeInputElement = new MaterialInput();
059    private Span thumb = new Span();
060
061    private static String VALUE = "value";
062    private static String MAX = "max";
063    private static String MIN = "min";
064    private MaterialLabel lblError = new MaterialLabel();
065
066    private final ErrorMixin<MaterialRange, MaterialLabel> errorMixin = new ErrorMixin<>(this, lblError, null);
067
068    /**
069     * Creates a range
070     */
071    public MaterialRange() {
072        super(Document.get().createFormElement());
073        getElement().setAttribute("action", "#");
074        lblError.setVisible(false);
075        paragraph.setStyleName(CssName.RANGE_FIELD);
076
077        rangeInputElement.setType(InputType.RANGE);
078        paragraph.add(rangeInputElement);
079
080        thumb.getElement().setClassName(CssName.THUMB);
081        Span value = new Span();
082        value.getElement().setClassName(CssName.VALUE);
083        thumb.add(value);
084
085        paragraph.add(thumb);
086        add(paragraph);
087
088        lblError.getElement().getStyle().setMarginTop(-10, Unit.PX);
089        add(lblError);
090    }
091
092    /**
093     * Creates a range with specified values
094     *
095     * @param min   - start min value
096     * @param max   - end max value
097     * @param value - default range value
098     */
099    public MaterialRange(Integer min, Integer max, Integer value) {
100        this();
101        setMin(min);
102        setMax(max);
103        setValue(value);
104    }
105
106    public void reset() {
107        setValue(getMin());
108        clearErrorOrSuccess();
109    }
110
111    /**
112     * Retrieve the Integer value from the given Attribute of the range element
113     *
114     * @param attribute The name of the attribute on the range element
115     * @return The Integer vaulue read from the given attribute or null
116     */
117    protected Integer getIntFromRangeElement(String attribute) {
118        Element ele = $(rangeInputElement).asElement();
119        if (ele != null) {
120            return ele.getPropertyInt(attribute);
121        }
122        return null;
123    }
124
125    /**
126     * Set the given Integer value to the attribute of the range element.
127     */
128    protected void setIntToRangeElement(String attribute, Integer val) {
129        Element ele = $(rangeInputElement).asElement();
130        if (ele != null) {
131            ele.setPropertyInt(attribute, val);
132        }
133    }
134
135    /**
136     * Read the current value
137     *
138     * @return The Integer value or null
139     */
140    @Override
141    public Integer getValue() {
142        return getIntFromRangeElement(VALUE);
143    }
144
145    @Override
146    public void setValue(Integer value, boolean fireEvents) {
147        if (value == null) {
148            GWT.log("Value must be null", new RuntimeException());
149            return;
150        }
151        if (value < getMin()) {
152            GWT.log("Value must not be less than the minimum range value.", new RuntimeException());
153            return;
154        }
155        if (value > getMax()) {
156            GWT.log("Value must not be greater than the maximum range value", new RuntimeException());
157            return;
158        }
159        setIntToRangeElement(VALUE, value);
160
161        super.setValue(value, fireEvents);
162    }
163
164    /**
165     * Read the min value
166     *
167     * @return The Integer or null
168     */
169    public Integer getMin() {
170        return getIntFromRangeElement(MIN);
171    }
172
173    /**
174     * Write the current min value
175     *
176     * @param min value must be &lt; max
177     */
178    public void setMin(Integer min) {
179        setIntToRangeElement(MIN, min);
180    }
181
182    /**
183     * Read the max value
184     *
185     * @return The Integer or null
186     */
187    public Integer getMax() {
188        return getIntFromRangeElement(MAX);
189    }
190
191    /**
192     * Write the current max value
193     *
194     * @param max value must be &gt; min
195     */
196    public void setMax(Integer max) {
197        setIntToRangeElement(MAX, max);
198    }
199
200    /**
201     * Register the ChangeHandler to become notified if the user changes the slider.
202     * The Handler is called when the user releases the mouse only at the end of the slide
203     * operation.
204     */
205    @Override
206    public HandlerRegistration addChangeHandler(final ChangeHandler handler) {
207        return addDomHandler(handler, ChangeEvent.getType());
208    }
209
210    @Override
211    public void setError(String error) {
212        errorMixin.setError(error);
213    }
214
215    @Override
216    public void setSuccess(String success) {
217        errorMixin.setSuccess(success);
218    }
219
220    @Override
221    public void setHelperText(String helperText) {
222        errorMixin.setHelperText(helperText);
223    }
224
225    @Override
226    public void clearErrorOrSuccess() {
227        errorMixin.clearErrorOrSuccess();
228    }
229
230    public MaterialLabel getLblError() {
231        return lblError;
232    }
233
234    public MaterialInput getRangeInputElement() {
235        return rangeInputElement;
236    }
237}
