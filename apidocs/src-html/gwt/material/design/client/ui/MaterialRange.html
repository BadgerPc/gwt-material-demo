020package gwt.material.design.client.ui;
021
022import com.google.gwt.dom.client.Document;
023import com.google.gwt.dom.client.Element;
024import com.google.gwt.event.dom.client.ChangeEvent;
025import com.google.gwt.event.dom.client.ChangeHandler;
026import com.google.gwt.event.dom.client.HasChangeHandlers;
027import com.google.gwt.event.shared.HandlerRegistration;
028import gwt.material.design.client.base.AbstractValueWidget;
029import gwt.material.design.client.base.HasError;
030import gwt.material.design.client.base.mixin.ErrorMixin;
031import gwt.material.design.client.constants.CssName;
032import gwt.material.design.client.constants.InputType;
033import gwt.material.design.client.ui.html.Paragraph;
034import gwt.material.design.client.ui.html.Span;
035
036import static gwt.material.design.jquery.client.api.JQuery.$;
037
038//@formatter:off
039
040/**
041 * Material Range - a slider that initialize the minimum and maximum values.
042 * <p>
043 * <h3>UiBinder Usage:</h3>
044 * <pre>
045 * {@code <m:MaterialRange value="2" min="20" max="50" value="25"/>}
046 * </pre>
047 *
048 * @author kevzlou7979
049 * @author Ben Dol
050 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#forms">Material Range</a>
051 * @see <a href="https://material.io/guidelines/components/sliders.html">Material Design Specification</a>
052 */
053//@formatter:on
054public class MaterialRange extends AbstractValueWidget<Integer> implements HasChangeHandlers, HasError {
055
056    private Paragraph paragraph = new Paragraph();
057    private MaterialInput rangeInputElement = new MaterialInput();
058    private Span thumb = new Span();
059
060    private static String VALUE = "value";
061    private static String MAX = "max";
062    private static String MIN = "min";
063    private MaterialLabel errorLabel = new MaterialLabel();
064    private HandlerRegistration changeHandler;
065
066    private final ErrorMixin<MaterialRange, MaterialLabel> errorMixin = new ErrorMixin<>(this, errorLabel, null);
067
068    /**
069     * Creates a range
070     */
071    public MaterialRange() {
072        super(Document.get().createFormElement());
073        build();
074    }
075
076    @Override
077    protected void build() {
078        getElement().setAttribute("action", "#");
079        errorLabel.setVisible(false);
080        paragraph.setStyleName(CssName.RANGE_FIELD);
081
082        rangeInputElement.setType(InputType.RANGE);
083        paragraph.add(rangeInputElement);
084
085        thumb.getElement().setClassName(CssName.THUMB);
086        Span value = new Span();
087        value.getElement().setClassName(CssName.VALUE);
088        thumb.add(value);
089
090        paragraph.add(thumb);
091        add(paragraph);
092
093        add(errorLabel);
094    }
095
096    @Override
097    protected void onLoad() {
098        super.onLoad();
099
100        if (changeHandler == null) {
101            changeHandler = addChangeHandler(changeEvent -> setValue(getValue(),true));
102        }
103    }
104
105    /**
106     * Creates a range with specified values
107     *
108     * @param min   - start min value
109     * @param max   - end max value
110     * @param value - default range value
111     */
112    public MaterialRange(Integer min, Integer max, Integer value) {
113        this();
114        setMin(min);
115        setMax(max);
116        setValue(value);
117    }
118
119    public void reset() {
120        setValue(getMin());
121        clearErrorOrSuccess();
122    }
123
124    /**
125     * Retrieve the Integer value from the given Attribute of the range element
126     *
127     * @param attribute The name of the attribute on the range element
128     * @return The Integer vaulue read from the given attribute or null
129     */
130    protected Integer getIntFromRangeElement(String attribute) {
131        Element ele = $(rangeInputElement).asElement();
132        if (ele != null) {
133            return ele.getPropertyInt(attribute);
134        }
135        return null;
136    }
137
138    /**
139     * Set the given Integer value to the attribute of the range element.
140     */
141    protected void setIntToRangeElement(String attribute, Integer val) {
142        Element ele = $(rangeInputElement).asElement();
143        if (ele != null) {
144            ele.setPropertyInt(attribute, val);
145        }
146    }
147
148    /**
149     * Read the current value
150     *
151     * @return The Integer value or null
152     */
153    @Override
154    public Integer getValue() {
155        return getIntFromRangeElement(VALUE);
156    }
157
158    @Override
159    public void setValue(Integer value, boolean fireEvents) {
160        if (value == null) {
161            throw new IllegalArgumentException("Value must not be null");
162        }
163        if (value < getMin()) {
164            throw new IllegalArgumentException("Value must not be less than the minimum range value.");
165        }
166        if (value > getMax()) {
167            throw new IllegalArgumentException("Value must not be greater than the maximum range value");
168        }
169        setIntToRangeElement(VALUE, value);
170
171        super.setValue(value, fireEvents);
172    }
173
174    /**
175     * Read the min value
176     *
177     * @return The Integer or null
178     */
179    public Integer getMin() {
180        return getIntFromRangeElement(MIN);
181    }
182
183    /**
184     * Write the current min value
185     *
186     * @param min value must be &lt; max
187     */
188    public void setMin(Integer min) {
189        setIntToRangeElement(MIN, min);
190    }
191
192    /**
193     * Read the max value
194     *
195     * @return The Integer or null
196     */
197    public Integer getMax() {
198        return getIntFromRangeElement(MAX);
199    }
200
201    /**
202     * Write the current max value
203     *
204     * @param max value must be &gt; min
205     */
206    public void setMax(Integer max) {
207        setIntToRangeElement(MAX, max);
208    }
209
210    /**
211     * Register the ChangeHandler to become notified if the user changes the slider.
212     * The Handler is called when the user releases the mouse only at the end of the slide
213     * operation.
214     */
215    @Override
216    public HandlerRegistration addChangeHandler(final ChangeHandler handler) {
217        return getRangeInputElement().addDomHandler(handler, ChangeEvent.getType());
218    }
219
220    @Override
221    public void setError(String error) {
222        errorMixin.setError(error);
223    }
224
225    @Override
226    public void setSuccess(String success) {
227        errorMixin.setSuccess(success);
228    }
229
230    @Override
231    public void setHelperText(String helperText) {
232        errorMixin.setHelperText(helperText);
233    }
234
235    @Override
236    public void clearErrorOrSuccess() {
237        errorMixin.clearErrorOrSuccess();
238    }
239
240    public MaterialLabel getErrorLabel() {
241        return errorLabel;
242    }
243
244    public MaterialInput getRangeInputElement() {
245        return rangeInputElement;
246    }
247
248    public Paragraph getParagraph() {
249        return paragraph;
250    }
251
252    public Span getThumb() {
253        return thumb;
254    }
255}
