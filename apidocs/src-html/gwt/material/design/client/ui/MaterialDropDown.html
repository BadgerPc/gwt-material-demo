023import com.google.gwt.dom.client.Element;
024import com.google.gwt.dom.client.Style;
025import com.google.gwt.uibinder.client.UiConstructor;
026import com.google.gwt.user.client.ui.Widget;
027import gwt.material.design.client.base.HasWaves;
028import gwt.material.design.client.base.helper.DOMHelper;
029import gwt.material.design.client.constants.Alignment;
030import gwt.material.design.client.ui.html.ListItem;
031import gwt.material.design.client.ui.html.UnorderedList;

033//@formatter:off
034
035/**
036 * You can add dropdown easily by specifying it's item
037 * content and add a UiHandler on it to implement any event.
038 *
039 * <h3>UiBinder Usage:</h3>
040 * <pre>
041 *{@code
042 *<m:MaterialDropDown>
043 *   <m:MaterialLink text="First" />
044 *   <m:MaterialLink text="Second" />
045 *   <m:MaterialLink text="Third" />
046 * </m:MaterialDropDown>
047 * }
048 * </pre>
049 * @author kevzlou7979
050 * @author Ben Dol
051 * @see <a href="http://gwt-material-demo.herokuapp.com/#dropdowns">Material DropDowns</a>
052 */
053//@formatter:on
054public class MaterialDropDown extends UnorderedList {
055
056    private String activator;
057    private Element activatorElem;
058
059    // Options
060    private int inDuration = 300;
061    private int outDuration = 225;
062    private boolean constrainWidth = true;
063    private boolean hover = false;
064    private boolean belowOrigin = false;
065    private int gutter = 0;
066    private String alignment = Alignment.LEFT.getCssName();
067
068    public MaterialDropDown() {
069        setStyleName("dropdown-content");
070    }
071
072    /**
073     * Material Dropdown - adds a list item selection when button, link, icon button pressed.
074     * @param activator - data-activates attribute name of your dropdown activator.
075     */
076    @UiConstructor
077    public MaterialDropDown(String activator) {
078        this();
079        this.activator = activator;
080
081        getElement().setId(this.activator);
082    }
083
084    @Override
085    protected void onLoad() {
086        super.onLoad();
087
088        initialize();
089    }
090
091    /**
092     * The duration of the transition enter in milliseconds. Default: 300
093     */
094    public void setInDuration(int durationMillis) {
095        this.inDuration = durationMillis;
096    }
097
098    public int getInDuration() {
099        return inDuration;
100    }
101
102    /**
103     * The duration of the transition out in milliseconds. Default: 225
104     */
105    public void setOutDuration(int durationMillis) {
106        this.outDuration = durationMillis;
107    }
108
109    public int getOutDuration() {
110        return outDuration;
111    }
112
113    /**
114     * If true, constrainWidth to the size of the dropdown activator. Default: true
115     */
116    public void setConstrainWidth(boolean constrainWidth) {
117        this.constrainWidth = constrainWidth;
118    }
119
120    public boolean isConstrainWidth() {
121        return constrainWidth;
122    }
123
124    /**
125     * If true, the dropdown will open on hover. Default: false
126     */
127    public void setHover(boolean hover) {
128        this.hover = hover;
129    }
130
131    public boolean isHover() {
132        return hover;
133    }
134
135    /**
136     * This defines the spacing from the aligned edge. Default: 0
137     */
138    public void setGutter(int gutter) {
139        this.gutter = gutter;
140    }
141
142    public int getGutter() {
143        return gutter;
144    }
145
146    /**
147     * If true, the dropdown will show below the activator. Default: false
148     */
149    public void setBelowOrigin(boolean belowOrigin) {
150        this.belowOrigin = belowOrigin;
151    }
152
153    public boolean isBelowOrigin() {
154        return belowOrigin;
155    }
156
157    /**
158     * Defines the edge the menu is aligned to. Default: 'left'
159     */
160    public void setAlignment(Alignment alignment) {
161        this.alignment = alignment.getCssName();
162    }
163
164    public Alignment getAlignment() {
165        return Alignment.fromStyleName(alignment);
166    }
167
168    @Override
169    public void add(Widget child) {
170        if(child instanceof ListItem) {
171            child.getElement().getStyle().setDisplay(Style.Display.BLOCK);
172            add(child, (Element) getElement());
173        } else {
174            ListItem li = new ListItem(child);
175            if(child instanceof HasWaves) {
176                li.setWaves(((HasWaves) child).getWaves());
177                ((HasWaves) child).setWaves(null);
178            }
179            li.getElement().getStyle().setDisplay(Style.Display.BLOCK);
180            add(li, (Element) getElement());
181        }
182    }
183
184    protected void initialize() {
185        Element activatorElem = DOMHelper.getElementByAttribute("data-activates", activator);
186        if(activatorElem == null) {
187            throw new IllegalStateException("There is no activator element with id: '" + activator
188                + "' in the DOM, cannot instantiate MaterialDropDown without an activator.");
189        } else {
190            this.activatorElem = activatorElem;
191        }
192
193        initialize(activatorElem);
194    }
195
196    /**
197     * Must be called after changing any options.
198     */
199    public void reinitialize() {
200        remove(activatorElem);
201        initialize(activatorElem);
202    }
203

205     * Initialize the dropdown components.
206     */

223        $wnd.jQuery(activator).dropdown("remove");

225}
