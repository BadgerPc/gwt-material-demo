051/**
052 * <p>Material ListBox is another dropdown component that will set / get the value depends on the selected index
053 * <h3>UiBinder Usage:</h3>
054 * <p>
055 * <pre>
056 * {@code
057 *    <m:MaterialListBox ui:field="lstBox" />
058 * }
059 * </pre>
060 * <h3>Java Usage:</h3>
061 * <p>
062 * <pre>
063 * {@code
064 *     // functions
065 *    lstBox.setSelectedIndex(2);
066 *    lstBox.getSelectedIndex();
067 *    lstBox.addValueChangeHandler(handler);
068 * }
069 * </pre>
070 * </p>
071 *
072 * @author kevzlou7979
073 * @author Ben Dol
074 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#forms">Material ListBox</a>
075 * @see <a href="https://material.io/guidelines/components/menus.html">Material Design Specification</a>
076 */
077//@formatter:on

107    @Override
108    protected void build() {
109        if (!isInitialize()) {
110            $(listBox.getElement()).change((e, param) -> {
111                try {
112                    ValueChangeEvent.fire(this, getValue());
113                } catch (IndexOutOfBoundsException ex) {
114                    GWT.log("ListBox value change handler threw an exception.", ex);
115                }
116                return true;
117            });
118            valueChangeHandler = addValueChangeHandler(valueChangeEvent -> {
119                if (isToggleReadOnly()) {
120                    setReadOnly(true);
121                }
122            });
123            add(listBox);
124            add(label);
125            add(errorLabel);
126        }
127    }
128
129    @Override
130    protected void onUnload() {
131        super.onUnload();
132
133        if (valueChangeHandler != null) {
134            valueChangeHandler.removeHandler();
135            valueChangeHandler = null;
136        }
137        $(listBox.getElement()).off("change");
138        $(listBox.getElement()).material_select("destroy");
139    }

184    @Override
185    protected void initialize() {
186        JsMaterialElement.$(listBox.getElement()).material_select(() -> JQuery.$("input.select-dropdown").trigger("close", null));
187        // Fixed auto hide when scrolling on IE Browsers
188        JQuery.$(listBox.getElement()).siblings("input.select-dropdown").off("mousedown").on("mousedown", (e, param1) -> {
189            if (!ViewPortHelper.isTouchScreenDevice()) {
190                e.preventDefault();
191            }
192            return true;
193        });
194    }

254    @Override
255    public void setValue(T value, boolean fireEvents) {
256        int index = values.indexOf(value);
257        if (index >= 0) {
258            T before = getValue();
259            setSelectedIndex(index);
260
261            if (fireEvents) {
262                ValueChangeEvent.fireIfNotEqual(this, before, value);
263            }
264        }
265    }

607    /**
608     * Returns all selected values of the list box, or empty array if none.
609     *
610     * @return the selected values of the list box
611     */
612    public String[] getItemsSelected() {
613        List<String> selected = new LinkedList<>();
614        for (int i = 0; i < listBox.getItemCount(); i++) {
615            if (listBox.isItemSelected(i)) {
616                selected.add(listBox.getValue(i));
617            }
618        }
619        return selected.toArray(new String[selected.size()]);
620    }

622    /**
623     * Sets the currently selected value.
624     * <p>
625     * After calling this method, only the specified item in the list will
626     * remain selected. For a ListBox with multiple selection enabled, see
627     * {@link #setValueSelected(String, boolean)} to select multiple items at a
628     * time.
629     *
630     * @param value the value of the item to be selected
631     */
632    public void setSelectedValue(String value) {
633        int idx = getIndex(value);
634        if (idx >= 0) {
635            setSelectedIndex(idx);
636        }
637    }
638
639    /**
640     * Gets the index of the specified value.
641     *
642     * @param value the value of the item to be found
643     * @return the index of the value
644     */
645    public int getIndex(String value) {
646        int count = getItemCount();
647        for (int i = 0; i < count; i++) {
648            if (getValue(i).equals(value)) {
649                return i;
650            }
651        }
652        return -1;
653    }

662        int idx = getIndex(value);
