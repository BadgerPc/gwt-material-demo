023import com.google.gwt.dom.client.Element;

039import java.util.List;

045/**
046 * The tabs structure consists of an unordered list of tabs that have hashes corresponding to tab ids.
047 * Then when you click on each tab, only the container with the corresponding tab id will become visible.
048 *
049 * <h3>UiBinder Usage:</h3>
050 * <pre>
051 * {@code
052 * <m:MaterialTab ui:field="tab"  backgroundColor="BLUE">
053 *     <m:MaterialTabItem waves="YELLOW" grid="l4"><i:Link text="Tab 1" href="#tab1" textColor="WHITE"/></m:MaterialTabItem>
054 *     <m:MaterialTabItem waves="YELLOW" grid="l4"><i:Link text="Tab 2" href="#tab2" textColor="WHITE"/></m:MaterialTabItem>
055 *     <m:MaterialTabItem waves="YELLOW" grid="l4"><i:Link text="Tab 3" href="#tab3" textColor="WHITE"/></m:MaterialTabItem>
056 * </m:MaterialTab>
057 *
058 * <m:MaterialPanel m:id="tab1">
059 *     <m:MaterialTitle title="Tab 1" description="Tab 1 Content"/>
060 * </m:MaterialPanel>
061 * <m:MaterialPanel m:id="tab2">
062 *     <m:MaterialTitle title="Tab 2" description="Tab 2 Content"/>
063 * <m:MaterialPanel>
064 * <i:Panel m:id="tab3">
065 *     <m:MaterialTitle title="Tab 3" description="Tab 3 Content"/>
066 * </m:MaterialPanel>
067 * }
068 * </pre>
069 *
070 * @author kevzlou7979
071 * @author Ben Dol
072 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#tabs">Material Tabs</a>
073 * @see <a href="https://material.io/guidelines/components/tabs.html">Material Design Specification</a>
074 */
075//@formatter:on
076public class MaterialTab extends UnorderedList implements HasType<TabType>, HasSelectionHandlers<Integer> {
077
078    private int tabIndex;
079    private Color indicatorColor;
080
081    private MaterialWidget indicator;
082    private ColorsMixin<MaterialWidget> indicatorColorMixin;
083
084    private final CssTypeMixin<TabType, MaterialTab> typeMixin = new CssTypeMixin<>(this);
085    private List<HandlerRegistration> handlers = new ArrayList<>();
086    private HandlerRegistration selectionHandler;
087
088    public MaterialTab() {
089        super(CssName.TABS);
090    }
091
092    @Override
093    protected void initialize() {
094        if (getWidgetCount() > 0) {
095            $(getElement()).tabs();
096
097            if (selectionHandler == null) {
098                selectionHandler = addSelectionHandler(selectionEvent -> this.tabIndex = selectionEvent.getSelectedItem());
099            }
100
101            if (handlers.size() > 0) {
102                for (HandlerRegistration handler : handlers) {
103                    handler.removeHandler();
104                }
105                handlers.clear();
106            }
107
108            for (Widget w : getChildren()) {
109                if (w instanceof MaterialTabItem) {
110                    HandlerRegistration handler = ((MaterialTabItem) w).addMouseDownHandler(e -> SelectionEvent.fire(MaterialTab.this, getChildren().indexOf(w)));
111                    handlers.add(handler);
112                }
113            }
114            applyIndicator();
115        }
116    }
117
118    @Override
119    public void reinitialize() {
120        clearAllIndicators();
121        $(getElement()).tabs();
122    }
123
124    protected void applyIndicator() {
125        indicator = new MaterialWidget(getIndicatorElement());
126        indicatorColorMixin = new ColorsMixin<>(indicator);
127        setIndicatorColor(indicatorColor);
128        clearAllIndicators();
129    }
130
131    protected void clearAllIndicators() {
132        Scheduler.get().scheduleDeferred(() -> {
133            for (int i = 1; i < $(getElement()).find(".indicator").length(); i++) {
134                $(getElement()).find(".indicator").eq(i).remove();
135            }
136        });
137    }
138
139    public int getTabIndex() {
140        return tabIndex;
141    }
142
143    public void setTabIndex(int tabIndex) {
144        this.tabIndex = tabIndex;
145        int i = 0;
146        for (Widget w : this) {
147            if (i == tabIndex) {
148                if (w instanceof MaterialTabItem) {
149                    ((MaterialTabItem) w).selectTab();
150                    break;
151                }
152            }
153            i++;
154        }
155    }
156
157    public void setIndicatorColor(Color indicatorColor) {
158        this.indicatorColor = indicatorColor;
159
160        if (indicatorColorMixin != null && indicatorColor != null) {
161            indicatorColorMixin.setBackgroundColor(indicatorColor);
162        }
163    }
164
165    /**
166     * Select a given tab by id.
167     *
168     * @param tabId Tab to selects id.
169     */
170    public void selectTab(String tabId) {
171        Scheduler.get().scheduleDeferred(() -> $(getElement()).tabs("select_tab", tabId));
172    }
173
174    protected Element getIndicatorElement() {
175        return $(getElement()).find(".indicator").last().asElement();
176    }
177
178    @Override
179    public void setType(TabType type) {
180        typeMixin.setType(type);
181    }
182
183    @Override
184    public TabType getType() {
185        return typeMixin.getType();
186    }
187
188    @Override
189    public HandlerRegistration addSelectionHandler(SelectionHandler<Integer> handler) {
190        return addHandler(handler, SelectionEvent.getType());
191    }
192}
