037/**
038 * Dialogs are content that are not original visible on a page but show up with
039 * extra information if needed. The transitions should make the appearance of
040 * the dialog make sense and not jarring to the user.
041 * <p>
042 * <h3>UiBinder Usage:</h3>
043 * <p>
044 * <pre>
045 * {@code
046 * <m:MaterialModal ui:field="modal" type="FIXED_FOOTER" dismissible="true" inDuration="500" outDuration="800">
047 *     <m:MaterialModalContent>
048 *         <m:MaterialTitle title="Title" description="Description" />
049 *     </m:MaterialModalContent>
050 *     <m:MaterialModalFooter>
051 *         <m:MaterialButton text="Close Modal" type="FLAT"/>
052 *     </m:MaterialModalFooter>
053 * </m:MaterialModal>
054 * }
055 * </pre>
056 * <p>
057 * *
058 * <h3>Java Usage:</h3>
059 * <p>
060 * <pre>
061 * {
062 *     &#064;code
063 *     &#064;UiField
064 *     MaterialModal modal;
065 *     modal.open();
066 * }
067 * </pre>
068 * <p>
069 * </p>
070 *
071 * @author kevzlou7979
072 * @author Ben Dol
073 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#dialogs">Material Modals</a>
074 * @see <a href="https://material.io/guidelines/components/dialogs.html#">Material Design Specification</a>
075 */
076// @formatter:on
077public class MaterialModal extends MaterialWidget implements HasType<ModalType>, HasInOutDurationTransition,
078        HasDismissible, HasCloseHandlers<MaterialModal>, HasOpenHandlers<MaterialModal>, HasFullscreen {
079
080    private final CssTypeMixin<ModalType, MaterialModal> typeMixin = new CssTypeMixin<>(this);
081    private final FullscreenMixin fullscreenMixin = new FullscreenMixin(this);
082    private int inDuration = 300;
083    private int outDuration = 200;
084    private boolean dismissible = false;
085    private double opacity = 0.5;
086    private JsModalOptions options;
087
088    public MaterialModal() {
089        super(Document.get().createDivElement(), CssName.MODAL);
090    }
091
092    /**
093     * Open the modal programmatically
094     * <p>
095     * Note: the MaterialModal component must be added to the document before
096     * calling this method. When declaring this modal on a UiBinder file, the
097     * MaterialModal is already added, but if you call it using pure Java, you
098     * must add it to a container before opening the modal. You can do it by
099     * calling, for example:
100     * </p>
101     * <pre>
102     * MaterialModal modal = new MaterialModal();
103     * RootPanel.get().add(modal);
104     * </pre>
105     *
106     * @throws IllegalStateException If the MaterialModal is not added to the document
107     */
108    public void open() {
109        open(true);
110    }
111
112    /**
113     * Open the modal programmatically
114     * <p>
115     * Note: the MaterialModal component must be added to the document before
116     * calling this method. When declaring this modal on a UiBinder file, the
117     * MaterialModal is already added, but if you call it using pure Java, you
118     * must add it to a container before opening the modal. You can do it by
119     * calling, for example:
120     * </p>
121     * <pre>
122     * MaterialModal modal = new MaterialModal();
123     * RootPanel.get().add(modal);
124     * </pre>
125     * @param fireEvent - Flag whether this component fires Open Event
126     *
127     * @throws IllegalStateException If the MaterialModal is not added to the document
128     */
129    public void open(boolean fireEvent) {
130        // the modal must be added to the document before opening
131        if (this.getParent() == null) {
132            throw new IllegalStateException(
133                    "The MaterialModal must be added to the document before calling open().");
134        }
135        open(getElement(), opacity, dismissible, inDuration, outDuration, fireEvent);
136    }
137
138    /**
139     * Open modal with additional properties
140     *
141     * @param e           - Modal Component
142     * @param opacity     - Opacity of modal background
143     * @param dismissible - Modal can be dismissed by clicking outside of the modal
144     * @param inDuration  - Transition in Duration
145     * @param outDuration - Transition out Duration
146     * @param fireEvent   - Flag whether this component fires Open Event
147     */
148    protected void open(Element e, double opacity, boolean dismissible, int inDuration, int outDuration, boolean fireEvent) {
149        options = new JsModalOptions();
150        options.opacity = opacity;
151        options.dismissible = dismissible;
152        options.in_duration = inDuration;
153        options.out_duration = outDuration;
154        options.complete = () -> onNativeClose(true, true);
155        options.ready = () -> onNativeOpen(fireEvent);
156        $(e).openModal(options);
157    }
158
159    protected void onNativeOpen(boolean fireEvent) {
160        if(fireEvent) {
161            OpenEvent.fire(this, this);
162        }
163    }
164
165    protected void onNativeClose(boolean autoClosed, boolean fireEvent) {
166        if (fireEvent) {
167            CloseEvent.fire(this, this, autoClosed);
168        }
169    }
170
171    /**
172     * Close the modal programmatically. It is the same as calling
173     * {@link #close(boolean)} with <code>false</code> as parameter.
174     * <p>
175     * Note: you may need to remove it MaterialModal from the document if you
176     * are not using UiBinder. See {@link #open()}.
177     * </p>
178     */
179    public void close() {
180        close(false);
181    }
182
183    /**
184     * Close the modal programmatically.
185     * <p>
186     * Note: you may need to remove it MaterialModal from the document if you
187     * are not using UiBinder. See {@link #open()}.
188     * </p>
189     *
190     * @param autoClosed Flag indicating if the modal was automatically dismissed
191     * @see CloseEvent
192     */
193    public void close(boolean autoClosed) {
194        close(autoClosed, true);
195    }
196
197    /**
198     * Close the modal programmatically.
199     * <p>
200     * Note: you may need to remove it MaterialModal from the document if you
201     * are not using UiBinder. See {@link #open()}.
202     * </p>
203     *
204     * @param autoClosed Flag indicating if the modal was automatically dismissed
205     * @param fireEvent Flag whether this component fires Close Event
206     * @see CloseEvent
207     */
208    public void close(boolean autoClosed, boolean fireEvent) {
209        close(getElement(), autoClosed, fireEvent);
210    }
211
212    protected void close(Element e, boolean autoClosed, boolean fireEvent) {
213        if (options != null) {
214            options.complete = () -> onNativeClose(autoClosed, fireEvent);
215            $(e).closeModal(options);
216        }
217    }
218
219    @Override
220    public void setType(ModalType type) {
221        typeMixin.setType(type);
222    }
223
224    @Override
225    public ModalType getType() {
226        return typeMixin.getType();
227    }
228
229    @Override
230    public void setInDuration(int inDuration) {
231        this.inDuration = inDuration;
232    }
233
234    @Override
235    public int getInDuration() {
236        return inDuration;
237    }
238
239    @Override
240    public void setOutDuration(int outDuration) {
241        this.outDuration = outDuration;
242    }
243
244    @Override
245    public int getOutDuration() {
246        return outDuration;
247    }
248
249    @Override
250    public void setDismissible(boolean dismissible) {
251        this.dismissible = dismissible;
252    }
253
254    @Override
255    public boolean isDismissible() {
256        return dismissible;
257    }
258
259    @Override
260    public void setOpacity(double opacity) {
261        this.opacity = opacity;
262    }
263
264    @Override
265    public HandlerRegistration addCloseHandler(CloseHandler<MaterialModal> handler) {
266        return addHandler(handler, CloseEvent.getType());
267    }
268
269    @Override
270    public HandlerRegistration addOpenHandler(OpenHandler<MaterialModal> handler) {
271        return addHandler(handler, OpenEvent.getType());
272    }
273
274    @Override
275    public void setEnabled(boolean enabled) {
276        getEnabledMixin().setEnabled(this, enabled);
277    }
278
279    @Override
280    public void setFullscreen(boolean value) {
281        if (getType() != ModalType.BOTTOM_SHEET) {
282            fullscreenMixin.setFullscreen(value);
283        }
284    }
285
286    @Override
287    public boolean isFullscreen() {
288        return fullscreenMixin.isFullscreen();
289    }
