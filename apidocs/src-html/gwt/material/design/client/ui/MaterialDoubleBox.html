022//@formatter:off
023
024/**
025 * Material Double Box is an input field that accepts any Double based string
026 * from user. <h3>UiBinder Usage:</h3>
027 * <p>
028 * <pre> {@code <m:MaterialDoubleBox ui:field="txtDoubleBox" label="Double value"/>
029 *
030 * // Setting value in Java
031 * txtDoubleBox.setValue(9999.90);
032 * }
033 * </pre>
034 * <p>
035 * The parsing and formatting of the number are done natively by the browser,
036 * using the i18n settings from the user.
037 *
038 * @author paulux84
039 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#forms">Material
040 * DoubleBox</a>
041 * @see <a href="https://material.io/guidelines/components/text-fields.html#">Material Design Specification</a>
042 */
043// @formatter:on
044public class MaterialDoubleBox extends MaterialNumberBox<Double> {
045
046    public MaterialDoubleBox() {
047        setStep("any");
048    }
049
050    public MaterialDoubleBox(String placeholder) {
051        this();
052        setPlaceholder(placeholder);
053    }
054
055    public MaterialDoubleBox(String placeholder, double value) {
056        this(placeholder);
057        setValue(value);
058    }
059
060    @Override
061    protected Double parseNumber(double number) {
062        if (Double.isNaN(number)) {
063            return null;
064        }
065        return number;
066    }
067}
