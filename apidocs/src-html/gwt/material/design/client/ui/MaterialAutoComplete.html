050//@formatter:off
051/**
052 * Use GWT Autocomplete to search for matches from local or remote data sources.
053 * We used MultiWordSuggestOracle to populate the list to be added on the
054 * autocomplete values.
055 *
056 * <h3>UiBinder Usage:</h3>
057 * <pre>
058 * {@code
059 *  <m:MaterialAutoComplete ui:field="autocomplete" placeholder="States" />}
060 * </pre>
061 *
062 * @author kevzlou7979
063 * @see <a href="http://gwt-material-demo.herokuapp.com/#autocompletes">Material AutoComplete</a>
064 */
065//@formatter:on
066public class MaterialAutoComplete extends MaterialWidget implements HasError, HasPlaceholder {
067
068    private List<String> itemValues = new ArrayList<>();
069    private List<ListItem> itemsHighlighted = new ArrayList<>();
070    private FlowPanel panel = new FlowPanel();
071    private UnorderedList list = new UnorderedList();
072    private MultiWordSuggestOracle suggestions;
073    private TextBox itemBox = new TextBox();
074    private int limit = 0;
075    private MaterialLabel lblError = new MaterialLabel();
076
077    private final ErrorMixin<MaterialAutoComplete, MaterialLabel> errorMixin = new ErrorMixin<>(this, lblError, list);
078    
079    /**
080     * Use MaterialAutocomplete to search for matches from
081     * local or remote data sources.
082     */
083    public MaterialAutoComplete() {
084         initWidget(panel);
085    }
086    
087    /**
088     * Use MaterialAutocomplete to search for matches from
089     * local or remote data sources.
090     */
091    public MaterialAutoComplete(MaterialSuggestionOracle suggestions) {
092        initWidget(panel);
093        generateAutoComplete(suggestions);
094    }
095 
096    /**
097     * Generate and build the List Items to be set on Auto Complete box.
098     */
099    private void generateAutoComplete(MaterialSuggestionOracle suggestions) {
100        list.setStyleName("multiValueSuggestBox-list");
101        this.suggestions = suggestions;
102        final ListItem item = new ListItem();
103
104        item.setStyleName("multiValueSuggestBox-input-token");
105        final SuggestBox box = new SuggestBox(suggestions, itemBox);
106        String autocompleteId = DOM.createUniqueId();
107        itemBox.getElement().setId(autocompleteId);
108
109        item.add(box);
110        list.add(item);
111        
112        itemBox.addKeyDownHandler(new KeyDownHandler() {
113            public void onKeyDown(KeyDownEvent event) {
114                switch (event.getNativeKeyCode()) {
115                case KeyCodes.KEY_ENTER:
116                   addItem(itemBox, list);
117                   break;
118
119                case KeyCodes.KEY_BACKSPACE:
120                    if (itemBox.getValue().trim().isEmpty()) {
121                        if (itemsHighlighted.isEmpty()) {
122                            if (itemValues.size() > 0) {
123                                ListItem li = (ListItem) list.getWidget(list.getWidgetCount() - 2);
124                                MaterialChip p = (MaterialChip) li.getWidget(0);
125                                if (itemValues.contains(p.getText())) {
126                                    itemValues.remove(p.getText());
127                                }
128                                list.remove(li);
129                            }
130                        }
131                    }
132
133                case KeyCodes.KEY_DELETE:
134                    if (itemBox.getValue().trim().isEmpty()) {
135                        for (ListItem li : itemsHighlighted) {
136                            li.removeFromParent();
137                            MaterialChip p = (MaterialChip) li.getWidget(0);
138                            itemValues.remove(p.getText());
139                        }
140                        itemsHighlighted.clear();
141                    }
142                    itemBox.setFocus(true);
143                    break;
144                }
145            }
146        });
147        
148        itemBox.addClickHandler(new ClickHandler() {
149
150            @Override
151            public void onClick(ClickEvent event) {
152                box.showSuggestionList();
153            }
154        });
155 
156        box.addSelectionHandler(new SelectionHandler<SuggestOracle.Suggestion>() {
157            public void onSelection(SelectionEvent<SuggestOracle.Suggestion> selectionEvent) {
158                addItem(itemBox, list);
159            }
160        });
161 
162        panel.add(list);
163        panel.getElement().setAttribute("onclick", "document.getElementById('"+autocompleteId+"').focus()");
164        panel.add(lblError);
165        box.setFocus(true);
166    }
167
168    /**
169     * Adding the item value using Material Chips added on auto complete box
170     */
171    private void addItem(final TextBox itemBox, final UnorderedList list) {
172        if(getLimit() > 0) {
173            if(itemValues.size() == getLimit()) {
174                itemBox.setValue("");
175                return;
176            }
177        }
178
179        String value = itemBox.getValue();
180        if (value != null && !"".equals(value.trim()) && !itemValues.contains(value.trim())) {
181
182            final ListItem displayItem = new ListItem();
183            displayItem.setStyleName("multiValueSuggestBox-token");
184
185            String imageChip = value;
186            String textChip = value;
187            final MaterialChip chip = new MaterialChip();
188
189            String s = "<img src=\"";
190            if(imageChip.contains(s)){
191                int ix = imageChip.indexOf(s)+s.length();
192                imageChip = imageChip.substring(ix, imageChip.indexOf("\"", ix+1));
193                chip.setUrl(imageChip);
194                textChip = textChip.replaceAll("[<](/)?img[^>]*[>]", "");
195            }
196            chip.setIconType(IconType.CLOSE);
197            chip.setText(textChip);
198            chip.addClickHandler(new ClickHandler() {
199                public void onClick(ClickEvent clickEvent) {
200                    if (itemsHighlighted.contains(displayItem)) {
201                        chip.removeStyleName("blue white-text");
202                        itemsHighlighted.remove(displayItem);
203                    } else {
204                        chip.addStyleName("blue white-text");
205                        itemsHighlighted.add(displayItem);
206                    }
207                }
208            });
209
210            chip.getIcon().addClickHandler(new ClickHandler() {
211                public void onClick(ClickEvent clickEvent) {
212                    itemValues.remove(chip.getText());
213                    list.remove(displayItem);
214                }
215            });
216
217            if(!itemValues.contains(textChip)){
218                displayItem.add(chip);
219                itemValues.add(chip.getText());
220                list.insert(displayItem, list.getWidgetCount() - 1);
221                itemBox.setValue("");
222                itemBox.setFocus(true);
223            } else {
224                itemBox.setValue("");
225            }
226        } else {
227            itemBox.setValue("");
228        }
229    }
230
231    /**
232     * Clear the chip items on the autocomplete box
233     */
234    public void clear() {
235        ListItem tagToRemove;
236        int num = list.getWidgetCount();
237        for (int i=num-1; i>=0; i--) {
238            tagToRemove = ((ListItem)list.getWidget(i));
239            tagToRemove.removeFromParent();
240        }
241    }
