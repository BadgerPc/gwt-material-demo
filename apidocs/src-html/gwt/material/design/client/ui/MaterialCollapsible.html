040/**
041 * Collapsible are accordion elements that expand when clicked on.
042 * They allow you to hide content that is not immediately relevant
043 * to the user.
044 * <p>
045 * <h3>UiBinder Usage:</h3>
046 * <p>
047 * <pre>
048 * {@code
049 * // Accordion
050 * <m:MaterialCollapsible type="ACCORDION" grid="s12 m6 l8">
051 *   <!-- ITEM 1 -->
052 *   <m:MaterialCollapsibleItem>
053 *     <m:MaterialCollapsibleHeader>
054 *       <m:MaterialLink text="First" iconType="POLYMER" iconPosition="LEFT" textColor="BLACK"/>
055 *     </m:MaterialCollapsibleHeader>
056 *     <m:MaterialCollapsibleBody>
057 *       <m:MaterialLabel text="Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat."/>
058 *     </m:MaterialCollapsibleBody>
059 *   </m:MaterialCollapsibleItem>
060 * </m:MaterialCollapsible>
061 *
062 * // Expandable
063 * <m:MaterialCollapsible type="EXPANDABLE" grid="s12 m6 l8">
064 *   <!-- ITEM 1 -->
065 *   <m:MaterialCollapsibleItem>
066 *     <m:MaterialCollapsibleHeader>
067 *       <m:MaterialLink text="First" iconType="POLYMER" iconPosition="LEFT" textColor="BLACK"/>
068 *     </m:MaterialCollapsibleHeader>
069 *     <m:MaterialCollapsibleBody>
070 *       <m:MaterialLabel text="Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat."/>
071 *     </m:MaterialCollapsibleBody>
072 *   </m:MaterialCollapsibleItem>
073 * </m:MaterialCollapsible>
074 *
075 * // Popout
076 * <m:MaterialCollapsible type="POPOUT" grid="s12 m6 l8">
077 *   <!-- ITEM 1 -->
078 *     <m:MaterialCollapsibleItem>
079 *     <m:MaterialCollapsibleHeader>
080 *       <m:MaterialLink text="First" iconType="POLYMER" iconPosition="LEFT" textColor="BLACK"/>
081 *     </m:MaterialCollapsibleHeader>
082 *     <m:MaterialCollapsibleBody>
083 *       <m:MaterialLabel text="Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat."/>
084 *     </m:MaterialCollapsibleBody>
085 *   </m:MaterialCollapsibleItem>
086 * </m:MaterialCollapsible>
087 * }
088 * </pre>
089 *
090 * @author kevzlou7979
091 * @author Ben Dol
092 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#!collapsible">Material Collapsibles</a>
093 */
094//@formatter:on
095public class MaterialCollapsible extends MaterialWidget implements HasType<CollapsibleType>, HasActiveParent {
096
097    protected interface HasCollapsibleParent {
098        void setParent(MaterialCollapsible parent);
099    }
100
101    private boolean initialized;
102    private boolean accordion = true;
103
104    private int activeIndex = -1;
105    private Widget activeWidget;
106    private final CssTypeMixin<CollapsibleType, MaterialCollapsible> typeMixin = new CssTypeMixin<>(this);
107
108    /**
109     * Creates an empty collapsible
110     */
111    public MaterialCollapsible() {
112        super(Document.get().createULElement(), CssName.COLLAPSIBLE);
113
114        // Items need to be added after the widget has loaded to avoid
115        // premature configuration issues.
116        enableFeature(Feature.ONLOAD_ADD_QUEUE, true);
117    }
118
119    /**
120     * Creates a list and adds the given widgets.
121     */
122    public MaterialCollapsible(final MaterialCollapsibleItem... widgets) {
123        this();
124
125        for (final MaterialCollapsibleItem item : widgets) {
126            add(item);
127        }
128    }
129
130    @Override
131    protected void onLoad() {
132        super.onLoad();
133
134        // Setup the expansion type
135        getElement().setAttribute("data-collapsible", isAccordion() ? CssName.ACCORDION : CssName.EXPANDABLE);
136
137        // Activate preset activation index
138        if (activeIndex != -1 && activeWidget == null) {
139            setActive(activeIndex);
140        }
141
142        // Initialize collapsible after all elements
143        // are attached and marked as active, etc.
144        collapsible(getElement(), accordion);
145    }
146
147    @Override
148    public void add(final Widget child) {
149        if (child instanceof MaterialCollapsibleItem) {
150            ((MaterialCollapsibleItem) child).setParent(this);
151        }
152        super.add(child);
153    }
154
155    @Override
156    public boolean remove(Widget w) {
157        if (w instanceof MaterialCollapsibleItem) {
158            ((MaterialCollapsibleItem) w).setParent(null);
159        }
160        w.removeStyleName(CssName.ACTIVE);
161
162        return super.remove(w);
163    }
164
165    @Override
166    public void setType(CollapsibleType type) {
167        typeMixin.setType(type);
168    }
169
170    @Override
171    public CollapsibleType getType() {
172        return typeMixin.getType();
173    }
174
175    /**
176     * Initialize the collapsible material component.
177     */
178    protected void collapsible() {
179        collapsible(getElement(), isAccordion());
180    }
181
182    protected void collapsible(final Element e, boolean accordion) {
183        $(e).collapsible(accordion);
184        initialized = true;
185    }
186
187    /**
188     * Configure if you want this collapsible container to
189     * accordion its child elements or use expandable.
190     */
191    public void setAccordion(boolean accordion) {
192        this.accordion = accordion;
193
194        if (initialized) {
195            // Since we have attached already reinitialize collapsible.
196            collapsible();
197        }
198    }
199
200    /**
201     * Is the collapsible an 'accordion' type.
202     */
203    public boolean isAccordion() {
204        return accordion;
205    }
206
207    @Override
208    public void setActive(int index) {
209        clearActive();
210        setActive(index, true);
211    }
212
213    @Override
214    public void setActive(int index, boolean active) {
215        activeIndex = index;
216        if (isAttached()) {
217            if (index <= getWidgetCount()) {
218                if (index != 0) {
219                    activeWidget = getWidget(index - 1);
220                    if (activeWidget != null && activeWidget instanceof MaterialCollapsibleItem) {
221                        ((MaterialCollapsibleItem) activeWidget).setActive(active);
222                        if (initialized) {
223                            collapsible();
224                        }
225                    }
226                } else {
227                    GWT.log("The active index must be a one-base index to mark as active.", new IndexOutOfBoundsException());
228                }
229            }
230        }
231    }
232
233    @Override
234    public Widget getActive() {
235        try {
236            return activeWidget;
237        } catch (IndexOutOfBoundsException ex) {
238            return null;
239        }
240    }
241
242    @Override
243    public void clearActive() {
244        clearActiveClass(this);
245        ClearActiveEvent.fire(this);
246    }
247
248    /**
249     * Open the given collapsible item.
250     *
251     * @param index the one-based collapsible item index.
252     */
253    public void open(int index) {
254        setActive(index, true);
255    }
256
257    /**
258     * Close the given collapsible item.
259     *
260     * @param index the one-based collapsible item index.
261     */
262    public void close(int index) {
263        setActive(index, false);
264    }
265
266    /**
267     * Close all the collapsible items.
268     */
269    public void closeAll() {
270        clearActive();
271        if (initialized) {
272            collapsible();
273        }
274    }
275
276    public HandlerRegistration addClearActiveHandler(final ClearActiveHandler handler) {
277        return addHandler(handler, ClearActiveEvent.TYPE);
278    }
279
280    @Override
281    public void setEnabled(boolean enabled) {
282        getEnabledMixin().setEnabled(this, enabled);
283    }
284}
