055/**
056 * SideNav is a material component that gives you a lists of menus and other navigation components.
057 * <p>
058 * <h3>UiBinder Usage:</h3>
059 * <pre>
060 * {@code
061 * <m:MaterialSideNav ui:field="sideNav" width="280" m:id="mysidebar"  type="OPEN" closeOnClick="false">
062 *     <m:MaterialLink href="#about" iconPosition="LEFT" iconType="OUTLINE" text="About" textColor="BLUE"  />
063 *     <m:MaterialLink href="#gettingStarted" iconPosition="LEFT" iconType="DOWNLOAD" text="Getting Started" textColor="BLUE"  >
064 * </m:MaterialSideNav>
065 * }
066 * </pre>
067 *
068 * @author kevzlou7979
069 * @author Ben Dol
070 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#!sidenavs">Material SideNav</a>
071 */

114    @Override
115    protected void onLoad() {
116        super.onLoad();
117
118        // Initialize the side nav
119        initialize();
120
121        if (showOnAttach != null) {
122            Scheduler.get().scheduleDeferred(() -> {
123                if (showOnAttach) {
124                    if (Window.getClientWidth() > 960) {
125                        show();
126                    }
127                } else {
128                    show();
129                    final HandlerRegistration[] openedHandler = new HandlerRegistration[1];
130                    openedHandler[0] = addOpenedHandler((event) -> {
131                        hide();
132
133                        if (openedHandler[0] != null) {
134                            openedHandler[0].removeHandler();
135                        }
136                    });
137                }
138            });
139        }
140    }

170    public Widget wrap(Widget child) {
171        if (child instanceof MaterialImage) {
172            child.getElement().getStyle().setProperty("border", "1px solid #e9e9e9");
173            child.getElement().getStyle().setProperty("textAlign", "center");
174        }
175
176        // Check whether the widget is not selectable by default
177        boolean isNotSelectable = false;
178        if (child instanceof MaterialWidget) {
179            MaterialWidget widget = (MaterialWidget) child;
180            if (widget.getInitialClasses() != null) {
181                if (widget.getInitialClasses().length > 0) {
182                    String initialClass = widget.getInitialClasses()[0];
183                    if (initialClass.contains(CssName.SIDE_PROFILE) || initialClass.contains(CssName.COLLAPSIBLE)) {
184                        isNotSelectable = true;
185                    }
186                }
187            }
188        }
189
190        if (!(child instanceof ListItem)) {
191            // Direct list item not collapsible
192            final ListItem listItem = new ListItem();
193            if (child instanceof MaterialCollapsible) {
194                listItem.getElement().getStyle().setBackgroundColor("transparent");
195            }
196            if (child instanceof HasWaves) {
197                listItem.setWaves(((HasWaves) child).getWaves());
198                ((HasWaves) child).setWaves(null);
199            }
200            listItem.add(child);
201
202            child = listItem;
203        }
204
205        // Collapsible and Side Porfile should not be selectable
206        final Widget finalChild = child;
207        if (!isNotSelectable) {
208            // Active click handler
209            finalChild.addDomHandler(event -> {
210                clearActive();
211                finalChild.addStyleName(CssName.ACTIVE);
212            }, ClickEvent.getType());
213        }
214        child.getElement().getStyle().setDisplay(Style.Display.BLOCK);
215        return child;
216    }

293    protected void processType(SideNavType type) {
294        if (activator != null && type != null) {
295            addStyleName(type.getCssName());
296            switch (type) {
297                case MINI:
298                    setWidth(64);
299                    break;
300                case CARD:
301                    new Timer() {
302                        @Override
303                        public void run() {
304                            if (isSmall()) {
305                                show();
306                            }
307                        }
308                    }.schedule(500);
309                    break;
310                case PUSH:
311                    applyPushType(width);
312                    break;
313            }
314        }
315    }
316
317    protected boolean isSmall() {
318        return !gwt.material.design.client.js.Window.matchMedia("all and (max-width: 992px)");
319    }

321    /**
322     * Push the header, footer, and main to the right part when Close type is applied.
323     */
324    protected void applyPushType(int width) {
325        $(JQuery.window()).off("resize");
326        $(JQuery.window()).resize((e, param1) -> {
327            pushElements(open, width);
328            return true;
329        });
330    }
331
332    protected void pushElements(boolean toggle, int width) {
333        int w = 0;
334        int dur = 200;
335        if (!gwt.material.design.client.js.Window.matchMedia("all and (max-width: 992px)")) {
336            if (toggle) {
337                w = width;
338                dur = 300;
339            }
340            applyTransition($("header").asElement(), w, dur);
341            applyTransition($("main").asElement(), w, dur);
342            applyTransition($("footer").asElement(), w, dur);
343        }
344        onPush(toggle, w, dur);
345    }
346
347    protected void applyTransition(Element elem, int width, int duration) {
348        $(elem).css("transition", duration + "ms");
349        $(elem).css("-moz-transition", duration + "ms");
350        $(elem).css("-webkit-transition", duration + "ms");
351        $(elem).css("margin-left", width + "px");
352    }
353
354    protected void onPush(boolean toggle, int width, int duration) {
355        SideNavPushEvent.fire(this, getElement(), activator, toggle, width, duration);
356    }

377    protected void initialize(boolean strict) {
378        if (activator == null) {
379            activator = DOMHelper.getElementByAttribute("data-activates", getId());
380            if (activator != null) {
381                if (alwaysShowActivator || !isFixed()) {
382                    String style = activator.getAttribute("style");
383                    activator.setAttribute("style", style + "; display: block !important");
384                    activator.removeClassName(CssName.NAVMENU_PERMANENT);
385                }
386            } else if (strict) {
387                throw new RuntimeException("Cannot find an activator for the MaterialSideNav, " +
388                    "please ensure you have a MaterialNavBar with an activator setup to match " +
389                    "this widgets id.");
390            }
391        }
392
393        SideNavType type = getType();
394        processType(type);
395
396        JsSideNavOptions options = new JsSideNavOptions();
397        options.menuWidth = width;
398        options.edge = edge != null ? edge.getCssName() : null;
399        options.closeOnClick = closeOnClick;
400
401        JsMaterialElement element = $(activator);
402        element.sideNav(options);
403
404        element.off("side-nav-closing");
405        element.on("side-nav-closing", e1 -> {
406            onClosing();
407            return true;
408        });
409
410        element.off("side-nav-closed");
411        element.on("side-nav-closed", e1 -> {
412            onClosed();
413            return true;
414        });
415
416        element.off("side-nav-opening");
417        element.on("side-nav-opening", e1 -> {
418            onOpening();
419            return true;
420        });
421
422        element.off("side-nav-opened");
423        element.on("side-nav-opened", e1 -> {
424            onOpened();
425            return true;
426        });
427    }
428
429    protected void onClosing() {
430        open = false;
431        if (getType().equals(SideNavType.PUSH)) {
432            pushElements(false, width);
433        }
434
435        SideNavClosingEvent.fire(this);
436    }
437
438    protected void onClosed() {
439        SideNavClosedEvent.fire(this);
440    }
441
442    protected void onOpening() {
443        open = true;
444        if (getType().equals(SideNavType.PUSH)) {
445            pushElements(true, width);
446        }
447
448        SideNavOpeningEvent.fire(this);
449    }
450
451    protected void onOpened() {
452        if (allowBodyScroll) {
453            RootPanel.getBodyElement().getStyle().clearOverflow();
454        }
455
456        SideNavOpenedEvent.fire(this);
457    }
