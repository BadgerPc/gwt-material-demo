054/**
055 * SideNav is a material component that gives you a lists of menus and other navigation components.
056 *
057 * <h3>UiBinder Usage:</h3>
058 * <pre>
059 * {@code
060 * <m:MaterialSideNav ui:field="sideNav" width="280" m:id="mysidebar"  type="OPEN" closeOnClick="false">
061 *     <m:MaterialLink href="#about" iconPosition="LEFT" icon="OUTLINE" text="About" textColor="blue"  />
062 *     <m:MaterialLink href="#gettingStarted" iconPosition="LEFT" icon="DOWNLOAD" text="Getting Started" textColor="blue"  >
063 * </m:MaterialSideNav>
064 * }
065 * </pre>
066 *
067 * @author kevzlou7979
068 * @author Ben Dol
069 * @see <a href="http://gwt-material-demo.herokuapp.com/#sidenav">Material SideNav</a>
070 */
071//@formatter:on
072public class MaterialSideNav extends UnorderedList implements HasType<SideNavType>, HasSelectables {
073
074    private int width = 240;
075    private Edge edge = Edge.LEFT;
076    private boolean closeOnClick = false;
077    private Element activator;
078    private HandlerRegistration observedHandler;
079
080    private StyleAttributeObserver observer = new StyleAttributeObserver(this, "left");
081
082    private final CssTypeMixin<SideNavType, MaterialSideNav> typeMixin = new CssTypeMixin<>(this);
083    private final ToggleStyleMixin<MaterialSideNav> fixedMixin = new ToggleStyleMixin<>(this, "fixed");
084
085    /**
086     * Container for App Toolbar and App Sidebar , contains Material Links,
087     * Icons or any other material components.
088     */
089    public MaterialSideNav() {
090        super();
091        setStyleName("side-nav");
092    }
093
094    /**
095     *  Creates a list and adds the given widgets.
096     */
097    public MaterialSideNav(final Widget... widgets){
098        this();
099        for (final Widget w : widgets) {
100            add(w);
101        }
102    }

106        this();

133    @Override
134    public void add(Widget child) {
135        if(child instanceof MaterialImage) {
136            child.getElement().getStyle().setProperty("border", "1px solid #e9e9e9");
137            child.getElement().getStyle().setProperty("textAlign", "center");
138        }
139
140        boolean collapsible = child instanceof MaterialCollapsible;
141        if(collapsible) {
142            // Since the collapsible is separ
143            ((MaterialCollapsible)child).addClearActiveHandler(new ClearActiveHandler() {
144                @Override
145                public void onClearActive(ClearActiveEvent event) {
146                    clearActive();
147                }
148            });
149        }
150
151        if(!(child instanceof ListItem)) {
152            // Direct list item not collapsible
153            final ListItem listItem = new ListItem();
154            if(child instanceof MaterialCollapsible) {
155                listItem.getElement().getStyle().setBackgroundColor("transparent");
156            }
157            if(child instanceof HasWaves) {
158                listItem.setWaves(((HasWaves) child).getWaves());
159                ((HasWaves) child).setWaves(null);
160            }
161            listItem.add(child);
162
163            child = listItem;
164        }
165
166        // Collapsible's should not be selectable
167        final Widget finalChild = child;
168        if(!collapsible) {
169            // Active click handler
170            finalChild.addDomHandler(new ClickHandler() {
171                @Override
172                public void onClick(ClickEvent event) {
173                    clearActive();
174                    finalChild.addStyleName("active");
175                }
176            }, ClickEvent.getType());
177        }
178        child.getElement().getStyle().setDisplay(Style.Display.BLOCK);
179        super.add(child);
180    }

224        return fixedMixin.isOn();
225    }

232        fixedMixin.setOn(fixed);

242    @Override
243    public SideNavType getType() {
244        return typeMixin.getType();
245    }
246
247    private void processType(SideNavType type) {
248        if(activator != null && type != null) {
249            switch (type) {
250                case MINI:
251                    setWidth(64);
252                    break;
253            }
254        }
255    }
256
257    @Override
258    public void clearActive() {
259        clearActive(this);
260
261        ClearActiveEvent.fire(this);
262    }
263
264    private void clearActive(HasWidgets widget) {
265        for(Widget child : widget) {
266            Element element = child.getElement();
267            if(StyleHelper.containsStyle(element.getClassName(), "active")) {
268                element.removeClassName("active");
269            }
270
271            if(child instanceof HasWidgets) {
272                clearActive((HasWidgets)child);
273            }
274        }
275    }
276
277    /**
278     * reinitialize the side nav configurations when changing
279     * properties.
280     */
281    public void reinitialize() {
282        activator = null;
283        initialize(false);
284    }
285
286    protected void initialize() {
287        initialize(true);
288    }
289
290    protected void initialize(boolean strict) {
291        if(activator == null) {
292            activator = DOMHelper.getElementByAttribute("data-activates", getId());
293            if (activator != null && activator.getClassName().contains("button-collapse")) {
294                SideNavType type = getType();
295                processType(type);
296
297                if(observedHandler != null) {
298                    observedHandler.removeHandler();
299                }
300                observedHandler = observer.addObservedHandler(new ObservedEvent.ObservedHandler() {
301                    @Override
302                    public void onObserved(ObservedEvent event) {
303                        if(event.getValue().equals("0px")) {
304                            SideNavShownEvent.fire(MaterialSideNav.this);
305                        } else if(event.getValue().equals("-"+getWidth()+"px")) {
306                            SideNavHiddenEvent.fire(MaterialSideNav.this);
307                        }
308                    }
309                });
310                observer.observe(getElement());
311
312                initialize(activator, width, closeOnClick, edge.getCssName());
313
314                if(!isFixed()) {
315                    String style = activator.getAttribute("style");
316                    activator.setAttribute("style", style + "; display: block !important");
317                }
318            } else if(strict) {
319                throw new RuntimeException("Cannot find an activator for the MaterialSideNav, " +
320                    "please ensure you have a MaterialNavBar with an activator setup to match " +
321                    "this widgets id.");
322            }
323        }
324    }
