053/**
054 * SideNav is a material component that gives you a lists of menus and other navigation components.
055 *
056 * <h3>UiBinder Usage:</h3>
057 * <pre>
058 * {@code
059 * <m:MaterialSideNav ui:field="sideNav" width="280" m:id="mysidebar"  type="OPEN" closeOnClick="false">
060 *     <m:MaterialLink href="#about" iconPosition="LEFT" iconType="OUTLINE" text="About" textColor="blue"  />
061 *     <m:MaterialLink href="#gettingStarted" iconPosition="LEFT" iconType="DOWNLOAD" text="Getting Started" textColor="blue"  >
062 * </m:MaterialSideNav>
063 * }
064 * </pre>
065 *
066 * @author kevzlou7979
067 * @author Ben Dol
068 * @see <a href="http://gwt-material-demo.herokuapp.com/#sidenav">Material SideNav</a>
069 */

157    @Override
158    public void add(Widget child) {
159        if(child instanceof MaterialImage) {
160            child.getElement().getStyle().setProperty("border", "1px solid #e9e9e9");
161            child.getElement().getStyle().setProperty("textAlign", "center");
162        }
163
164        boolean collapsible = child instanceof MaterialCollapsible;
165        if(collapsible) {
166            // Since the collapsible is separ
167            ((MaterialCollapsible)child).addClearActiveHandler(new ClearActiveHandler() {
168                @Override
169                public void onClearActive(ClearActiveEvent event) {
170                    clearActive();
171                }
172            });
173        }
174
175        if(!(child instanceof ListItem)) {
176            // Direct list item not collapsible
177            final ListItem listItem = new ListItem();
178            if(child instanceof MaterialCollapsible) {
179                listItem.getElement().getStyle().setBackgroundColor("transparent");
180            }
181            if(child instanceof HasWaves) {
182                listItem.setWaves(((HasWaves) child).getWaves());
183                ((HasWaves) child).setWaves(null);
184            }
185            listItem.add(child);
186
187            child = listItem;
188        }
189
190        // Collapsible's should not be selectable
191        final Widget finalChild = child;
192        if(!collapsible) {
193            // Active click handler
194            finalChild.addDomHandler(new ClickHandler() {
195                @Override
196                public void onClick(ClickEvent event) {
197                    clearActive();
198                    finalChild.addStyleName("active");
199                }
200            }, ClickEvent.getType());
201        }
202        child.getElement().getStyle().setDisplay(Style.Display.BLOCK);
203        super.add(child);
204    }

271    private void processType(SideNavType type) {
272        if(activator != null && type != null) {
273            addStyleName(type.getCssName());
274            switch (type) {
275                case MINI:
276                    setWidth(64);
277                    break;
278                case CARD:
279                    new Timer() {
280                        @Override
281                        public void run() {
282                            if(isSmall()) { show(); }
283                        }}.schedule(500);
284                    break;
285                case PUSH:
286                    applyPushType(getElement(), activator, width);
287                    break;
288            }
289        }
290    }

361    protected void initialize(boolean strict) {
362        if(activator == null) {
363            activator = DOMHelper.getElementByAttribute("data-activates", getId());
364            if (activator != null) {
365                SideNavType type = getType();
366                processType(type);
367
368                initialize(activator, width, closeOnClick, edge.getCssName());
369
370                if(alwaysShowActivator || !isFixed()) {
371                    String style = activator.getAttribute("style");
372                    activator.setAttribute("style", style + "; display: block !important");
373                    activator.removeClassName("navmenu-permanent");
374                }
375            } else if(strict) {
376                throw new RuntimeException("Cannot find an activator for the MaterialSideNav, " +
377                        "please ensure you have a MaterialNavBar with an activator setup to match " +
378                        "this widgets id.");
379            }
380        }
381    }

413    protected void onClosing() {
414        open = false;
415        if(getType().equals(SideNavType.PUSH)) {
416            pushElements(false, width);
417        }
418
419        SideNavClosingEvent.fire(this);
420    }
421
422    protected void onClosed() {
423        SideNavClosedEvent.fire(this);
424    }
425
426    protected void onOpening() {
427        open = true;
428        if(getType().equals(SideNavType.PUSH)) {
429            pushElements(true, width);
430        }
431
432        SideNavOpeningEvent.fire(this);
433    }
434
435    protected void onOpened() {
436        if(allowBodyScroll) {
437            RootPanel.getBodyElement().getStyle().clearOverflow();
438        }
439
440        SideNavOpenedEvent.fire(this);
441    }
