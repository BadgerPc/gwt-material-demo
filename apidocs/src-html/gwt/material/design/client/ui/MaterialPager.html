023import com.google.gwt.dom.client.Document;
024import com.google.gwt.event.dom.client.ClickEvent;
025import com.google.gwt.user.client.Event;
026import gwt.material.design.client.base.MaterialWidget;
027import gwt.material.design.client.constants.CssName;
028import gwt.material.design.client.constants.IconPosition;
029import gwt.material.design.client.constants.IconType;
030import gwt.material.design.client.constants.WavesType;
031import gwt.material.design.client.events.PageSelectionEvent;

034import static gwt.material.design.client.events.PageSelectionEvent.PageSelectionHandler;
035import static gwt.material.design.client.events.PageSelectionEvent.TYPE;

039/**
040 * Material Pager with page event
041 * <h3>UiBinder Usage:</h3>
042 * <pre>
043 * {@code <m:MaterialPager  ui:field='pager' />}
044 * </pre>
045 *
046 * @author Guaido79
047 */
048public class MaterialPager extends MaterialWidget {
049
050    private int total;
051    private int pageSize = 10;
052    private int currentPage = 1;
053    private int maxPageLinksShown = 10;
054    private boolean enableIndicator;
055    private String indicatorTemplate = "Page {page} of {total}";
056
057    private int calcTotalPages;
058    private int calcShowingPageFrom;
059    private int calcShowingPageTo;
060    private boolean calcInitialized;
061
062    private PagerListItem linkLeft;
063    private PagerListItem linkRight;
064
065    private MaterialChip indicator;
066
067    public MaterialPager() {
068        super(Document.get().createULElement(), CssName.PAGINATION);
069        setWaves(WavesType.DEFAULT);
070        removeStyleName(CssName.WAVES_EFFECT);
071    }
072
073    public MaterialPager(int total, int pageSize) {
074        this();
075        this.total = total;
076        this.pageSize = pageSize;
077    }
078
079    public MaterialPager(int total, int pageSize, int currentPage) {
080        this(total, pageSize);
081        this.currentPage = currentPage;
082    }
083
084    @Override
085    protected void onLoad() {
086        super.onLoad();
087        initialize();
088    }
089
090    private void initialize() {
091        if (!calcInitialized) {
092            calcTotalPages = total / pageSize + (((double) total % (double) pageSize) > 0 ? 1 : 0);
093
094            add(getOrCreateLiElementLeft());
095            moveNextPagesRange();
096            add(getOrCreateLiElementRight());
097            if (enableIndicator) {
098                add(createLiElementIndicator());
099            }
100            onPageSelection(1);
101
102            calcInitialized = true;
103        }
104    }
105
106    protected void moveNextPagesRange() {
107        calcShowingPageFrom = currentPage;
108        calcShowingPageTo = Math.min(currentPage + maxPageLinksShown - 1, calcTotalPages);
109        createPageNumberLinks();
110    }
111
112    protected void movePreviousPagesRange() {
113        calcShowingPageFrom = currentPage - maxPageLinksShown + 1;
114        calcShowingPageTo = currentPage;
115        createPageNumberLinks();
116    }
117
118    protected void createPageNumberLinks() {
119        for (int i = 0; i < getWidgetCount(); i++) {
120            final PagerListItem widget = (PagerListItem) getWidget(i);
121            if (!widget.isFixed()) {
122                Scheduler.get().scheduleDeferred(widget::removeFromParent);
123            }
124        }
125        int insertionIndex = 1;
126        for (int i = calcShowingPageFrom; i <= calcShowingPageTo; i++) {
127            final PagerListItem liElementForPage = createLiElementForPage(i);
128
129            Scheduler.get().scheduleDeferred(new InsertElementAtPositionCommand(insertionIndex++) {
130                @Override
131                public void execute() {
132                    insert(liElementForPage, insertionIndex);
133                }
134            });
135        }
136    }
137
138    protected PagerListItem createLiElementForPage(final int page) {
139        final PagerListItem pageLiElement = new PagerListItem();
140        pageLiElement.setFixed(false);
141        pageLiElement.add(createLinkPage(page));
142
143        addPageSelectionHandler(event -> {
144            pageLiElement.setActive(event.getPageTo() == page);
145        });
146
147        pageLiElement.addHandler(event -> {
148            onPageSelection(page);
149            event.preventDefault();
150            event.stopPropagation();
151        }, ClickEvent.getType());
152
153        return pageLiElement;
154    }
155
156    protected PagerListItem getOrCreateLiElementLeft() {
157        linkLeft = new PagerListItem();
158        linkLeft.setFixed(true);
159        linkLeft.addHandler(event -> {
160            if (linkLeft.isEnabled())
161                onPageSelection(currentPage - 1);
162            event.preventDefault();
163            event.stopPropagation();
164        }, ClickEvent.getType());
165        linkLeft.add(createLinkLeft());
166
167        addPageSelectionHandler(event -> {
168            MaterialPager.this.linkLeft.setEnabled(event.getPageTo() > 1);
169        });
170
171        return linkLeft;
172    }
173
174    protected PagerListItem getOrCreateLiElementRight() {
175        linkRight = new PagerListItem();
176        linkRight.setFixed(true);
177        linkRight.addHandler(event -> {
178            if (linkRight.isEnabled())
179                onPageSelection(currentPage + 1);
180            event.stopPropagation();
181            event.preventDefault();
182        }, ClickEvent.getType());
183        linkRight.add(createLinkRight());
184
185        addPageSelectionHandler(event -> {
186            MaterialPager.this.linkRight.setEnabled(event.getPageTo() < calcTotalPages);
187        });
188
189        return linkRight;
190    }
191
192    protected PagerListItem createLiElementIndicator() {
193        PagerListItem indicatorLi = new PagerListItem(false);
194        indicatorLi.setFixed(true);
195        indicatorLi.add(getOrCreateIndicator());
196        return indicatorLi;
197    }
198
199    protected MaterialChip getOrCreateIndicator() {
200        indicator = new MaterialChip();
201        indicator.getElement().getStyle().setBackgroundColor("inherit");
202        addPageSelectionHandler(event -> {
203            indicator.setText(indicatorTemplate
204                    .replaceAll("\\{page\\}", String.valueOf(event.getPageTo()))
205                    .replaceAll("\\{total\\}", String.valueOf(event.getTotalPage()))
206            );
207        });
208        return indicator;
209    }
210
211    protected MaterialLink createLinkPage(int page) {
212        return new MaterialLink(String.valueOf(page));
213    }
214
215    protected MaterialLink createLinkLeft() {
216        final MaterialLink linkLeft = new MaterialLink(IconType.CHEVRON_LEFT);
217        linkLeft.setIconPosition(IconPosition.NONE);
218        return linkLeft;
219    }
220
221    protected MaterialLink createLinkRight() {
222        final MaterialLink linkRight = new MaterialLink(IconType.CHEVRON_RIGHT);
223        linkRight.setIconPosition(IconPosition.NONE);
224        return linkRight;
225    }
226
227    protected void onPageSelection(int page) {
228        this.currentPage = page;
229
230        if (this.currentPage > calcShowingPageTo) {
231            moveNextPagesRange();
232        }
233        if (this.currentPage < calcShowingPageFrom) {
234            movePreviousPagesRange();
235        }
236
237        PageSelectionEvent event = new PageSelectionEvent();
238        event.setPageFrom(this.currentPage);
239        event.setPageTo(page);
240        event.setTotalPage(this.calcTotalPages);
241
242        fireEvent(event);
243    }
244
245    public void addPageSelectionHandler(PageSelectionHandler handler) {
246        this.addHandler(handler, TYPE);
247    }
248
249    public boolean isEnableIndicator() {
250        return enableIndicator;
251    }
252
253    public void setEnableIndicator(boolean enableIndicator) {
254        this.enableIndicator = enableIndicator;
255    }
256
257    public int getTotal() {
258        return total;
259    }
260
261    public void setTotal(final int total) {
262        boolean needToClear = total != this.total;
263        this.total = total;
264        currentPage = 1;
265        if (calcInitialized && needToClear) {
266            clear();
267            initialize();
268        }
269    }
270
271    public int getPageSize() {
272        return pageSize;
273    }
274
275    public void setPageSize(int pageSize) {
276        this.pageSize = pageSize;
277    }
278
279    public int getCurrentPage() {
280        return currentPage;
281    }
282
283    public void setCurrentPage(int currentPage) {
284        this.currentPage = currentPage;
285    }
286
287    public int getMaxPageLinksShown() {
288        return maxPageLinksShown;
289    }
290
291    public void setMaxPageLinksShown(int maxPageLinksShown) {
292        this.maxPageLinksShown = maxPageLinksShown;
293    }
294
295    public String getIndicatorTemplate() {
296        return indicatorTemplate;
297    }
298
299    /**
300     * Set the paging indicator label with a custom template
301     * <ul>
302     * <li><strong>{page}</strong> is the current page</li>
303     * <li><strong>{total}</strong> is the total page</li>
304     * </ul>
305     * Example
306     * <pre>
307     * {@code
308     * Page {page} of {total}
309     * }</pre>
310     */
311    public void setIndicatorTemplate(String indicatorTemplate) {
312        this.indicatorTemplate = indicatorTemplate;
313    }
314
315    static abstract class InsertElementAtPositionCommand implements Scheduler.ScheduledCommand {
316        protected int insertionIndex;
317
318        InsertElementAtPositionCommand(int insertionIndex) {
319            this.insertionIndex = insertionIndex;
320        }
321    }
322
323    static class PagerListItem extends ListItem {
324
325        private boolean fixed;
326        private boolean enabled;
327
328        public PagerListItem() {
329            this(true);
330        }
331
332        public PagerListItem(boolean clickable) {
333            if (clickable) {
334                addStyleName(CssName.WAVES_EFFECT);
335                sinkEvents(Event.ONCLICK | Event.TOUCHEVENTS);
336            }
337        }
338
339        public boolean isFixed() {
340            return fixed;
341        }
342
343        public void setFixed(boolean fixed) {
344            this.fixed = fixed;
345        }
346
347        public void setActive(boolean active) {
348            if (active) {
349                addStyleName(CssName.ACTIVE);
350            } else {
351                removeStyleName(CssName.ACTIVE);
352            }
353        }
354
355        @Override
356        public boolean isEnabled() {
357            return enabled;
358        }
359
360        @Override
361        public void setEnabled(boolean enabled) {
362            this.enabled = enabled;
363            if (!enabled) {
364                addStyleName(CssName.DISABLED);
365            } else {
366                removeStyleName(CssName.DISABLED);
367            }
368        }
369    }
