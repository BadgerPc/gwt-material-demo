020package gwt.material.design.client.ui;
021
022import com.google.gwt.event.logical.shared.*;
023import com.google.gwt.event.shared.HandlerRegistration;
024import com.google.gwt.user.client.ui.Widget;
025import gwt.material.design.client.base.HasPosition;
026import gwt.material.design.client.base.mixin.CssNameMixin;
027import gwt.material.design.client.constants.HideOn;
028import gwt.material.design.client.constants.Position;
029import gwt.material.design.client.ui.html.ListItem;
030import gwt.material.design.client.ui.html.UnorderedList;
031
032import java.util.ArrayList;
033import java.util.List;
034
035//@formatter:off
036
037/**
038 * <p>Material NavSection is a child of MaterialNavBar that will contain toolbar items such as link, icon and other components
039 * <h3>UiBinder Usage:</h3>
040 * <p>
041 * <pre>
042 * {@code
043 * <m:MaterialNavSection align="RIGHT">
044 *     <m:MaterialLink  iconType="ACCOUNT_CIRCLE" iconPosition="LEFT" text="Account"  textColor="WHITE" waves="LIGHT"/>
045 *     <m:MaterialLink  iconType="AUTORENEW" iconPosition="LEFT" text="Refresh" textColor="WHITE" waves="LIGHT"/>
046 *     <m:MaterialLink  iconType="SEARCH" tooltip="Menu" textColor="WHITE" waves="LIGHT"/>
047 *     <m:MaterialLink  iconType="MORE_VERT" tooltip="Starter" textColor="WHITE" waves="LIGHT"/>
048 * </m:MaterialNavSection>
049 * }
050 * </pre>
051 * </p>
052 *
053 * @author kevzlou7979
054 * @author Ben Dol
055 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#navbar">Material NavSection</a>
056 * @see <a href="https://material.io/guidelines/components/toolbars.html#">Material Design Specification</a>
057 */
058//@formatter:on
059public class MaterialNavSection extends UnorderedList implements HasPosition, HasSelectionHandlers<Integer> {
060
061    private final CssNameMixin<MaterialNavSection, Position> posMixin = new CssNameMixin<>(this);
062    private List<HandlerRegistration> handlers = new ArrayList<>();
063
064    /**
065     * Container for App Toolbar and App Sidebar, contains Material
066     * Links, Icons or any other material components.
067     */
068    public MaterialNavSection() {
069        super();
070        setHideOn(HideOn.HIDE_ON_MED_DOWN);
071    }
072
073    @Override
074    public void add(Widget child) {
075        super.add(new ListItem(child));
076    }
077
078    @Override
079    protected void onLoad() {
080        super.onLoad();
081        build();
082    }
083
084    @Override
085    protected void build() {
086        for (Widget widget : getChildren()) {
087            if (widget instanceof ListItem) {
088                handlers.add(((ListItem) widget).addClickHandler(clickEvent -> SelectionEvent.fire(this, getWidgetIndex(widget))));
089            }
090        }
091    }
092
093    @Override
094    protected void onUnload() {
095        super.onUnload();
096
097        for (HandlerRegistration hr : handlers) {
098            hr.removeHandler();
099        }
100    }
101
102    @Override
103    public Position getPosition() {
104        return posMixin.getCssName();
105    }
106
107    @Override
108    public void setPosition(Position position) {
109        posMixin.setCssName(position);
110    }
111
112    @Override
113    public HandlerRegistration addSelectionHandler(SelectionHandler<Integer> selectionHandler) {
114        return addHandler(selectionHandler, SelectionEvent.getType());
115    }
116}
